set(MADMISC_HEADERS misc.h ran.h phandler.h interpolation_1d.h cfft.h info.h gnuplot.h array_of_bools.h kahan_accumulator.h)
set(MADMISC_SOURCES
    checksum_file.cc position_stream.cc gprofexit.cc ran.cc cfft.cc info.cc numerics.cc unique_filename.cc)
# retrieve git metadata
include(GetGitMetadata)
vgkit_cmake_git_metadata()
# feed MADNESS_GIT_{REVISION,DESCRIPTION} to info.cc only to avoid recompiling everything
set_source_files_properties(
        info.cc
        PROPERTIES
          COMPILE_DEFINITIONS
          "MADNESS_GIT_REVISION=\"${MADNESS_GIT_REVISION}\";MADNESS_GIT_DESCRIPTION=\"${MADNESS_GIT_DESCRIPTION}\""
)

# Generate build_timestamp.cc (a real source file, not a header) so that CMake
# manages the dependency with a consistent path in the Makefile.  The previous
# header-based approach broke because the compiler wrote an absolute path into
# the .d file while the Makefile rule used a relative path; make compares these
# as strings and never connected them, causing a two-build lag.
#
# With a generated .cc file that is listed in MADMISC_SOURCES, CMake inserts a
# proper explicit dependency (same path form on both sides) so the chain
#   source-file-change → rebuild build_timestamp.cc → recompile build_timestamp.cc.o
# all happens in one make pass.
set(BUILD_TIME_CC ${CMAKE_CURRENT_BINARY_DIR}/build_timestamp.cc)
file(WRITE ${CMAKE_CURRENT_BINARY_DIR}/gen_build_timestamp.cmake [=[
string(TIMESTAMP _time "%H:%M:%S")
string(TIMESTAMP _date "%b %d %Y")
file(WRITE "${OUTPUT}"
    "// Generated by CMake — do not edit\n"
    "namespace madness { namespace info {\n"
    "const char* build_time() { return \"${_time}\"; }\n"
    "const char* build_date() { return \"${_date}\"; }\n"
    "} }\n"
)
]=])
file(GLOB_RECURSE _all_sources
    ${CMAKE_SOURCE_DIR}/src/madness/*.cc
    ${CMAKE_SOURCE_DIR}/src/madness/*.h
    ${CMAKE_SOURCE_DIR}/src/madness/*.hpp
    ${CMAKE_SOURCE_DIR}/src/apps/*.cc
    ${CMAKE_SOURCE_DIR}/src/apps/*.cpp
    ${CMAKE_SOURCE_DIR}/src/apps/*.h
    ${CMAKE_SOURCE_DIR}/src/apps/*.hpp
)
add_custom_command(
    OUTPUT ${BUILD_TIME_CC}
    COMMAND ${CMAKE_COMMAND}
        -DOUTPUT=${BUILD_TIME_CC}
        -P ${CMAKE_CURRENT_BINARY_DIR}/gen_build_timestamp.cmake
    DEPENDS ${_all_sources}
    COMMENT "Generating build timestamp"
)

# Add generated source to the library sources
list(APPEND MADMISC_SOURCES ${BUILD_TIME_CC})

# Create the MADmisc library
add_mad_library(misc MADMISC_SOURCES MADMISC_HEADERS "world" "madness/misc/")
