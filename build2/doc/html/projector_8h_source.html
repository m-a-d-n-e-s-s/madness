17 
18  /// use this class to project a function or a set of functions on
19  /// another space of function. The projector can handle different sets of
20  /// functions for the bra and the ket space, e.g. in case of regularized
21  /// orbitals: |f> <-> <f|R^2

60  /// project f on p:
61 
62  /// \f[
63  /// | result > = \sum_p | p > <p|f>
64  /// \f]
65  /// @param[in] f the function to be projected
66  /// @return the projection of f on the space of p
67  funcT operator()(const funcT& f) const {
68  return this->operator()(vecfuncT(1,f)).front();
69  }
70 
71  /// project f on p:
72 
73  /// \f[
74  /// | result > = \sum_p | p > <p|f>
75  /// \f]
76  /// @param[in] f the vector of functions to be projected
77  /// @return the projection of f on the space of p
78  vecfuncT operator()(const vecfuncT& f) const {
79  if (f.size()==0) return vecfuncT();
80  World& world=f[0].world();
81  Tensor<T> ovlp=matrix_inner(world,mo_bra_,f);
82  vecfuncT result=transform(world,mo_ket_,ovlp,true);
83  truncate(world,result);
84  return result;
85  }
86 
87  /// apply 3D Projector to one particle of a 6D function
88  /// \f[
89  /// |result> = \sum_p |p(particle)> <p(particle)|f(1,2)>_{particle}
90  /// \f]
91  /// @param[in] f the 6D function to be projected
92  /// @param[in] the particle that is projected (1 or 2)
93  /// @return the projected function
94  real_function_6d operator()(const real_function_6d&f, const size_t particle)const{
95  real_function_6d result = real_factory_6d(f.world());
96  MADNESS_ASSERT(particle==1 or particle==2);
97  for(size_t i=0;i<mo_ket_.size();i++){
98  real_function_3d tmp1 = mo_ket_[i];
99  real_function_3d tmp2 = f.project_out(mo_bra_[i],particle-1);
100  real_function_6d tmp12;
101  if(particle==1){
102  tmp12 = CompositeFactory<double,6,3>(f.world()).particle1(copy(tmp1)).particle2(copy(tmp2));
103  tmp12.fill_tree();
104  }else{
105  tmp12 = CompositeFactory<double,6,3>(f.world()).particle1(copy(tmp2)).particle2(copy(tmp1));
106  tmp12.fill_tree();
107  }
108  result += tmp12;
109  }
110  return result;
111  }

218  /// apply the strong orthogonality operator Q12 on a function f
219 
220  /// notation of the equations follows
221  /// J. Chem. Phys., vol. 139, no. 11, p. 114106, 2013.
222  Function<T,2*NDIM> operator()(const Function<T,2*NDIM>& f) const {
223 
224  // simple and it works for higher accuracies, but might be
225  // imprecise for lower accuracies
226 // return (f-O1(f)-O2(f)+O1(O2(f))).truncate().reduce_rank();
227 
228  const double thresh=FunctionDefaults<2*NDIM>::get_thresh();
229  const double tight_thresh=FunctionDefaults<2*NDIM>::get_thresh()*0.1;
230 
231  // Eq. (A9): g_kl = < k(1) l(2) | f(1,2) >
232  // note no (kl) symmetry here!
233  Tensor<double> g_kl(bra1_.size(),bra2_.size());
234  for (size_t k=0; k<bra1_.size(); ++k) {
235  for (size_t l=0; l<bra2_.size(); ++l) {
236  Function<T,2*NDIM> kl=CompositeFactory<T,2*NDIM,NDIM>(world)
237  .particle1(copy(bra1_[k])).particle2(copy(bra2_[l]));
238  g_kl(k,l)=inner(f,kl);
239  }
240  }
241 // if (world.rank()==0) {print(g_kl);};
242 
243  // Eq. (A12)
244  // project out the mainly first particle: O1 (1 - 1/2 O2)
245  Function<T,2*NDIM> r1=FunctionFactory<T,2*NDIM>(world).thresh(tight_thresh);
246  for (size_t k=0; k<bra1_.size(); ++k) {
247 
248  // Eq. (A10)
249  Function<T,NDIM> h2=f.project_out(bra1_[k],0);
250 
251  // Eq. (A12)
252  for (size_t l=0; l<ket2_.size(); ++l) {
253  h2-=0.5*g_kl(k,l)*ket2_[l];
254  }
255 
256  // Eq. (A7), second term rhs
257  // the hartree product tends to be inaccurate; tighten threshold
258  FunctionDefaults<2*NDIM>::set_thresh(tight_thresh);
259  r1=(r1+hartree_product(ket1_[k],h2));
260  FunctionDefaults<2*NDIM>::set_thresh(thresh);
261  r1.set_thresh(thresh);
262  r1.print_size("r1"+stringify(k));
263  }
264 
265  // project out the mainly second particle: O2 (1 - 1/2 O1)
266  Function<T,2*NDIM> r2=FunctionFactory<T,2*NDIM>(world).thresh(tight_thresh);
267  for (size_t l=0; l<ket2_.size(); ++l) {
268 
269  // Eq. (A11)
270  Function<T,NDIM> h1=f.project_out(bra2_[l],1);
271 
272  // Eq. (A13)
273  for (size_t k=0; k<ket1_.size(); ++k) {
274  h1-=0.5*g_kl(k,l)*ket1_[k]; // ordering g(k,l) is correct
275  }
276 
277  // Eq. (A7), third term rhs
278  // the hartree product tends to be inaccurate; tighten threshold
279  FunctionDefaults<2*NDIM>::set_thresh(tight_thresh);
280  r2=(r2+hartree_product(h1,ket2_[l]));
281  r2.set_thresh(thresh);
282  FunctionDefaults<2*NDIM>::set_thresh(thresh);
283  r2.print_size("r2"+stringify(l));
284  }
285  FunctionDefaults<2*NDIM>::set_thresh(tight_thresh);
286  Function<T,2*NDIM> result=(f-r1-r2).truncate().reduce_rank();
287  FunctionDefaults<2*NDIM>::set_thresh(thresh);
