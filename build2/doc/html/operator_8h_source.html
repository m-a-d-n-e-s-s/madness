593  /// Computes the operator norm of one of the separated terms of the modified NS form
594  /// ... WITHOUT FACTOR INCLUDED
595  /// compute for 1 term, all dim, 1 disp, essentially for SeparatedConvolutionInternal
596  double munorm2_modified(Level n, const ConvolutionData1D<Q>* ops_1d[]) const {
597  PROFILE_MEMBER_FUNC(SeparatedConvolution);
598 
599  // follows Eq. (21) ff of Beylkin 2008 (Beylkin Appl. Comput. Harmon. Anal. 24, pp 354)
600 
601  // we have all combinations of difference, upsampled, F terms (d, u, f),
602  // with the constraint that d is in each term exactly once. In the mixed terms (udf)
603  // we just get all possible combinations, in the pure terms (dff, duu) we have
604  // to multiply each term (dff, fdf, ffd) with (NDIM-1)!, to get the right number.
605 
606  double dff = 0.0;
607  double duu = 0.0;
608  double udf = 0.0;
609 
610  // loop over d shifting over the dimensions dxx, xdx, xxd,
611  for (size_t d=0; d<NDIM; ++d) {
612  double dff_tmp = ops_1d[d]->N_diff;
613  double duu_tmp = ops_1d[d]->N_diff;
614  double udf_tmp = ops_1d[d]->N_diff;
615 
616 
617  for (size_t dd=0; dd<NDIM; ++dd) {
618  if (dd!=d) {
619  dff_tmp *= ops_1d[dd]->N_F;
620  duu_tmp *= ops_1d[dd]->N_up;
621 
622  udf_tmp *= ops_1d[dd]->N_F;
623  for (size_t ddd=0; ddd<NDIM; ++ddd) {
624  if (ddd!=dd) udf += udf_tmp * ops_1d[ddd]->N_up;
625  }
626  }
627  }
628 
629  dff+=dff_tmp;
630  duu+=duu_tmp;
631  }
632 
633  // finalize with the factorial
634  double factorial=1.0;
635  for (int i=1; i<static_cast<int>(NDIM)-1; ++i) factorial*=double(i);
636  dff*=factorial;
637  duu*=factorial;
638 
639  // Eq. (23) of Beylkin 2008, for one separated term WITHOUT the factor
640  double norm=(dff + udf + duu) /(factorial * double(NDIM));
641 
642 // // double check
643 // if (NDIM==3) {
644 // Tensor<Q> R_full=outer(ops_1d[0]->R,outer(ops_1d[1]->R,ops_1d[2]->R));
645 // Tensor<Q> T_full=outer(ops_1d[0]->T,outer(ops_1d[1]->T,ops_1d[2]->T));
646 // double n2=(R_full-T_full).normf();
647 //// print("norm estimate, norm",norm, n2, norm<n2);
648 // norm=n2;
649 // }

803 
804  /// @param[in] coeff the coeffs of dim 2*NDIM that will be upsampled
805  /// @param[in] key the key indicating the child -- only some dimensions will be "reproductive"
806  /// @param[in] particle if 0: upsample dimensions 0-2
807  /// if 1: upsample dimensions 3-5
808  /// @return a partially upsampled coefficient tensor
809  template<typename T, size_t FDIM>
810  GenTensor<T> partial_upsample(const Key<FDIM>& key, const GenTensor<T>& coeff, const int particle) const {
811 
812  if (coeff.rank()==0) return GenTensor<T>();
813  MADNESS_ASSERT(coeff.dim(0)==k);
814  if (NDIM==coeff.ndim()) {
815  MADNESS_ASSERT(particle==1); // other particle, leave this particle unchanged
816  return coeff;
817  }
818 
819  MADNESS_ASSERT(coeff.ndim()==FDIM);
820  MADNESS_ASSERT(particle==0 or (2*NDIM==FDIM));
821 
822  // the twoscale coefficients: for upsampling use h0/h1; see Alpert Eq (3.35a/b)
823  // handle the spectator dimensions with the identity matrix
824  const Tensor<T> h[2] = {cdata.h0, cdata.h1};
825  Tensor<T> identity(k,k);
826  for (int i=0; i<k; ++i) identity(i,i)=1.0;
827  Tensor<T> matrices[2*NDIM];
828 
829  // get the appropriate twoscale coefficients for each dimension
830  if (particle==0) {
831  for (size_t ii=0; ii<NDIM; ++ii) matrices[ii]=h[key.translation()[ii]%2];
832  for (size_t ii=0; ii<NDIM; ++ii) matrices[ii+NDIM]=identity;
833  } else if (particle==1) {
834  for (size_t ii=0; ii<NDIM; ++ii) matrices[ii]=identity;
835  for (size_t ii=0; ii<NDIM; ++ii) matrices[ii+NDIM]=h[key.translation()[ii+NDIM]%2];
836  } else {
837  MADNESS_EXCEPTION("unknown particle",1);
838  }
839 
840  // transform and accumulate on the result
841  const GenTensor<T> result=general_transform(coeff,matrices);
842  return result;
843  }

847 
848  /// specialization of the unfilter method, will transform only the sum coefficients
849  /// @param[in] key key of level n+1
850  /// @param[in] coeff sum coefficients of level n (does NOT belong to key!!)
851  /// @return sum coefficients on level n+1
852  template<typename T, size_t FDIM>
853  GenTensor<T> upsample(const Key<FDIM>& key, const GenTensor<T>& coeff) const {
