142  /// This exception is thrown whenever an MPI error occurs.
143  class Exception : public std::exception {
144  private:
145  char mpi_error_string_[MPI_MAX_ERROR_STRING];
146  std::string mpi_statuses_error_string_;
147  public:
148 
149  Exception(const int mpi_error) throw() {
150  int len = 0;
151  if(MPI_Error_string(mpi_error, mpi_error_string_, &len) != MPI_SUCCESS)
152  std::strncpy(mpi_error_string_, "UNKNOWN MPI ERROR!", MPI_MAX_ERROR_STRING);
153  }
154 
155  Exception(const int mpi_error, const int nstatuses,
156  const int* indices,
157  MPI_Status* const statuses) noexcept {
158  try {
159  if (mpi_error == MPI_ERR_IN_STATUS) {
160  std::ostringstream oss;
161  for(auto s=0; s!=nstatuses; ++s) {
162  int len = 0;
163  auto status_error = statuses[s].MPI_ERROR;
164  if (status_error != MPI_SUCCESS) {
165  oss << "request " << indices[s] << ":";
166  if (MPI_Error_string(status_error, mpi_error_string_, &len) != MPI_SUCCESS)
167  oss << " unknown error!" << std::endl;
168  else
169  oss << mpi_error_string_ << std::endl;
170  }
171  }
172  mpi_statuses_error_string_ = oss.str();
173  }
174  }
175  catch (...) {}
176 
177  int len = 0;
178  if(MPI_Error_string(mpi_error, mpi_error_string_, &len) != MPI_SUCCESS)
179  std::strncpy(mpi_error_string_, "UNKNOWN MPI ERROR!", MPI_MAX_ERROR_STRING);
180  }
181 
182  Exception(const Exception& other) throw() {
183  std::strncpy(mpi_error_string_, other.mpi_error_string_, MPI_MAX_ERROR_STRING);
184  try {
185  mpi_statuses_error_string_ = other.mpi_statuses_error_string_;
186  } catch(...) { mpi_statuses_error_string_.clear(); }
187  }
188 
189  Exception& operator=(const Exception& other) {
190  std::strncpy(mpi_error_string_, other.mpi_error_string_, MPI_MAX_ERROR_STRING);
191  try {
192  mpi_statuses_error_string_ = other.mpi_statuses_error_string_;
193  } catch(...) { mpi_statuses_error_string_.clear(); }
194  return *this;
195  }
196 
197  virtual ~Exception() throw() { }
198 
199  virtual const char* what() const throw() { return mpi_error_string_; }
200  bool can_elaborate() const noexcept {
201  return !mpi_statuses_error_string_.empty();
202  }
203  const char* elaborate() const noexcept {
204  return mpi_statuses_error_string_.c_str();
205  }
206 
207  friend std::ostream& operator<<(std::ostream& os, const Exception& e) {
208  os << e.what();
209  if (e.can_elaborate()) {
210  os << e.elaborate();
211  }
212  return os;

500  Impl(const MPI_Comm& c, int m, int n, bool o) :

502  { MADNESS_ASSERT(comm != MPI_COMM_NULL); }

510  /// Returns a unique tag for temporary use (1023<tag<=4095)
511 
512  /// These tags are intended for one time use to avoid tag
513  /// collisions with other messages around the same time period.
514  /// It simply increments/wraps a counter and returns the next
515  /// legal value.
516  ///
517  /// So that send and receiver agree on the tag all processes
518  /// need to call this routine in the same sequence.
519  int unique_tag() {
520  // Cannot use MPI mutex for anything else!
521  // It will preprocess to nothing for MPI_THREAD_MULTIPLE!
522  madness::ScopedMutex<madness::SCALABLE_MUTEX_TYPE> obolus(SafeMPI::charon);
523  int result = utag++;
524  if (utag >= 4095) utag = 1024;
525  return result;
526  }
527 
528  /// Returns a unique tag reserved for long-term use (0<tag<1000)
529 
530  /// Get a tag from this routine for long-term/repeated use.
531  ///
532  /// Tags in [1000,1023] are statically assigned.
533  int unique_reserved_tag() {
534  // Cannot use MPI mutex for anything else!
535  // It will preprocess to nothing for MPI_THREAD_MULTIPLE!
536  madness::ScopedMutex<madness::SCALABLE_MUTEX_TYPE> obolus(SafeMPI::charon);
537  int result = urtag++;
538  if (result >= 1000) MADNESS_EXCEPTION( "too many reserved tags in use" , result );
539  return result;
540  }
