304  /// symmetric orthonormalization (see e.g. Szabo/Ostlund)
305  /// @param[in] the vector to orthonormalize
306  /// @param[in] overlap matrix
307  template <typename T, std::size_t NDIM>
308  std::vector<Function<T,NDIM> > orthonormalize_symmetric(
309  const std::vector<Function<T,NDIM> >& v,
310  const Tensor<T>& ovlp) {
311  if(v.empty()) return v;
312 
313  Tensor<T> U;
314  Tensor< typename Tensor<T>::scalar_type > s;
315  syev(ovlp,U,s);
316 
317  // transform s to s^{-1}
318  for(size_t i=0;i<v.size();++i) s(i)=1.0/(sqrt(s(i)));
319 
320  // save Ut before U gets modified with s^{-1}
321  const Tensor<T> Ut=transpose(U);
322  for(size_t i=0;i<v.size();++i){
323  for(size_t j=0;j<v.size();++j){
324  U(i,j)=U(i,j)*s(j);
325  }
326  }
327 
328  Tensor<T> X=inner(U,Ut,1,0);
329 
330  World& world=v.front().world();
331  return transform(world,v,X);
332 
333  }

348  /// canonical orthonormalization (see e.g. Szabo/Ostlund)
349  /// @param[in] the vector to orthonormalize
350  /// @param[in] overlap matrix
351  /// @param[in] lindep linear dependency threshold relative to largest eigenvalue
352  template <typename T, std::size_t NDIM>
353  std::vector<Function<T,NDIM> > orthonormalize_canonical(
354  const std::vector<Function<T,NDIM> >& v,
355  const Tensor<T>& ovlp,
356  double lindep) {
357 
358  if(v.empty()) return v;
359 
360  Tensor<T> U;
361  Tensor< typename Tensor<T>::scalar_type > s;
362  syev(ovlp,U,s);
363  lindep*=s(s.size()-1); // eigenvalues are in ascending order
364 
365  // transform s to s^{-1}
366  int rank=0,lo=0;
367  Tensor< typename Tensor<T>::scalar_type > sqrts(v.size());
368  for(size_t i=0;i<v.size();++i) {
369  if (s(i)>lindep) {
370  sqrts(i)=1.0/(sqrt(s(i)));
371  rank++;
372  } else {
373  sqrts(i)=0.0;
374  lo++;
375  }
376  }
377  MADNESS_ASSERT(size_t(lo+rank)==v.size());
378 
379  for(size_t i=0;i<v.size();++i){
380  for(size_t j=0;j<v.size();++j){
381  U(i,j)=U(i,j)*(sqrts(j));
382  }
383  }
384  Tensor<T> X=U(_,Slice(lo,-1));
385 
386  World& world=v.front().world();
387  return transform(world,v,X);
388 
389  }
