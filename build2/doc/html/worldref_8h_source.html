193  /// Remote reference counter
194 
195  /// Automatically counts local and remote references to an object. The
196  /// reference count is incremented when the object is copied locally or
197  /// serialized as part of communication.
198  class RemoteCounter {
199  private:
200  typedef RemoteCounterBase implT;
201  typedef ConcurrentHashMap<void*, WorldPtr<implT> > pimpl_mapT;
202 
203  static pimpl_mapT pimpl_map_; ///< A map of currently registered
204  ///< implementation objects. The key is
205  ///< it's referenced pointer.
206 
207  /// Pointer to the shared counter implementation object
208  mutable WorldPtr<implT> pimpl_;
209 
210  /// Clean-up the implementation object
211 
212  /// Here we check that the pimpl has been initialized, and if so, we
213  /// release the current reference. If the count drops to zero, then
214  /// this is the last reference to the pimpl and it should be deleted.
215  void destroy() {
216  if(pimpl_.is_local()) {
217  if(pimpl_->release()) {
218  // No one else is referencing this pointer.
219  // We can safely dispose of it.
220 
221 #ifdef MADNESS_REMOTE_REFERENCE_DEBUG
222  print(">>> RemoteCounter::unregister_ptr_: key=", pimpl_->key(), ", value=", pimpl_);
223 #endif
224  unregister_ptr_(pimpl_->key());
225  delete pimpl_.get();
226  }
227  }
228 
229  pimpl_ = WorldPtr<implT>();
230  }
231 
232  /// Register a local shared pointer
233 
234  /// This function will first search the local pointer register for
235  /// the shared pointer \c p. If found the pimpl for that pointer
236  /// will be returned. Otherwise a new pimpl will be created and
237  /// returned.
238  /// \tparam T The shared pointer type to register
239  /// \param w The world where the pointer lives
240  /// \param p The shared pointer to register
241  /// \return A world pointer to the pimpl
242  /// \throw std::bad_alloc If pimpl allocation fails.
243  /// \throw madness::MadnessException If pointer cannot be inserted
244  /// into the pointer registration map.
245  template <typename T>
246  static WorldPtr<implT> register_ptr_(World& w, const std::shared_ptr<T>& p) {
247  // Check for a null pointer
248  if(p.get() == nullptr)
249  return WorldPtr<implT>(w, nullptr);
250 
251  pimpl_mapT::accessor acc;
252  // Pointer is local and non-null
253  if(pimpl_map_.insert(acc,static_cast<void*>(p.get()))) {
254  // The pointer is not registered so we need to make a
255  // new pimpl.
256  implT* pimpl = new RemoteCounterImpl<T>(p);
257 
258  try{
259  acc->second = WorldPtr<implT>(w, pimpl);
260  } catch(...) {
261  delete pimpl;
262  throw;
263  }
264 
265 #ifdef MADNESS_REMOTE_REFERENCE_DEBUG
266  print(">>> RemoteCounter::register_ptr_(new): key=", p.get(), ", pimpl=", acc->second);
267 #endif
268  } else {
269  // The pointer is already registered, so we just need
270  // increment the counter.
271 #ifdef MADNESS_REMOTE_REFERENCE_DEBUG
272  print(">>> RemoteCounter::register_ptr_(existing): key=", acc->second->key(), ", pimpl=", acc->second);
273 #endif
274  acc->second->add_ref();
275  }
276 
277  return acc->second;
278  }

282  /// \param key The key of the \c RemoteReference object to be unregistered.
283  /// \throw MadnessException If \c key is not found in the pointer map.
284  static void unregister_ptr_(void* key) {
285  std::size_t erased = pimpl_map_.erase(key);
286  if (!erased) MADNESS_EXCEPTION("worldref: unregister_ptr failed", erased);

390  /// Simple structure used to manage references/pointers to remote instances
391 
392  /// This class was intended only for internal use and is still rather
393  /// poorly thought through, however, it seems to fill a wider need.
394  /// \note Do not serialize via wrap_opaque().
395  /// \note Ownership of a reference is transfered when serialized on a remote
396  /// node. You should not attempt to send a remote reference to more than one
397  /// node except from the owning node. If you do serialize more than once,
398  /// this will cause an invalid memory access on the owning node.
399  /// \note !!! It is YOUR RESPONSIBILITY to release the reference count. This
400  /// can be done by sending the remote reference back to the owner or by
401  /// calling reset(). If this is not done, you will have a memory leak.
402  template <typename T>
403  class RemoteReference {
404  public:
405  typedef typename detail::ptr_traits<T>::reference referenceT;
406  typedef T* pointerT;
407 
408  private:
409  mutable pointerT pointer_; ///< World pointer
410  detail::RemoteCounter counter_; ///< Remote reference counter
411 
412  // This is for RemoteReferences of other types, so they can still access
413  // private members.
414  template <typename>
415  friend class RemoteReference;
416 
417  // Handles reset of a remote reference from another node.
418  static void reset_handler(const AmArg& arg) {
419  RemoteReference<T> r;
420  arg & r;
421  // r resets on scope exit.
422  }

425

427  RemoteReference() :
428  pointer_(), counter_() {};
