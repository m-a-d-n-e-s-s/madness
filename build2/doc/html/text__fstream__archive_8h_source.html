71  /// Stores the cookie for runtime type-checking into the archive as a tag.
72 
73  /// \tparam T The type of data to be stored between the tags.
74  template <class T>
75  void store_start_tag() const {
76  char tag[256];
77  unsigned char cookie = archive_typeinfo<T>::cookie;
78  sprintf(tag,"<t%d>", cookie);
79  os << tag << std::endl;
80  MAD_ARCHIVE_DEBUG(std::cout << "textarchive: tag = " << tag << std::endl);
81  }
82 
83  /// Closes the "cookie" tag for runtime type-checking.
84 
85  /// \tparam T The type of data to be stored between the tags.
86  template <class T>
87  void store_end_tag() const {
88  char tag[256];
89  unsigned char cookie = archive_typeinfo<T>::cookie;
90  sprintf(tag,"</t%d>",cookie);
91  os << tag << std::endl;
92  }
93 
94  /// Store data to the filestream.
95 
96  /// The function only appears (due to \c enable_if) if \c T is
97  /// serializable.
98  /// \tparam T The type of data to be written.
99  /// \param[in] t Location of the data to be written.
100  /// \param[in] n The number of data items to be written.
101  template <class T>
102  typename std::enable_if< madness::is_iostreammable_v<T> >::type
103  store(const T* t, long n) const {
104  using madness::operator<<;
105  for (long i=0; i<n; ++i)
106  os << t[i] << std::endl;
107  }

113

118  void store(const unsigned char* t, long n) const {

167  /// Check the "cookie" tag in the archive for runtime type-checking.
168 
169  /// \tparam T The expected data type.
170  /// \throw MadnessException if the tag does not match that of the
171  /// expected type.
172  template <class T>
173  void check_start_tag(bool end=false) const {
174  char tag[256], ftag[256];
175  is.getline(ftag,256);
176  unsigned char cookie = archive_typeinfo<T>::cookie;
177  if (end)
178  sprintf(tag,"</t%d>",cookie);
179  else
180  sprintf(tag,"<t%d>",cookie);
181 
182  if (strcmp(tag,ftag) != 0) {
183  std::cout << "TextFstreamInputArchive: type mismatch: expected=" << tag
184  << " "
185  << archive_type_names[cookie]
186  << " "
187  << " got=" << ftag << std::endl;
188  MADNESS_EXCEPTION("TextFstreamInputArchive: check_tag: types do not match/corrupt file", 1);
189  }
190  }
191 
192  /// Read the closing "cookie" tag.
193 
194  /// \tparam T The expected data type between the tags.
195  template <class T>
196  inline void check_end_tag() const {
197  check_start_tag<T>(true);
198  }
199 
200  /// Load from the filestream.
201 
202  /// The function only appears (due to \c enable_if) if \c T is
203  /// serializable.
204  /// \tparam T The type of data to be read.
205  /// \param[out] t Where to put the loaded data.
206  /// \param[in] n The number of data items to be loaded.
207  template <class T>
208  typename std::enable_if< madness::is_iostreammable_v<T> >::type
209  load(T* t, long n) const {
210  for (long i=0; i<n; ++i) is >> t[i];
