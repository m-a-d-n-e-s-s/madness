236  bool optimize_conjugate_gradients(Tensor<double>& x) {
237 
238 // Tensor<double> project_ext=projector_external_dof(target->molecule());
239 
240 
241  // initial energy and gradient gradient
242  double energy=0.0;
243  Tensor<double> gradient;
244 
245  // first step is steepest descent
246  Tensor<double> displacement(x.size());
247  Tensor<double> oldgradient;
248  Tensor<double> old_displacement(x.size());
249  old_displacement.fill(0.0);

251  for (int iter=1; iter<parameters.maxiter(); ++iter) {
252 
253  // displace coordinates
254  if (iter>1) x+=displacement;
255 
256  // compute energy and gradient
257  target->value_and_gradient(x, energy, gradient);
258  print("gopt: new energy",energy);
259  gnorm = gradient.normf()/sqrt(gradient.size());
260  print("gopt: raw gradient norm ",gnorm);
261 
262  // remove external degrees of freedom (translation and rotation)
263  Tensor<double> project_ext=projector_external_dof(target->molecule(),parameters.remove_dof());
264  gradient=inner(gradient,project_ext);
265  gnorm = gradient.normf()/sqrt(gradient.size());
266  print("gopt: projected gradient norm ",gnorm);
267 
268  // compute new displacement
269  if (iter==1) {
270  displacement=-gradient;
271  } else {
272  double beta=0.0;
273  if (parameters.cg_method()=="fletcher_reeves")
274  beta=gradient.normf()/oldgradient.normf();
275  if (parameters.cg_method()=="polak_ribiere")
276  beta=gradient.normf()/(gradient-oldgradient).normf();
277  displacement=-gradient + beta * old_displacement;
278  }
279 
280  // save displacement for the next step
281  old_displacement=displacement;
282 
283  if (converged() and (displacement.normf()/displacement.size()<parameters.tol())) break;
284  }
285 
286  return converged();

289  /// effectively invert the hessian and multiply with the gradient
290  Tensor<double> new_search_direction2(const Tensor<double>& g,
291  const Tensor<double>& hessian) const {
292  Tensor<double> dx, s;
293  double tol = parameters.gradient_precision();
294  double trust = 1.0; // This applied in spectral basis
295 
296  // diagonalize the hessian:
297  // VT H V = lambda
298  // H^-1 = V lambda^-1 VT
299  Tensor<double> v, e;
300  syev(hessian, v, e);
301 
302  // Transform gradient into spectral basis
303  // H^-1 g = V lambda^-1 VT g
304  Tensor<double> gv = inner(g,v); // this is VT g == gT V == gv
305 
306  // Take step applying restriction
307  int nneg=0, nsmall=0, nrestrict=0;
308  for (int i=0; i<e.size(); ++i) {
309  if (e[i] < -tol) {
310  if (parameters.printtest()) printf(
311  " forcing negative eigenvalue to be positive %d %.1e\n", i, e[i]);
312  nneg++;
313  //e[i] = -2.0*e[i]; // Enforce positive search direction
314  e[i] = -0.1*e[i]; // Enforce positive search direction
315  }
316  else if (e[i] < tol) {
317  if (parameters.printtest()) printf(
318  " forcing small eigenvalue to be zero %d %.1e\n", i, e[i]);
319  nsmall++;
320  e[i] = tol;
321  gv[i]=0.0; // effectively removing this direction
322  }
323 
324  // this is the step -lambda^-1 gv
325  gv[i] = -gv[i] / e[i];
326  if (std::abs(gv[i]) > trust) { // Step restriction
327  double gvnew = trust*std::abs(gv(i))/gv[i];
328  if (parameters.printtest()) printf(
329  " restricting step in spectral direction %d %.1e --> %.1e\n",
330  i, gv[i], gvnew);
331  nrestrict++;
332  gv[i] = gvnew;
333  }
334  }
335  if (nneg || nsmall || nrestrict)
336  printf(" nneg=%d nsmall=%d nrestrict=%d\n", nneg, nsmall, nrestrict);
337 
338  // Transform back from spectral basis to give the displacements
339  // disp = -V lambda^-1 VT g = V lambda^-1 gv
340  return inner(v,gv);
341  }

345  /// compute the projector to remove transl. and rot. degrees of freedom
346 
347  /// taken from http://www.gaussian.com/g_whitepap/vib.htm
348  /// I don't really understand the concept behind the projectors, but it
349  /// seems to work, and it is not written down explicitly anywhere.
350  /// NOTE THE ERROR IN THE FORMULAS ON THE WEBPAGE !
351  /// @param[in] do_remove_dof which dof to remove: x,y,z,Rx,Ry,Rz (transl/rot)
352  static Tensor<double> projector_external_dof(const Molecule& mol,
353  const std::vector<std::string>& remove_dof) {
354 
355  // compute the translation vectors
356  Tensor<double> transx(3*mol.natom());
357  Tensor<double> transy(3*mol.natom());
358  Tensor<double> transz(3*mol.natom());
359  for (size_t i=0; i<mol.natom(); ++i) {
360  transx[3*i ]=sqrt(mol.get_atom(i).get_mass_in_au());
361  transy[3*i+1]=sqrt(mol.get_atom(i).get_mass_in_au());
362  transz[3*i+2]=sqrt(mol.get_atom(i).get_mass_in_au());
363  }
364 
365  // compute the rotation vectors
366 
367  // move the molecule to its center of mass and compute
368  // the moment of inertia tensor
369  Tensor<double> com=mol.center_of_mass();
370  Molecule mol2=mol;
371  mol2.translate(-1.0*com);
372  Tensor<double> I=mol2.moment_of_inertia();
373  I.scale(constants::atomic_mass_in_au);
374 
375  // diagonalize the moment of inertia
376  Tensor<double> v,e;
377  syev(I, v, e); // v being the "X" tensor on the web site
378  v=transpose(v);
379 
380 // Tensor<double> B(e.size());
381 // for (long i=0; i<e.size(); ++i) B(i)=1.0/(2.0*e(i));
382 // print("rotational constants in cm-1");
383 // print(constants::au2invcm*B);
384 
385  // rotation vectors
386  Tensor<double> rotx(3*mol.natom());
387  Tensor<double> roty(3*mol.natom());
388  Tensor<double> rotz(3*mol.natom());
389 
390  for (size_t iatom=0; iatom<mol.natom(); ++iatom) {
391 
392  // coordinates wrt the center of mass ("R" on the web site)
393  Tensor<double> coord(3);
394  coord(0l)=mol.get_atom(iatom).x-com(0l);
395  coord(1l)=mol.get_atom(iatom).y-com(1l);
396  coord(2l)=mol.get_atom(iatom).z-com(2l);
397 
398  // note the wrong formula on the Gaussian website:
399  // multiply with sqrt(mass), do not divide!
400  coord.scale(sqrt(mol.get_atom(iatom).get_mass_in_au()));
401 
402  // p is the dot product of R and X on the web site
403  Tensor<double> p=inner(coord,v);
404 
405  // Eq. (5)
406  rotx(3*iatom + 0)=p(1)*v(0,2)-p(2)*v(0,1);
407  rotx(3*iatom + 1)=p(1)*v(1,2)-p(2)*v(1,1);
408  rotx(3*iatom + 2)=p(1)*v(2,2)-p(2)*v(2,1);
409 
410  roty(3*iatom + 0)=p(2)*v(0,0)-p(0l)*v(0,2);
411  roty(3*iatom + 1)=p(2)*v(1,0)-p(0l)*v(1,2);
412  roty(3*iatom + 2)=p(2)*v(2,0)-p(0l)*v(2,2);
413 
414  rotz(3*iatom + 0)=p(0l)*v(0,1)-p(1)*v(0,0);
415  rotz(3*iatom + 1)=p(0l)*v(1,1)-p(1)*v(1,0);
416  rotz(3*iatom + 2)=p(0l)*v(2,1)-p(1)*v(2,0);
417 
418  }
419 
420  // move the translational and rotational vectors to a common tensor
421  bool remove_Tx=std::find(remove_dof.begin(), remove_dof.end(), "Tx")!=remove_dof.end();
422  bool remove_Ty=std::find(remove_dof.begin(), remove_dof.end(), "Ty")!=remove_dof.end();
423  bool remove_Tz=std::find(remove_dof.begin(), remove_dof.end(), "Tz")!=remove_dof.end();
424  bool remove_Rx=std::find(remove_dof.begin(), remove_dof.end(), "Rx")!=remove_dof.end();
425  bool remove_Ry=std::find(remove_dof.begin(), remove_dof.end(), "Ry")!=remove_dof.end();
426  bool remove_Rz=std::find(remove_dof.begin(), remove_dof.end(), "Rz")!=remove_dof.end();
427  Tensor<double> ext_dof(6,3*mol.natom());
428  if (remove_Tx) ext_dof(0l,_)=transx;
429  if (remove_Ty) ext_dof(1l,_)=transy;
430  if (remove_Tz) ext_dof(2l,_)=transz;
431 
432  if (remove_Rx) ext_dof(3l,_)=rotx;
433  if (remove_Ry) ext_dof(4l,_)=roty;
434  if (remove_Rz) ext_dof(5l,_)=rotz;
435  print("removing dof ",remove_Tx, remove_Ty, remove_Tz, remove_Rx, remove_Ry, remove_Rz);
436 
437  // normalize
438  for (int i=0; i<6; ++i) {
439  double norm=ext_dof(i,_).normf();
440  if (norm>1.e-14) ext_dof(i,_).scale(1.0/norm);
441  else ext_dof(i,_)=0.0;
442  }
443 
444  // compute overlap to orthonormalize the projectors
445  Tensor<double> ovlp=inner(ext_dof,ext_dof,1,1);
446  syev(ovlp,v,e);
447  ext_dof=inner(v,ext_dof,0,0);
448 
449  // normalize or remove the dof if necessary (e.g. linear molecules)
450  for (int i=0; i<6; ++i) {
451  if (e(i)<1.e-14) {
452  ext_dof(i,_).scale(0.0); // take out this degree of freedom
453  } else {
454  ext_dof(i,_).scale(1.0/sqrt(e(i))); // normalize
455  }
456  }
457

459  Tensor<double> projector(3*mol.natom(),3*mol.natom());
460  for (size_t i=0; i<3*mol.natom(); ++i) projector(i,i)=1.0;
461 
462  // compute the outer products of the projectors
463  // 1- \sum_i | t_i >< t_i |
464  projector-=inner(ext_dof,ext_dof,0,0);
465 
466  return projector;
