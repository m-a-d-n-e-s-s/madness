63  /// A thread safe, fast but simple doubled-ended queue.
64 
65  /// Since the point is speed, the implementation is a circular
66  /// buffer rather than a linked list so as to avoid the new/del
67  /// overhead. It will grow as needed, but presently will not
68  /// shrink. Had to modify STL API to make things thread safe.
69  ///
70  /// It is now rather heavily specialized to its only use.
71  template <typename T>
72  class DQueue : private CONDITION_VARIABLE_TYPE {
73  char pad[64]; ///< To put the lock and the data in separate cache lines
74  volatile size_t n __attribute__((aligned(64))); ///< Number of elements in the buffer
75  volatile size_t sz; ///< Current capacity
76  volatile T* volatile buf; ///< Actual buffer
77  volatile int _front; ///< Index of element at front of buffer
78  volatile int _back; ///< Index of element at back of buffer
79  DQStats stats;

81  void grow() {
82  // ASSUME WE ALREADY HAVE THE MUTEX WHEN IN HERE
83  ++(stats.ngrow);
84  if (sz != n) MADNESS_EXCEPTION("assertion failure in dqueue::grow", static_cast<int>(sz));
85  size_t oldsz = sz;
86  if (sz < 32768)
87  sz = 65536;
88  else if (sz <= 1048576)
89  sz *= 2;
90  else
91  sz += 1048576;
92  volatile T* volatile nbuf = new T[sz];
93  int lo = sz/2 - oldsz/2;
94  for (int i=_front; i<int(oldsz); ++i,++lo) {
95  nbuf[lo] = buf[i];
96  }
97  if (_front > 0) {
98  for (int i=0; i<=_back; ++i,++lo) {
99  nbuf[lo] = buf[i];
100  }
101  }
102  _front = sz/2 - oldsz/2;
103  _back = _front + n - 1;
104  delete [] buf;
105  buf = nbuf;
106  //sanity_check();
107  }
108 
109  void sanity_check() const {
110  // ASSUME WE ALREADY HAVE THE MUTEX WHEN IN HERE
111  int num = _back - _front + 1;
112  if (num < 0) num += sz;
113  if (num==int(sz) && n==0) num=0;
114  if (num==0 && n==sz) num=sz;
115  //if (long(n) != num) print("size",sz,"front",_front,"back",_back,"n",n,"num",num);
116  MADNESS_ASSERT(long(n) == num);
117  }

119  void push_back_with_lock(const T& value) {
120  size_t nn = n;
121  size_t ss = sz;
122  if (nn == ss) {
123  grow();
124  ss = sz;
125  }
126  ++nn;
127  if (nn > stats.nmax) stats.nmax = nn;
128  n = nn;
129 
130  int b = _back + 1;
131  if (b >= int(ss)) b = 0;
132  buf[b] = value;
133  _back = b;
134  ++(stats.npush_back);
135 
136  signal();
137  }
