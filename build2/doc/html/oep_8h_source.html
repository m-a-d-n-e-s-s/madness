75 struct interpolate_munge_refdens{
76 
77  double thresh_high, thresh_low, log_high, log_low;
78  typedef double resultT;
79 
80  // same default values as for dens_thresh_hi and dens_thresh_lo
81  interpolate_munge_refdens(const double hi = 1.0e-4, const double lo = 1.0e-7) {
82  thresh_high = hi;
83  thresh_low = lo;
84  log_high = log10(thresh_high);
85  log_low = log10(thresh_low);
86  }
87 
88  /// @param[out] U result
89  /// @param[in] r reference density
90  Tensor<double> operator()(const Key<3>& key, const Tensor<double>& refdens) const {
91 
92  // set 1 if above munge interval (dens > thresh_high) or 0 if below munge interval (dens < thresh_low)
93  // interpolate inbetween with logarithmic form of (r - lo)/(hi - lo) because this yields near linear behavior
94  Tensor<double> U = copy(refdens); // copy refdens in oder to have the same dimension in the tensor
95  U.fill(1.0); // start with a fuction that is 1.0 everywhere
96  ITERATOR(
97  U, double r = refdens(IND);
98  if (r > thresh_high) {
99  U(IND) = 1.0;
100  } else if (r < thresh_low) {
101  U(IND) = 0.0;
102  } else {
103  U(IND) = (log10(refdens(IND)) - log_low)/(log_high - log_low);
104  }
105  );
106  return U;
107 
108  }
109 
110  template <typename Archive>
111  void serialize(Archive& ar) {}

435  /// compute Slater potential (Kohut, 2014, equation (15))
436  real_function_3d compute_slater_potential(const vecfuncT& nemo, const long homo_ind) const {
437 
438  Exchange<double,3> K(world, this, 0);
439  vecfuncT Knemo = K(nemo);
440  // 2.0*R_square in numerator and density (rho) cancel out upon division
441  real_function_3d numerator = dot(world, nemo, Knemo);
442  real_function_3d rho = dot(world, nemo, nemo);
443 // save(numerator, "Slaterpotential_numerator");
444 
445  // dividing by rho: the minimum value for rho is dens_thresh_lo
446  real_function_3d Vs = -1.0*binary_op(numerator, rho, dens_inv1(oep_param.dens_thresh_inv())); // in Kohut_2014, -1.0 is included in K
447  if (oep_param.saving_amount() >= 3) save(Vs, "Slaterpotential_nolra");
448 
449  // long-range asymptotic behavior for Slater potential is \int 1/|r-r'| * |phi_HOMO|^2 dr'
450  // in order to compute this lra, use Coulomb potential with only HOMO density (= |phi_HOMO|^2)
451  Coulomb J(world, this);
452  real_function_3d lra = -1.0*J.compute_potential(R_square*square(nemo[homo_ind]));
453  if (oep_param.saving_amount() >= 3) save(lra, "lra_slater");

475  /// compute average ionization energy I like Kohut, 2014, equations (21) and (25)
476  real_function_3d compute_average_I(const vecfuncT& nemo, const tensorT eigvals) const {
477 
478  // transform 1d tensor eigvals to vector epsilon
479  std::vector<double> epsilon(eigvals.size());
480  for (int i = 0; i < eigvals.size(); i++) epsilon[i] = eigvals(i);
481 
482  vecfuncT nemo_square = square(world, nemo); // |nemo|^2
483  scale(world, nemo_square, epsilon); // epsilon*|nemo|^2
484  // 2.0*R_square in numerator and density (rho) cancel out upon division
485  real_function_3d numerator = sum(world, nemo_square);
486  real_function_3d rho = dot(world, nemo, nemo);
487 
488  // like Kohut, 2014, equations (21) and (25)
489  real_function_3d I = -1.0*binary_op(numerator, rho, dens_inv1(oep_param.dens_thresh_inv()));
490 
491  // munge I for long-range asymptotic behavior which is -epsilon_HOMO
492  print("computing I: index of HOMO is", homo_ind(eigvals));
493  double lra = -1.0*eigvals(homo_ind(eigvals));
494  real_function_3d weight = compute_weighting_function(nemo);
495  I = I*weight + lra*(1.0 - weight);
496 
497  return I;
498 
499  }

501  /// compute the total kinetic energy density devided by the density tau/rho with equation (6) from Kohut
502  real_function_3d compute_total_kinetic_density(const vecfuncT& nemo, const tensorT eigvals) const {
503 
504  // compute the denominator rho (density)
505  real_function_3d rho = compute_density(nemo);
506 
507  // compute the numerator tau
508 
509  // get \nabla R and (\nabla R)^2 via and U1 = -1/R \nabla R and U1dot = (1/R \nabla R)^2
510  const vecfuncT U1 = this->ncf->U1vec();
511  NuclearCorrelationFactor::U1_dot_U1_functor u1_dot_u1(ncf.get());
512  const real_function_3d U1dot = real_factory_3d(world).functor(u1_dot_u1).truncate_on_project();
513 
514  // get \nabla nemo
515  std::vector<vecfuncT> grad_nemo(nemo.size());
516  for (long i = 0; i < nemo.size(); i++) {
517  if(calc->param.dft_deriv() == "bspline") grad_nemo[i] = grad_bspline_one(nemo[i]); // gradient using b-spline
518  else grad_nemo[i] = grad(nemo[i]); // default gradient using abgv
519  }
520 
521  // compute tau = 1/2 * sum |phi_i|^2
522  // = 1/2 * sum {(\nabla R)^2 * nemo_i^2 + 2 * R * nemo_i * (\nabla R) * (\nabla nemo_i)) + R^2 * (\nabla nemo_i)^2}
523  // = 1/2 * R^2 * sum {U1dot * nemo_i^2 + 2 * nemo_i * U1 * (\nabla nemo_i)) + (\nabla nemo_i)^2}
524  vecfuncT grad_nemo_squared(nemo.size());
525  for (long i = 0; i < nemo.size(); i++) {
526  grad_nemo_squared[i] = U1dot*square(nemo[i])
527  - 2.0*nemo[i]*dot(world, U1, grad_nemo[i])
528  + dot(world, grad_nemo[i], grad_nemo[i]);
529  }
530  real_function_3d tau = R_square*sum(world, grad_nemo_squared); // 1/2 * sum |Nabla nemo|^2 (* 2 because closed shell)
531 
532  // calculate quotient = tau/rho
533  real_function_3d quotient = binary_op(tau, rho, dens_inv1(oep_param.dens_thresh_inv()));
534 
535  // munge quotient for long-range asymptotic behavior which is -epsilon_HOMO
536  print("computing tau/rho: index of HOMO is", homo_ind(eigvals));
537  double lra = -1.0*eigvals(homo_ind(eigvals));
538  real_function_3d weight = compute_weighting_function(nemo);
539  quotient = quotient*weight + lra*(1.0 - weight);
540 
541  return quotient;
542 
543  }

545  /// compute the Pauli kinetic energy density devided by the density tau_P/rho with equation (16) from Ospadov, 2017
546  real_function_3d compute_Pauli_kinetic_density(const vecfuncT& nemo, const tensorT eigvals) const {
547 
548  // compute the denominator rho (density)
549  real_function_3d rho_square = square(dot(world, nemo, nemo)); // density squared without 2.0*R_square
550 
551  // compute the numerator tau_P
552 
553  // get \nabla nemo
554  std::vector<vecfuncT> grad_nemo(nemo.size());
555  for (long i = 0; i < nemo.size(); i++) {
556  if(calc->param.dft_deriv() == "bspline") grad_nemo[i] = grad_bspline_one(nemo[i]); // gradient using b-spline
557  else grad_nemo[i] = grad(nemo[i]); // default gradient using abgv
558  }
559 
560  vecfuncT grad_nemo_term;
561  for (long i = 0; i < nemo.size(); i++) {
562  for (long j = i + 1; j < nemo.size(); j++) {
563  vecfuncT tmp = nemo[i]*grad_nemo[j] - nemo[j]*grad_nemo[i];
564  grad_nemo_term.push_back(dot(world, tmp, tmp));
565  }
566  }
567  real_function_3d numerator = sum(world, grad_nemo_term); // numerator = tau_P * 2 * rho / R^4
568 
569  // calculate quotient = tau_P/rho
570  real_function_3d quotient = 0.5*binary_op(numerator, rho_square, dens_inv1(oep_param.dens_thresh_inv()));
571 
572  // munge quotient for long-range asymptotic behavior which is 0
573  print("computing tau_P/rho: index of HOMO is", homo_ind(eigvals));
574  real_function_3d weight = compute_weighting_function(nemo);
575  quotient = quotient*weight; // + lra*(1.0 - weight) but lra is 0 here
576 
577  return quotient;
578 
579  }

633  double compute_kinetic_energy(const vecfuncT phi) const {
634

636  double E_kin = 0.0;

648  /// compute Evir using Levy-Perdew virial relation (Kohut_2014, (43) or Ospadov_2017, (25))
649  double compute_exchange_energy_vir(const vecfuncT phi, const real_function_3d Vx) const {
650 
651  // make vector of functions r = (x, y, z)
652  auto monomial_x = [] (const coord_3d& r) {return r[0];};
653  auto monomial_y = [] (const coord_3d& r) {return r[1];};
654  auto monomial_z = [] (const coord_3d& r) {return r[2];};
655  vecfuncT r(3);
656  r[0]=real_factory_3d(world).functor(monomial_x);
657  r[1]=real_factory_3d(world).functor(monomial_y);
658  r[2]=real_factory_3d(world).functor(monomial_z);
659 
660  // for density note that phi is R*nemo, so no R_square is needed
661  real_function_3d rho = 2.0*dot(world, phi, phi); // 2 because closed shell
662  real_function_3d rhoterm = 3*rho + dot(world, r, grad(rho));
663  double Ex = inner(Vx, rhoterm);
664  return Ex;
665
