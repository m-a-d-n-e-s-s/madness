55 template<typename T, std::size_t NDIM>
56 class Function;
57 /// This class implements the symmetry operations (not the point groups)
58 class pg_operator {
59 
60 public:
61 
62  /// ctor with explicit mirror maps
63  pg_operator(std::string symbol, std::string name, const std::vector<long> mm,
64  const std::vector<long> md)
65  : symbol_(symbol), name_(name), mirrormap(mm), mapdim_(md) {}
66 
67  /// return the name of the symmetry operator
68  std::string name() const {return name_;}
69 
70  std::string symbol() const {return symbol_;}
71 
72  /// apply the operator on an n-dimensional MRA function
73  template<typename T, std::size_t NDIM>
74  Function<T,NDIM> operator()(const Function<T,NDIM>& f, bool fence=true) const {
75  Function<T,NDIM> result;
76 
77  if (name_=="identity") {
78  result=copy(f,fence);
79 
80  } else if (name_=="inversion"){
81 
82  std::vector<long> mm(NDIM,-1);
83  result=mirror(f,mm,fence);
84 
85  } else {
86  result=map_and_mirror(f,mapdim_,mirrormap,fence);
87  }
88 
89  return result;
90  }
91 
92  /// apply the operator on an n-dimensional MRA function
93  template<typename T, std::size_t NDIM>
94  std::vector<Function<T,NDIM> > operator()(const std::vector<Function<T,NDIM> >& vf, bool fence=true) const {
95  std::vector<Function<T,NDIM> > result(vf.size());
96 
97  // fast return
98  if (vf.size()==0) return result;
99  World& world=vf.begin()->world();
100 
101  for (size_t i=0; i<vf.size(); ++i) result[i]=this->operator()(vf[i],false);
102 
103  if (fence) world.gop.fence();
104  return result;
105  }
106 
107 
108  friend
109  std::ostream& operator<<(std::ostream& s, const pg_operator& pg_op) {
110  s << "Symmetry operator " << pg_op.name() ;
111  return s;
112  }
113 
114 
115 private:
116 
117  std::string symbol_;
118  std::string name_;
119  std::vector<long> mirrormap;
120  std::vector<long> mapdim_;
121 
122 };
