46 /// struct for running a protocol of subsequently tightening precision
47 class SCFProtocol {
48 public:
49  SCFProtocol(World& w, const CalculationParameters& param,
50  const std::string name, const bool restart)
51  : world(w), filename(name), restart(restart), converged(false),
52  start_prec(1.e-4), current_prec(start_prec), end_prec(param.econv()),
53  thresh(1.e-4), econv(1.e-4), dconv(1.e-3), user_dconv(1.e-20) {
54  user_dconv=param.dconv();
55  }
56 
57  World& world;
58 
59  std::string filename; ///< name for the restart data file
60  bool restart; ///< do a restart from file (if possible)
61  bool converged; ///< flag if protocol has converged
62 
63  double start_prec; ///< starting precision, typically 1.e-4
64  double current_prec; ///< current precision
65  double end_prec; ///< final precision
66 
67  double thresh; ///< numerical precision of representing functions
68  double econv; ///< energy convergence of SCF calculations
69  double dconv; ///< density convergence of SCF calculations
70  double user_dconv; ///< density convergence provided by user
71 
72  void initialize() {
73 
74  // don't do anything if this protocol is already converged
75  if (converged) return;
76 
77  // try to read restart data file
78  if (restart) {
79  std::ifstream f(filename.c_str());
80  if(not f.fail()) {
81  std::string s, tag;
82  while (std::getline(f,s)) {
83  std::istringstream ss(s);
84  ss >> tag;
85  if (tag=="start_prec") ss >> start_prec;
86  if (tag=="end_prec") ss >> end_prec;
87  if (tag=="current_prec") ss >> current_prec;
88  if (tag=="econv") ss >> econv;
89  if (tag=="dconv") ss >> dconv;
90  if (tag=="thresh") ss >> thresh;
91  if (tag=="user_dconv") ss >> user_dconv;
92  }
93  }
94  } else {
95  current_prec=start_prec;
96  infer_thresholds(current_prec);
97  }
98  if (world.rank()==0) {
99  printf("\nstarting protocol at time %8.1fs \n",wall_time());
100  print("precision steps ",start_prec," --> ",end_prec);
101  print("protocol: thresh",thresh,"econv ",econv,"dconv",dconv);
102  }
103  }
104 
105  bool finished() const {
106  if(world.rank()==0) printf("\nending protocol at time %8.1fs \n",wall_time());
107  return converged;
108  }
109 
110  /// go to the next level
111  SCFProtocol& operator++() {
112  if (current_prec*0.9999>end_prec) {
113  current_prec*=0.1;
114  if (current_prec<end_prec) current_prec=end_prec;
115  infer_thresholds(current_prec);
116  if(world.rank()==0) print("protocol: thresh",thresh,"econv ",econv,"dconv",dconv);
117  } else {
118  converged=true;
119  }
120 
121  // update restart data on file
122  std::ofstream f(filename.c_str());
123  f << "start_prec " << start_prec << std::endl;;
124  f << "end_prec " << end_prec << std::endl;;
125  f << "current_prec " << current_prec << std::endl;;
126  f << "econv " << econv << std::endl;;
127  f << "dconv " << dconv << std::endl;;
128  f << "thresh " << thresh << std::endl;;
129  f << "user_dconv " << user_dconv << std::endl;;
130 
131  return *this;
132  }
133 
134  /// infer thresholds starting from a target precision
135  void infer_thresholds(const double prec) {
136  econv=prec;
137  thresh=econv;
138  dconv=std::min(1.e-3,sqrt(econv)*0.1);
139 // dconv=std::min(1.e-3,econv*10.0);
140  if (approx(current_prec,end_prec)) dconv=user_dconv; // respect the user
141  }
142 
143  /// compare two positive doubles to be equal
144  bool approx(const double a, const double b) const {
145  return (std::abs(a/b-1.0)<1.e-12);
146  }
147 };
