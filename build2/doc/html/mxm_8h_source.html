59  /// Matrix \c += Matrix * matrix reference implementation (slow but correct)
60  template <typename T, typename Q, typename S>
61  static inline void mxm_reference(long dimi, long dimj, long dimk,
62  T* MADNESS_RESTRICT c, const Q* MADNESS_RESTRICT a,
63  const S* MADNESS_RESTRICT b) {
64  /*
65  c(i,j) = c(i,j) + sum(k) a(i,k)*b(k,j)
66 
67  where it is assumed that the last index in each array is has unit
68  stride and the dimensions are as provided.
69  */
70 
71  for (long i=0; i<dimi; ++i) {
72  for (long k=0; k<dimk; ++k) {
73  for (long j=0; j<dimj; ++j) {
74  c[i*dimj+j] += a[i*dimk+k]*b[k*dimj+j];
75  }
76  }
77  }
78  }

151  /// Matrix = Matrix transpose * matrix ... slow reference implementation
152 
153  /// This routine does \c C=AT*B whereas mTxm does C=C+AT*B.
154  /// \code
155  /// c(i,j) = sum(k) a(k,i)*b(k,j) <------ does not accumulate into C
156  /// \endcode
157  ///
158  /// \c ldb is the last dimension of b in C storage (the leading dimension
159  /// in fortran storage). It is here to accomodate multiplying by a matrix
160  /// stored with \c ldb>dimj which happens in madness when transforming with
161  /// low rank matrices. A matrix in dense storage has \c ldb=dimj which is
162  /// the default for backward compatibility.
163  template <typename aT, typename bT, typename cT>
164  void mTxmq_reference(long dimi, long dimj, long dimk,
165  cT* MADNESS_RESTRICT c, const aT* a, const bT* b, long ldb=-1) {
166  if (ldb == -1) ldb=dimj;
167  MADNESS_ASSERT(ldb>=dimj);
168  //std::cout << "IN GENERIC mTxmq " << tensor_type_names[TensorTypeData<aT>::id] << " " << tensor_type_names[TensorTypeData<bT>::id] << " " << tensor_type_names[TensorTypeData<cT>::id] << "\n";
169  for (long i=0; i<dimi; ++i,c+=dimj,++a) {
170  for (long j=0; j<dimj; ++j) c[j] = 0.0;
171  const aT *aik_ptr = a;
172  for (long k=0; k<dimk; ++k,aik_ptr+=dimi) {
173  aT aki = *aik_ptr;
174  for (long j=0; j<dimj; ++j) {
175  c[j] += aki*b[k*ldb+j];
176  }
177  }
178  }
179  }

305  /// Matrix transpose * matrix (hand unrolled version)
306 
307  template <>
308  inline void mTxm(long dimi, long dimj, long dimk,
309  double* MADNESS_RESTRICT c, const double* MADNESS_RESTRICT a,
310  const double* MADNESS_RESTRICT b) {
311  /*
312  c(i,j) = c(i,j) + sum(k) a(k,i)*b(k,j) <--- NOTE ACCUMULATION INTO C
313 
314  where it is assumed that the last index in each array is has unit
315  stride and the dimensions are as provided.
316 
317  i loop might be long in anticipated application
318 
319  4-way unrolled k loop ... empirically fastest on PIII
320  compared to 2/3 way unrolling (though not by much).
321  */
322 
323  long dimk4 = (dimk/4)*4;
324  for (long i=0; i<dimi; ++i,c+=dimj) {
325  const double* ai = a+i;
326  const double* p = b;
327  for (long k=0; k<dimk4; k+=4,ai+=4*dimi,p+=4*dimj) {
328  double ak0i = ai[0 ];
329  double ak1i = ai[dimi];
330  double ak2i = ai[dimi+dimi];
331  double ak3i = ai[dimi+dimi+dimi];
332  const double* bk0 = p;
333  const double* bk1 = p+dimj;
334  const double* bk2 = p+dimj+dimj;
335  const double* bk3 = p+dimj+dimj+dimj;
336  for (long j=0; j<dimj; ++j) {
337  c[j] += ak0i*bk0[j] + ak1i*bk1[j] + ak2i*bk2[j] + ak3i*bk3[j];
338  }
339  }
340  for (long k=dimk4; k<dimk; ++k) {
341  double aki = a[k*dimi+i];
342  const double* bk = b+k*dimj;
343  for (long j=0; j<dimj; ++j) {
344  c[j] += aki*bk[j];
345  }
346  }
347  }
348  }

352  /// Matrix * matrix transpose (hand unrolled version)
353 
354  template <>
355  inline void mxmT(long dimi, long dimj, long dimk,
356  double* MADNESS_RESTRICT c,
357  const double* MADNESS_RESTRICT a, const double* MADNESS_RESTRICT b) {
358  /*
359  c(i,j) = c(i,j) + sum(k) a(i,k)*b(j,k)
360 
361  where it is assumed that the last index in each array is has unit
362  stride and the dimensions are as provided.
363 
364  j loop might be long in anticipated application
365 
366  Unrolled i loop. Empirically fastest on PIII compared
367  to unrolling j, or both i&j.
368  */
369 
370  long dimi2 = (dimi/2)*2;
371  for (long i=0; i<dimi2; i+=2) {
372  const double* ai0 = a+i*dimk;
373  const double* ai1 = a+i*dimk+dimk;
374  double* MADNESS_RESTRICT ci0 = c+i*dimj;
375  double* MADNESS_RESTRICT ci1 = c+i*dimj+dimj;
376  for (long j=0; j<dimj; ++j) {
377  double sum0 = 0;
378  double sum1 = 0;
379  const double* bj = b + j*dimk;
380  for (long k=0; k<dimk; ++k) {
381  sum0 += ai0[k]*bj[k];
382  sum1 += ai1[k]*bj[k];
383  }
384  ci0[j] += sum0;
385  ci1[j] += sum1;
386  }
387  }
388  for (long i=dimi2; i<dimi; ++i) {
389  const double* ai = a+i*dimk;
390  double* MADNESS_RESTRICT ci = c+i*dimj;
391  for (long j=0; j<dimj; ++j) {
392  double sum = 0;
393  const double* bj = b+j*dimk;
394  for (long k=0; k<dimk; ++k) {
395  sum += ai[k]*bj[k];
396  }
397  ci[j] += sum;
398  }
399  }
400  }

402  /// Matrix * matrix (hand unrolled version)
403  template <>
404  inline void mxm(long dimi, long dimj, long dimk,
405  double* MADNESS_RESTRICT c, const double* MADNESS_RESTRICT a, const double* MADNESS_RESTRICT b) {
406  /*
407  c(i,j) = c(i,j) + sum(k) a(i,k)*b(k,j)
408 
409  where it is assumed that the last index in each array is has unit
410  stride and the dimensions are as provided.
411 
412  4-way unrolled k loop ... empirically fastest on PIII
413  compared to 2/3 way unrolling (though not by much).
414  */
415 
416  long dimk4 = (dimk/4)*4;
417  for (long i=0; i<dimi; ++i, c+=dimj,a+=dimk) {
418  const double* p = b;
419  for (long k=0; k<dimk4; k+=4,p+=4*dimj) {
420  double aik0 = a[k ];
421  double aik1 = a[k+1];
422  double aik2 = a[k+2];
423  double aik3 = a[k+3];
424  const double* bk0 = p;
425  const double* bk1 = bk0+dimj;
426  const double* bk2 = bk1+dimj;
427  const double* bk3 = bk2+dimj;
428  for (long j=0; j<dimj; ++j) {
429  c[j] += aik0*bk0[j] + aik1*bk1[j] + aik2*bk2[j] + aik3*bk3[j];
430  }
431  }
432  for (long k=dimk4; k<dimk; ++k) {
433  double aik = a[k];
434  for (long j=0; j<dimj; ++j) {
435  c[j] += aik*b[k*dimj+j];
436  }
437  }
438  }
439  }

441  /// Matrix transpose * matrix transpose (hand tiled and unrolled)
442  template <>
443  inline void mTxmT(long dimi, long dimj, long dimk,
444  double* MADNESS_RESTRICT csave, const double* MADNESS_RESTRICT asave, const double* MADNESS_RESTRICT b) {
445  /*
446  c(i,j) = c(i,j) + sum(k) a(k,i)*b(j,k)
447 
448  where it is assumed that the last index in each array is has unit
449  stride and the dimensions are as provided.
450 
451  Tiled k, copy row of a into temporary, and unroll j once.
452  */
453 
454  const int ktile=32;
455  double ai[ktile];
456  long dimj2 = (dimj/2)*2;
457 
458  for (long klo=0; klo<dimk; klo+=ktile, asave+=ktile*dimi, b+=ktile) {
459  long khi = klo+ktile;
460  if (khi > dimk) khi = dimk;
461  long nk = khi-klo;
462 
463  const double * MADNESS_RESTRICT a = asave;
464  double * MADNESS_RESTRICT c = csave;
465  for (long i=0; i<dimi; ++i,c+=dimj,++a) {
466  const double* q = a;
467  for (long k=0; k<nk; ++k,q+=dimi) ai[k] = *q;
468 
469  const double* bj0 = b;
470  for (long j=0; j<dimj2; j+=2,bj0+=2*dimk) {
471  const double* bj1 = bj0+dimk;
472  double sum0 = 0;
473  double sum1 = 0;
474  for (long k=0; k<nk; ++k) {
475  sum0 += ai[k]*bj0[k];
476  sum1 += ai[k]*bj1[k];
477  }
478  c[j ] += sum0;
479  c[j+1] += sum1;
480  }
481 
482  for (long j=dimj2; j<dimj; ++j,bj0+=dimk) {
483  double sum = 0;
484  for (long k=0; k<nk; ++k) {
485  sum += ai[k]*bj0[k];
486  }
487  c[j] += sum;
488  }
489  }
490  }
491  }

493  /*
494  * mtxm, but with padded buffers.
495  *
496  * ext_b is the extent of the b array, so shrink() isn't needed.
497  */
498  template <typename aT, typename bT, typename cT>
499  void mTxmq_padding(long dimi, long dimj, long dimk, long ext_b,
500  cT* c, const aT* a, const bT* b) {
501  const int alignment = 4;
502  bool free_b = false;
503  long effj = dimj;
504 
505  /* Setup a buffer for c if needed */
506  cT* c_buf = c;
507  if (dimj%alignment) {
508  effj = (dimj | 3) + 1;
509  c_buf = (cT*)malloc(sizeof(cT)*dimi*effj);
510  }
511 
512  /* Copy b into a buffer if needed */
513  if (ext_b%alignment) {
514  free_b = true;
515  bT* b_buf = (bT*)malloc(sizeof(bT)*dimk*effj);
516 
517  bT* bp = b_buf;
518  for (long k=0; k<dimk; k++, bp += effj, b += ext_b)
519  memcpy(bp, b, sizeof(bT)*dimj);
520 
521  b = b_buf;
522  ext_b = effj;
523  }
524 
525  cT* c_work = c_buf;
526  /* mTxm */
527  for (long i=0; i<dimi; ++i,c_work+=effj,++a) {
528  for (long j=0; j<dimj; ++j) c_work[j] = 0.0;
529  const aT *aik_ptr = a;
530  for (long k=0; k<dimk; ++k,aik_ptr+=dimi) {
531  aT aki = *aik_ptr;
532  for (long j=0; j<dimj; ++j) {
533  c_work[j] += aki*b[k*ext_b+j];
534  }
535  }
536  }
537 
538  /* Copy c out if needed */
539  if (dimj%alignment) {
540  cT* ct = c_buf;
541  for (long i=0; i<dimi; i++, ct += effj, c += dimj)
542  memcpy(c, ct, sizeof(cT)*dimj);
543 
544  free(c_buf);
545  }
546 
547  /* Free the buffer for b */
548  if (free_b) free((bT*)b);
549  }
