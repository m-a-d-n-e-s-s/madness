56  /// \brief Range, vaguely a la Intel TBB, to encapsulate a random-access,
57  /// STL-like start and end iterator with chunksize.
58 
59  /// \tparam iteratorT The iterator type.
60  template <typename iteratorT>
61  class Range {
62  long n; ///< Number of items to iterator over. \todo Could this be replaced by size_t?
63  iteratorT start; ///< First item for iteration.
64  iteratorT finish; ///< Last item for iteration (first past the end, conventionally).
65  int chunksize; ///< Number of items to give to each thread/process.
66 
67  public:
68  using iterator = iteratorT; ///< Alias for the iterator type.
69 
70  /// Makes the range [start, finish).
71 
72  /// The motivated reader should look at the Intel TBB range,
73  /// partitioner, split, concepts, etc.
74  /// \param[in] start The first item to iterate over.
75  /// \param[in] finish The last item for iteration (one past the end).
76  /// \param[in] chunk The number of items to give to each thread/process.
77  Range(const iterator& start, const iterator& finish, int chunk=1)
78  : n(distance(start,finish))
79  , start(start)
80  , finish(finish)
81  , chunksize(chunk)
82  {
83  if (chunksize < 1) chunksize = 1;
84  }
85 
86  /// Copy constructor. Cost is O(1).
87 
88  /// \todo Can we make this `= default`?
89  /// \param[in] r The \c Range to copy.
90  Range(const Range& r)
91  : n(r.n)
92  , start(r.start)
93  , finish(r.finish)
94  , chunksize(r.chunksize)
95  {}
96 
97  /// Splits range between new and old (r) objects. Cost is O(1).
98 
99  /// \param[in] left The range to be split.
100  Range(Range& left, const Split& /*split*/)
101  : n(0)
102  , start(left.finish)
103  , finish(left.finish)
104  , chunksize(left.chunksize)
105  {
106  if (left.n > chunksize) {
107  int nleft = (left.n+1)/2;
108 
109  start = left.start;
110  advance(start,nleft);
111  finish = left.finish;
112  n = left.n - nleft;
113 
114  left.finish = start;
115  left.n = nleft;
116  }
117  }

151  private:
152  /// Advance by \c n elements in the range.
153 
154  /// This version is for cases where the "iterator type" is integral.
155  /// \tparam integralT The integral iterator type.
156  /// \tparam distanceT The distance type.
157  /// \param[in,out] i The integral iterator.
158  /// \param[in] n The number of elements to advance.
159  template<typename integralT, typename distanceT>
160  inline static typename std::enable_if<std::is_integral<integralT>::value, void>::type
161  advance(integralT& i, distanceT n) { i += n; }
162 
163  /// Advance by \c n elements in the range.
164 
165  /// This version is for cases where the "iterator type" is not integral.
166  /// \tparam iterT The non-integral iterator type.
167  /// \tparam distanceT The distance type.
168  /// \param[in,out] it The iterator.
169  /// \param[in] n The number of elements to advance.
170  template<typename iterT, typename distanceT>
171  inline static typename std::enable_if<!std::is_integral<iterT>::value, void>::type
172  advance(iterT& it, distanceT n) { std::advance(it, n); }
173 
174  /// Calculate the distance between two iterators.
175 
176  /// This version is for cases where the "iterator type" is integral.
177  /// \tparam integralT The integral iterator type.
178  /// \param[in] first One iterator.
179  /// \param[in] last The other iterator.
180  /// \return The distance between the first and last iterators.
181  template<class integralT>
182  inline static typename std::enable_if<std::is_integral<integralT>::value, integralT>::type
183  distance(integralT first, integralT last) { return last - first; }
184 
185  /// Calculate the distance between two iterators.
186 
187  /// This version is for cases where the "iterator type" is not integral.
188  /// \tparam iterT The non-integral iterator type.
189  /// \param[in] first One iterator.
190  /// \param[in] last The other iterator.
191  /// \return The distance between the first and last iterators.
192  template<class iterT>
193  inline static auto
194  distance(iterT first, iterT last,
195  typename std::enable_if<!std::is_integral<iterT>::value>::type* = nullptr)
196  -> decltype(std::distance(first, last))
197  { return std::distance(first, last); }
198  };
