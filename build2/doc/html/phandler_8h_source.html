59 // T can be double or complex, but muParser results will always be real
60 // NDIM can be 1 to 6
61 // Varables allowed in strings are x,y,z,u,v,w, and r
62 // "r" will always mean magnitude of given vector
63 
64 template <typename T, int NDIM>
65 class ParserHandler : public madness::FunctionFunctorInterface<T, NDIM> {
66 
67  private:
68  mu::Parser parser; // the string-to-function parser
69  static const int MAX_DIM = 6;
70  mutable double vars[MAX_DIM]; // variables used in expression
71  mutable double r; // distance to origin
72  typedef madness::Vector<double, NDIM> coordT;
73 
74  public:
75  ParserHandler(std::string expr) {
76  if (NDIM > MAX_DIM) MADNESS_EXCEPTION("too many dim for parser!",0);
77  try {
78  if (NDIM >= 1) parser.DefineVar("x", &vars[0]);
79  if (NDIM >= 2) parser.DefineVar("y", &vars[1]);
80  if (NDIM >= 3) parser.DefineVar("z", &vars[2]);
81  if (NDIM >= 4) parser.DefineVar("u", &vars[3]);
82  if (NDIM >= 5) parser.DefineVar("v", &vars[4]);
83  if (NDIM >= 6) parser.DefineVar("w", &vars[5]);
84  parser.DefineVar("r", &r);
85  parser.SetExpr(expr);
86  } catch (mu::Parser::exception_type &e) {
87  std::cout << "muParser: " << e.GetMsg() << std::endl;
88  }
89  }
90 
91  virtual T operator() (const coordT &vals_in) const {
92  r = 0;
93  for (int i = 0; i < NDIM; ++i) {
94  vars[i] = vals_in[i];
95  r += vals_in[i]*vals_in[i];
96  }
97  r = sqrt(r);
98 
99  try {
100  return parser.Eval();
101  } catch (mu::Parser::exception_type &e) {
102  std::cout << "muParser: " << e.GetMsg() << std::endl;
103  return 0.0;
104  }
105  } // end operator()
106 
107  void changeExpr(const std::string newExpr) {
108  try {
109  parser.SetExpr(newExpr);
110  } catch (mu::Parser::exception_type &e) {
111  std::cout << "muParser: " << e.GetMsg() << std::endl;
112  }
113  }
114 }; // end class parserhandler
