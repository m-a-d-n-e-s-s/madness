325  /// Multiplies a matrix by a vector
326 
327  /// \f[
328  /// \mathbf{C} \leftarrow \alpha \mathbf{A}^{\mathrm{OpA}} \mathbf{B}^{\mathrm{OpB}} + \beta \mathbf{C}
329  /// \f]
330  /// \param OpA Operation to be applied to matrix \f$ \mathbf{A} \f$
331  /// \param OpB Operation to be applied to matrix \f$ \mathbf{B} \f$
332  /// \param m Rows in matrix \f$ \mathbf{C} \f$
333  /// \param n Columns in matrix \f$ \mathbf{C} \f$
334  /// \param k Inner dimension size for matrices \f$ \mathbf{A} \f$ and \f$ \mathbf{B} \f$
335  /// \param alpha Scaling factor applied to \f$ \mathbf{A} \f$ \c * \f$ \mathbf{B} \f$
336  /// \param a Pointer to matrix \f$ \mathbf{A} \f$
337  /// \param lda The size of the leading-order dimension of matrix \f$ \mathbf{A} \f$
338  /// \param b Pointer to matrix \f$ \mathbf{A} \f$
339  /// \param ldb The size of the leading-order dimension of matrix \f$ \mathbf{B} \f$
340  /// \param beta Scaling factor for matrix \f$ \mathbf{C} \f$
341  /// \param c Pointer to matrix \f$ \mathbf{C} \f$
342  /// \param ldc The size of the leading-order dimension of matrix \f$ \mathbf{C} \f$
343  ///@{
344  inline void gemm(const CBLAS_TRANSPOSE OpA, const CBLAS_TRANSPOSE OpB,
345  const integer m, const integer n, const integer k, const float alpha,
346  const float* a, const integer lda, const float* b, const integer ldb,
347  const float beta, float* c, const integer ldc)
348  {
349  MADNESS_ASSERT(OpA != ConjTrans);
350  MADNESS_ASSERT(OpB != ConjTrans);
351  static const char *op[] = { "n","t" };
352  F77_SGEMM(op[OpA], op[OpB], &m, &n, &k, &alpha, a, &lda, b, &ldb, &beta, c, &ldc);
353  }
