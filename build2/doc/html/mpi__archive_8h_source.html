113  /// Archive allowing buffering, serialization of data, and point-to-point communication between processes with MPI.
114  class MPIOutputArchive : public BaseOutputArchive {
115  mutable World* world; ///< The world.
116  ProcessID dest; ///< The destination process.
117  int tag; ///< MPI communication tag.
118  const std::size_t bufsize; ///< Size of the buffer.
119  mutable std::vector<unsigned char> v; ///< The buffer.
120  madness::archive::VectorOutputArchive var; ///< Archive for storing the buffer.
121 
122  public:
123  /// Construct an archive for sending data via MPI.
124 
125  /// \param[in] world The world.
126  /// \param[in] dest The destination process.
127  /// \param[in] tag MPI communication tag.
128  MPIOutputArchive(World& world, const ProcessID& dest, int tag=SafeMPI::MPIAR_TAG)
129  : world(&world), dest(dest), tag(tag), bufsize(1024*1024), v(), var(v) {
130  v.reserve(2*bufsize);
131  };
132 
133  /// Serialize data and store it in the buffer.
134 
135  /// The function only appears (due to \c enable_if) if \c T is
136  /// fundamental.
137  /// \tparam T The data type to be serialized.
138  /// \param[in] t Pointer to the data.
139  /// \param[in] n Number of data items to serialize.
140  template <class T>
141  inline
142  typename std::enable_if< is_trivially_serializable<T>::value, void >::type
143  store(const T* t, long n) const {
144  if (v.size() > bufsize) flush();
145  var.store(t,n);
146  if (v.size() > bufsize) flush();
147  }
148 
149  /// Send all data in the buffer to the destination process.
150 
151  /// \todo Check out the "?? why ??" comment.
152  void flush() const {
153  if (v.size()) {
154  world->mpi.Send(v.size(), dest, tag);
155  world->mpi.Send(&v[0], v.size(), dest, tag);
156  v.clear();
157  if (v.capacity() < 2*bufsize)
158  v.reserve(2*bufsize); // ?? why ??
159  }
160  };
161 
162  /// Close the archive (i.e., send any data in the buffer).
163  void close() {
164  flush();
165  };
166 
167  /// Destructor. Close the archive first, which may entail sending data.
168  ~MPIOutputArchive() {
169  close();
170  };

173  /// Archive allowing buffering, deserialization of data, and point-to-point communication between processes with MPI.
174  class MPIInputArchive : public BaseInputArchive {
175  mutable World* world; ///< The world.
176  ProcessID src; ///< The source process.
177  int tag; ///< MPI communication tag.
178  mutable std::vector<unsigned char> v; ///< The buffer.
179  madness::archive::VectorInputArchive var; ///< Archive for loading the buffer.
180 
181  public:
182  /// Construct an archive for receiving data via MPI.
183 
184  /// \param[in] world The world.
185  /// \param[in] src The source process.
186  /// \param[in] tag MPI communication tag.
187  MPIInputArchive(World& world, const ProcessID& src, int tag=SafeMPI::MPIAR_TAG)
188  : world(&world), src(src), tag(tag), v(), var(v) {};
189 
190  /// Deserialize data and store it in the buffer.
191 
192  /// The function only appears (due to \c enable_if) if \c T is
193  /// fundamental.
194  /// \tparam T The data type to be deserialized.
195  /// \param[out] t Pointer to the data.
196  /// \param[in] n Number of data items to serialize.
197  template <class T>
198  inline
199  typename std::enable_if< is_trivially_serializable<T>::value, void >::type
200  load(T* t, long n) const {
201  if (!var.nbyte_avail()) {
202  var.rewind();
203  std::size_t m;
204  world->mpi.Recv(m, src, tag);
205  v.resize(m);
206  world->mpi.Recv(v.data(), m, src, tag);
207  }
208  var.load(t,n);
209  }
210  };
