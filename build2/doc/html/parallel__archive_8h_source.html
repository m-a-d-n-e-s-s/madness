123  /// Opens the parallel archive.
124 
125  /// \attention When writing to a new archive, the number of writers
126  /// specified is used. When reading from an existing archive,
127  /// the number of `ionode`s is adjusted to to be the same as
128  /// the number that wrote the original archive. Presently,
129  /// we don't have logic to handle reading an archive using
130  /// fewer processes originally used to write it. If you
131  /// want to fix this have a look in worlddc.h for the only
132  /// spot that currently needs changing to make that work.
133  ///
134  /// \note The default number of I/O nodes is one and there is an
135  /// arbitrary maximum of 50 set. On IBM BG/P the maximum
136  /// is nproc/64.
137  /// \param[in] world The world.
138  /// \param[in] filename Name of the file.
139  /// \param[in] nwriter The number of writers.
140  void open(World& world, const char* filename, int nwriter=1) {
141  this->world = &world;
142  nio = nwriter;
143 #if defined(HAVE_IBMBGP) || defined(HAVE_IBMBGQ)
144  /* Jeff believes that BG is designed to handle up to *
145  * one file per node and I assume no more than 8 ppn */
146  int maxio = world.size()/8;
147 #else
148  int maxio = 50;
149 #endif
150  if (nio > maxio) nio = maxio; // Sanity?
151  if (nio > world.size()) nio = world.size();
152 
153  MADNESS_ASSERT(filename);
154  MADNESS_ASSERT(strlen(filename)-1<sizeof(fname));
155  strcpy(fname,filename); // Save the filename for later
156  char buf[256];
157  MADNESS_ASSERT(strlen(filename)+7 <= sizeof(buf));
158  sprintf(buf, "%s.%5.5d", filename, world.rank());
159 
160  if (world.rank() == 0) {
161  ar.open(buf);
162  ar & nio; // read/write nio from/to the archive
163  MADNESS_ASSERT(nio <= world.size());
164  }
165 
166  // Ensure all agree on value of nio that may also have changed if reading
167  world.gop.broadcast(nio, 0);
168 
169  // Other reader/writers can now open the local archive
170  if (is_io_node() && world.rank()) {
171  ar.open(buf);
172  }
173 
174  // Count #client
175  ProcessID me = world.rank();
176  nclient=0;
177  for (ProcessID p=0; p<world.size(); ++p) if (io_node(p) == me) ++nclient;
178 
179 // if (is_io_node()) {
180 // madness::print("I am an IO node with",nclient,"clients and file",buf);
181 // }
182 // else {
183 // madness::print("I am a client served by",my_io_node(),fname);
184 // }
185  }

187  /// Returns true if the named, unopened archive exists on disk with read access.
188 
189  /// This is a collective operation.
190  /// \param[in] world The world.
191  /// \param[in] filename Name of the file.
192  /// \return True if the named, unopened archive exists and is readable.
193  static bool exists(World& world, const char* filename) {
194  char buf[256];
195  MADNESS_ASSERT(strlen(filename)+7 <= sizeof(buf));
196  sprintf(buf, "%s.%5.5d", filename, world.rank());
197  bool status;
198  if (world.rank() == 0)
199  status = (access(buf, F_OK|R_OK) == 0);
200 
201  world.gop.broadcast(status);
202 
203  return status;
204  }

271  /// An archive for storing local or parallel data wrapping a \c BinaryFstreamOutputArchive.
272 
273  /// \note Writes of process-local objects only store the data from process zero.
274  ///
275  /// \note Writes of parallel containers (presently only \c WorldContainer) store all data.
276  ///
277  /// Each of the server or I/O nodes creates a
278  /// \c BinaryFstreamOutputArchive with the name `filename.rank`. Client
279  /// processes send their data to servers in a round-robin fashion.

303  /// An archive for storing local or parallel data, wrapping a \c BinaryFstreamInputArchive.
304 
305  /// \note Reads of process-local objects load the values originally stored by process zero,
