485  /// This class is used to record the task trace information, including
486  /// submit, start, and stop times, as well as identification information.
487  class TaskEvent {
488  private:
489  double times_[3]; ///< Task trace times: { submit, start, stop }.
490  std::pair<void*, unsigned short> id_; ///< Task identification information.
491  unsigned short threads_; ///< Number of threads used by the task.
492 
493  /// Print demangled symbol name.
494 
495  /// Add the demangled symbol name to \c os. If demangling fails,
496  /// the unmodified symbol name is used instead. If symbol is NULL,
497  /// "UNKNOWN" is used instead. A tab character is added after the
498  /// symbol name.
499  /// \param[in,out] os The output stream.
500  /// \param[in] symbol The symbol to add to the stream.
501  static void print_demangled(std::ostream& os, const char* symbol) {
502  // Get the demagled symbol name
503  if(symbol) {
504  int status = 0;
505 #ifndef USE_LIBIBERTY
506  const char* name = abi::__cxa_demangle(symbol, 0, 0, &status);
507 #else
508  char* name = cplus_demangle(symbol, DMGL_NO_OPTS);
509 #endif
510  // Append the demangled symbol name to the output stream
511  if(status == 0) {
512  os << name << "\t";
513  free((void*)name);
514  } else {
515  os << symbol << "\t";
516  }
517  } else {
518  os << "UNKNOWN\t";
519  }
520  }
521 
522  /// Get name of the function pointer.
523 
524  /// \return The mangled function name.
525  std::string get_name() const {
526 
527  // Get the backtrace symbol for the function address,
528  // which contains the function name.
529  void* const * func_ptr = const_cast<void* const *>(& id_.first);
530  char** bt_sym = backtrace_symbols(func_ptr, 1);
531 
532  // Extract the mangled function name from the backtrace
533  // symbol.
534  std::string mangled_name;
535 
536 #ifdef ON_A_MAC
537  // Format of bt_sym is:
538  // <frame #> <file name> <address> <mangled name> + <function offset>
539  std::istringstream iss(bt_sym[0]);
540  long frame;
541  std::string file, address;
542  iss >> frame >> file >> address >> mangled_name;
543 #else // Assume Linux
544  // Format of bt_sym is:
545  // <file>(<mangled name>+<function offset>) [<address>]
546  const char* first = strchr(bt_sym[0],'(');
547  if(first) {
548  ++first;
549  const char* last = strrchr(first,'+');
550  if(last)
551  mangled_name.assign(first, (last - first) - 1);
552  }
553 #endif // ON_A_MAC
554 
555  // Free the backtrace buffer
556  free(bt_sym);
557 
558  return mangled_name;
559  }
560 
561  public:
562 
563  // Only default constructors are needed.
564 
565  /// Record the start time of the task and collect task information.
566 
567  /// \param[in,out] id The task identifier (a function pointer or const char*)
568  /// and an integer to differentiate the different types.
569  /// \param[in] threads The number of threads this task uses.
570  /// \param[in] submit_time The time that the task was submitted to the
571  /// task queue.
572  void start(const std::pair<void*, unsigned short>& id,
573  const unsigned short threads, const double submit_time)
574  {
575  id_ = id;
576  threads_ = threads;
577  times_[0] = submit_time;
578  times_[1] = wall_time();
579  }
580 
581  /// Record the stop time of the task.
582  void stop() {
583  times_[2] = wall_time();
584  }
585 
586  /// Output the task data using a tab-separated list.
587 
588  /// Output information includes
589  /// - the ID pointer
590  /// - the function, member function, and object type name
591  /// - the number of threads used by the task
592  /// - the submit time
593  /// - the start time
594  /// - the stop time.
595  ///
596  /// \param[in,out] os The output stream.
597  /// \param[in] te The task event to be output.
598  /// \return The \c os reference.
599  friend std::ostream& operator<<(std::ostream& os, const TaskEvent& te) {
600  // Add address to output stream
601  os << std::hex << std::showbase << te.id_.first <<
602  std::dec << std::noshowbase << "\t";
603 
604  // Print the name
605  switch(te.id_.second) {
606  case 1:
607  {
608  const std::string mangled_name = te.get_name();
609 
610  // Print the demangled name
611  if(! mangled_name.empty())
612  print_demangled(os, mangled_name.c_str());
613  else
614  os << "UNKNOWN\t";
615  }
616  break;
617  case 2:
618  print_demangled(os, static_cast<const char*>(te.id_.first));
619  break;
620  default:
621  os << "UNKNOWN\t";
622  }
623 
624  // Print:
625  // # of threads, submit time, start time, stop time
626  os << te.threads_;
627  const std::streamsize precision = os.precision();
628  os.precision(6);
629  os << std::fixed << "\t" << te.times_[0]
630  << "\t" << te.times_[1] << "\t" << te.times_[2];
631  os.precision(precision);
632  return os;
633  }
