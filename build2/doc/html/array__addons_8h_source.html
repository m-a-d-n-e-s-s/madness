50  /// Output \c std::array to stream for human consumption.
51 
52  /// \tparam T The type of data stored in the array.
53  /// \tparam N The size of the array.
54  /// \param[in,out] s The output stream.
55  /// \param[in] a The array to be output.
56  /// \return The output stream.
57  template <typename T, std::size_t N>
58  std::ostream& operator<<(std::ostream& s, const std::array<T,N>& a) {
59  s << "[";
60  for(std::size_t i=0; i<N; ++i) {
61  s << a[i];
62  if (i != (N-1)) s << ",";
63  }
64  s << "]";
65  return s;
66  }
