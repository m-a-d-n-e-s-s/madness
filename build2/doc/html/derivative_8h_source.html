110  void forward_do_diff1(const implT* f, implT* df, const keyT& key,
111  const argT& left,
112  const argT& center,
113  const argT& right) const {
114 
115  const dcT& coeffs = f->get_coeffs();
116  ProcessID owner = coeffs.owner(key);
117 
118  if (owner == world.rank()) {
119  if (!left.second.has_data()) {
120  woT::task(owner, &madness::DerivativeBase<T,NDIM>::do_diff1,
121  f, df, key, find_neighbor(f, key,-1), center, right,
122  TaskAttributes::hipri());
123  }
124  else if (!right.second.has_data()) {
125  woT::task(owner, &madness::DerivativeBase<T,NDIM>::do_diff1,
126  f, df, key, left, center, find_neighbor(f, key,1),
127  TaskAttributes::hipri());
128  }
129  // Boundary node
130  else if (left.first.is_invalid() || right.first.is_invalid()) {
131  woT::task(owner, &madness::DerivativeBase<T,NDIM>::do_diff2b,
132  f, df, key, left, center, right);
133  }
134  // Interior node
135  else {
136  woT::task(owner, &madness::DerivativeBase<T,NDIM>::do_diff2i,
137  f, df, key, left, center, right);
138  }
139  }
140  else {
141  df->task(owner, &madness::FunctionImpl<T,NDIM>::forward_do_diff1,
142  this, f, key, left, center, right, TaskAttributes::hipri());
143  }
144  }

208  static bool enforce_bc(int bc_left, int bc_right, Level n, Translation& l) {
209  Translation two2n = 1ul << n;
210  if (l < 0) {
211  if (bc_left == BC_ZERO || bc_left == BC_FREE || bc_left == BC_DIRICHLET || bc_left == BC_ZERONEUMANN || bc_left == BC_NEUMANN) {
212  return false; // f=0 BC, or no BC, or nonzero f BC, or zero deriv BC, or nonzero deriv BC
213  }
214  else if (bc_left == BC_PERIODIC) {
215  l += two2n; // Periodic BC
216  MADNESS_ASSERT(bc_left == bc_right); //check that both BCs are periodic
217  }
218  else {
219  MADNESS_EXCEPTION("enforce_bc: confused left BC?",bc_left);
220  }
221  }
222  else if (l >= two2n) {
223  if (bc_right == BC_ZERO || bc_right == BC_FREE || bc_right == BC_DIRICHLET || bc_right == BC_ZERONEUMANN || bc_right == BC_NEUMANN) {
224  return false; // f=0 BC, or no BC, or nonzero f BC, or zero deriv BC, or nonzero deriv BC
225  }
226  else if (bc_right == BC_PERIODIC) {
227  l -= two2n; // Periodic BC
228  MADNESS_ASSERT(bc_left == bc_right); //check that both BCs are periodic
229  }
230  else {
231  MADNESS_EXCEPTION("enforce_bc: confused BC right?",bc_right);
232  }
233  }
234  return true;
235  }

237  Key<NDIM> neighbor(const keyT& key, int step) const {
238  Vector<Translation,NDIM> l = key.translation();
239  l[axis] += step;
240  if (!enforce_bc(bc(axis,0), bc(axis,1), key.level(), l[axis])) {
241  return keyT::invalid();
242  }
243  else {
244  return keyT(key.level(),l);

313  void do_diff2b(const implT* f, implT* df, const keyT& key,
314  const argT& left,
315  const argT& center,
316  const argT& right) const {
317  Vector<Translation,NDIM> l = key.translation();
318  double lev = (double) key.level();
319 
320  coeffT d;
321 
322  //left boundary
323  if (l[this->axis] == 0) {
324 
325  coeffT tensor_right=df->parent_to_child(right.second, right.first, this->neighbor(key,1));
326  coeffT tensor_center=df->parent_to_child(center.second, center.first, key);
327 
328  d= transform_dir(tensor_right,left_rmt,this->axis);
329  d+=transform_dir(tensor_center,left_r0t,this->axis);
330  }
331  else {
332 
333  coeffT tensor_left=df->parent_to_child(left.second, left.first, this->neighbor(key,-1));
334  coeffT tensor_center=df->parent_to_child(center.second, center.first, key);
335 
336  d= transform_dir(tensor_left,right_rpt,this->axis);
337  d+=transform_dir(tensor_center,right_r0t,this->axis);
338  }
339 
340  double fac = FunctionDefaults<NDIM>::get_rcell_width()[this->axis]*pow(2.0,lev);
341  if (is_second) fac *= fac;
342  else if (is_third) fac *= fac*fac;
343 
344  d.scale(fac);
345  d.reduce_rank(df->get_thresh());
346  df->get_coeffs().replace(key,nodeT(d,false));
347 
348 
349  // This is the boundary contribution (formally in BoundaryDerivative)
350  int bc_left = this->bc(this->axis,0);
351  int bc_right = this->bc(this->axis,1);
352 
353  Future<argT> found_argT;
354  tensorT bf, bdry_t;
355  //left boundary
356  if (l[this->axis] == 0) {
357  if (bc_left != BC_PERIODIC && bc_left != BC_FREE && bc_left != BC_ZERO && bc_left != BC_ZERONEUMANN) {
358  bf = copy(bv_left);
359  found_argT = g1.get_impl()->find_me(key);
360  }
361  else {
362  return;
363  }
364  }
365  else { //right boundary
366  if (bc_right != BC_PERIODIC && bc_right != BC_FREE && bc_right != BC_ZERO && bc_right != BC_ZERONEUMANN) {
367  bf = copy(bv_right);
368  found_argT = g2.get_impl()->find_me(key);
369  }
370  else {
371  return;
372  }
373  }
374  tensorT gcoeffs = df->parent_to_child(found_argT.get().second, found_argT.get().first,key).full_tensor_copy();
375 
376  //if (this->bc.get_bc().dim(0) == 1) {
377  if (NDIM == 1) {
378  bdry_t = gcoeffs[0]*bf;
379  }
380  else {
381  tensorT slice_aid(this->k); //vector of zeros
382  slice_aid[0] = 1;
383  tensorT tmp = inner(slice_aid, gcoeffs, 0, this->axis);
384  bdry_t = outer(bf,tmp);
385  if (this->axis) bdry_t = copy(bdry_t.cycledim(this->axis,0,this->axis)); // make it contiguous
386  }
387  bdry_t.scale(FunctionDefaults<NDIM>::get_rcell_width()[this->axis]);
388 
389  if (l[this->axis]==0) {
390  if (bc_left == BC_DIRICHLET)
391  bdry_t.scale( pow(2.0,lev));
392  else if (bc_left ==BC_NEUMANN)
393  bdry_t.scale(FunctionDefaults<NDIM>::get_cell_width()[this->axis]);
394  }
395  else {
396  if (bc_right == BC_DIRICHLET)
397  bdry_t.scale( pow(2.0,lev));
398  else if (bc_right ==BC_NEUMANN)
399  bdry_t.scale(FunctionDefaults<NDIM>::get_cell_width()[this->axis]);
400  }
401 
402  bdry_t += d.full_tensor_copy();;
403  df->get_coeffs().replace(key,nodeT(coeffT(bdry_t,df->get_thresh(),df->get_tensor_type()),false));
404  }

447  void initCoefficients() {
448  is_second = false;
449  is_third = false;
450 
451  r0 = Tensor<double>(this->k,this->k);
452  rp = Tensor<double>(this->k,this->k);
453  rm = Tensor<double>(this->k,this->k);
454 
455  left_rm = Tensor<double>(this->k,this->k);
456  left_r0 = Tensor<double>(this->k,this->k);
457 
458  right_r0 = Tensor<double>(this->k,this->k);
459  right_rp = Tensor<double>(this->k,this->k);
460 
461  // These are the coefficients for the boundary contribution
462  bv_left = Tensor<double>(this->k);
463  bv_right = Tensor<double>(this->k);
464 
465  int bc_left = this->bc(this->axis,0);
466  int bc_right = this->bc(this->axis,1);
467 
468  double kphase = -1.0;
469  if (this->k%2 == 0) kphase = 1.0;
470  double iphase = 1.0;
471  for (int i=0; i<this->k; ++i) {
472  double jphase = 1.0;
473  for (int j=0; j<this->k; ++j) {
474  double gammaij = sqrt(double((2*i+1)*(2*j+1)));
475  double Kij;
476  if (((i-j)>0) && (((i-j)%2)==1))
477  Kij = 2.0;
478  else
479  Kij = 0.0;
480 
481  r0(i,j) = 0.5*(1.0 - iphase*jphase - 2.0*Kij)*gammaij;
482  rm(i,j) = 0.5*jphase*gammaij;
483  rp(i,j) =-0.5*iphase*gammaij;
484 
485  // Constraints on the derivative
486  if (bc_left == BC_ZERONEUMANN || bc_left == BC_NEUMANN) {
487  left_rm(i,j) = jphase*gammaij*0.5*(1.0 + iphase*kphase/this->k);
488 
489  double phi_tmpj_left = 0;
490 
491  for (int l=0; l<this->k; ++l) {
492  double gammalj = sqrt(double((2*l+1)*(2*j+1)));
493  double Klj;
494 
495  if (((l-j)>0) && (((l-j)%2)==1)) Klj = 2.0;
496  else Klj = 0.0;
497 
498  phi_tmpj_left += sqrt(double(2*l+1))*Klj*gammalj;
499  }
500  phi_tmpj_left = -jphase*phi_tmpj_left;
501  left_r0(i,j) = (0.5*(1.0 + iphase*kphase/this->k) - Kij)*gammaij + iphase*sqrt(double(2*i+1))*phi_tmpj_left/pow(this->k,2.);
502  }
503  else if (bc_left == BC_ZERO || bc_left == BC_DIRICHLET || bc_left == BC_FREE) {
504  left_rm(i,j) = rm(i,j);
505 
506  // B.C. with a function
507  if (bc_left == BC_ZERO || bc_left == BC_DIRICHLET)
508  left_r0(i,j) = (0.5 - Kij)*gammaij;
509 
510  // No B.C.
511  else if (bc_left == BC_FREE)
512  left_r0(i,j) = (0.5 - iphase*jphase - Kij)*gammaij;
513  }
514 
515  // Constraints on the derivative
516  if (bc_right == BC_ZERONEUMANN || bc_right == BC_NEUMANN) {
517  right_rp(i,j) = -0.5*(iphase + kphase / this->k)*gammaij;
518 
519  double phi_tmpj_right = 0;
520  for (int l=0; l<this->k; ++l) {
521  double gammalj = sqrt(double((2*l+1)*(2*j+1)));
522  double Klj;
523  if (((l-j)>0) && (((l-j)%2)==1)) Klj = 2.0;
524  else Klj = 0.0;
525  phi_tmpj_right += sqrt(double(2*l+1))*Klj*gammalj;
526  }
527  right_r0(i,j) = -(0.5*jphase*(iphase+ kphase/this->k) + Kij)*gammaij + sqrt(double(2*i+1))*phi_tmpj_right/pow(this->k,2.);
528  }
529  else if (bc_right == BC_ZERO || bc_right == BC_FREE || bc_right == BC_DIRICHLET) {
530  right_rp(i,j) = rp(i,j);
531 
532  // Zero BC
533  if (bc_right == BC_ZERO || bc_right == BC_DIRICHLET)
534  right_r0(i,j) = -(0.5*iphase*jphase + Kij)*gammaij;
535 
536  // No BC
537  else if (bc_right == BC_FREE)
538  right_r0(i,j) = (1.0 - 0.5*iphase*jphase - Kij)*gammaij;
539 
540  }
541 
542  jphase = -jphase;
543  }
544  iphase = -iphase;
545  }
546 
547  // Coefficients for the boundary contributions
548  iphase = 1.0;
549  for (int i=0; i<this->k; ++i) {
550  iphase = -iphase;
551 
552  if (bc_left == BC_DIRICHLET)
553  bv_left(i) = iphase*sqrt(double(2*i+1)); // vector for left dirichlet BC
554  else if(bc_left == BC_NEUMANN)
555  bv_left(i) = -iphase*sqrt(double(2*i+1))/pow(this->k,2.); // vector for left deriv BC
556  else
557  bv_left(i) = 0.0;
558 
559  if (bc_right == BC_DIRICHLET)
560  bv_right(i) = sqrt(double(2*i+1)); // vector for right dirichlet BC
561  else if (bc_right == BC_NEUMANN)
562  bv_right(i) = sqrt(double(2*i+1))/pow(this->k,2.); // vector for right deriv BC
563  else
564  bv_right(i) = 0.0;
565  }
566 
567  r0t = transpose(r0);
568  rpt = transpose(rp);
569  rmt = transpose(rm);
570 
571  right_r0t = transpose(right_r0);
572  right_rpt = transpose(right_rp);
573 
574  left_rmt = transpose(left_rm);

650  void read_from_file(const std::string& filename, unsigned int order = 1) {
651 
652  Tensor<double> r0_bsp(this->k,this->k);
653  Tensor<double> rp_bsp(this->k,this->k);
654  Tensor<double> rm_bsp(this->k,this->k);
655 
656  std::ifstream f(filename);
657  bool found=false;
658 
659  for (int m; f >> m; ) {
660  if (m == this->k) {
661  for (int i=0; i<m; i++)
662  for (int j=0; j<m; j++)
663  MADNESS_ASSERT(f >> rp_bsp(i,j));
664  for (int i=0; i<m; i++)
665  for (int j=0; j<m; j++)
666  MADNESS_ASSERT(f >> r0_bsp(i,j));
667  for (int i=0; i<m; i++)
668  for (int j=0; j<m; j++)
669  MADNESS_ASSERT(f >> rm_bsp(i,j));
670  found = true;
671  break;
672  }
673  else {
674  double junk;
675  for (int i=0; i<3*m*m; i++)
676  MADNESS_ASSERT(f >> junk);
677  }
678  }
679  MADNESS_ASSERT(found);
680  Tensor<double> r0_bsp_t = transpose(r0_bsp);
681  Tensor<double> rp_bsp_t = transpose(rp_bsp);
682  Tensor<double> rm_bsp_t = transpose(rm_bsp);
683 
684  r0=r0_bsp; r0t=r0_bsp_t; left_r0=r0_bsp; left_r0t=r0_bsp_t; right_r0=r0_bsp; right_r0t=r0_bsp_t;
685 
686  rp=rp_bsp; rpt=rp_bsp_t; right_rp=rp_bsp; right_rpt=rp_bsp_t;
687 
688  rm=rm_bsp; rmt=rm_bsp_t; left_rm=rm_bsp; left_rmt=rm_bsp_t;
689 
690  // Get scaling factor right for higher order derivatives
691  if (order == 1) {
692  set_is_first();
693  }
694  else if(order == 2) {
695  set_is_second();
696  }
697  else if(order == 3) {
698  set_is_third();
699  }
700  }
