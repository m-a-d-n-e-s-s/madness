178  /// Evaluates the function at a point in user coordinates. Possible non-blocking comm.
179 
180  /// Only the invoking process will receive the result via the future
181  /// though other processes may be involved in the evaluation.
182  ///
183  /// Throws if function is not initialized.
184  Future<T> eval(const coordT& xuser) const {
185  PROFILE_MEMBER_FUNC(Function);
186  const double eps=1e-15;
187  verify();
188  MADNESS_ASSERT(!is_compressed());
189  coordT xsim;
190  user_to_sim(xuser,xsim);
191  // If on the boundary, move the point just inside the
192  // volume so that the evaluation logic does not fail
193  for (std::size_t d=0; d<NDIM; ++d) {
194  if (xsim[d] < -eps) {
195  MADNESS_EXCEPTION("eval: coordinate lower-bound error in dimension", d);
196  }
197  else if (xsim[d] < eps) {
198  xsim[d] = eps;
199  }
200 
201  if (xsim[d] > 1.0+eps) {
202  MADNESS_EXCEPTION("eval: coordinate upper-bound error in dimension", d);
203  }
204  else if (xsim[d] > 1.0-eps) {
205  xsim[d] = 1.0-eps;
206  }
207  }
208 
209  Future<T> result;
210  impl->eval(xsim, impl->key0(), result.remote_ref(impl->world));
211  return result;
212  }

317  /// Evaluates a cube/slice of points (probably for plotting) ... collective but no fence necessary
318 
319  /// All processes recieve the entire result (which is a rather severe limit
320  /// on the size of the cube that is possible).
321 
322  /// Set eval_refine=true to return the refinment levels of
323  /// the given function.
324 
325  /// @param[in] cell A Tensor describe the cube where the function to be evaluated in
326  /// @param[in] npt How many points to evaluate in each dimension
327  /// @param[in] eval_refine Wether to return the refinment levels of the given function
328  Tensor<T> eval_cube(const Tensor<double>& cell,
329  const std::vector<long>& npt,
330  bool eval_refine = false) const {
331  MADNESS_ASSERT(static_cast<std::size_t>(cell.dim(0))>=NDIM && cell.dim(1)==2 && npt.size()>=NDIM);
332  PROFILE_MEMBER_FUNC(Function);
333  const double eps=1e-14;
334  verify();
335  reconstruct();
336  coordT simlo, simhi;
337  for (std::size_t d=0; d<NDIM; ++d) {
338  simlo[d] = cell(d,0);
339  simhi[d] = cell(d,1);
340  }
341  user_to_sim(simlo, simlo);
342  user_to_sim(simhi, simhi);
343 
344  // Move the bounding box infintesimally inside dyadic
345  // points so that the evaluation logic does not fail
346  for (std::size_t d=0; d<NDIM; ++d) {
347  MADNESS_ASSERT(simhi[d] >= simlo[d]);
348  MADNESS_ASSERT(simlo[d] >= 0.0);
349  MADNESS_ASSERT(simhi[d] <= 1.0);
350 
351  double delta = eps*(simhi[d]-simlo[d]);
352  simlo[d] += delta;
353  simhi[d] -= 2*delta; // deliberate asymmetry
354  }
355  return impl->eval_plot_cube(simlo, simhi, npt, eval_refine);
356  }
