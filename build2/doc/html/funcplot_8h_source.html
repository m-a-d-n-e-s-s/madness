73  /// Writes the header information of a VTK file for plotting in an external
74  /// post-processing package (such as Paraview)
75  //
76  /// @param world World communicator
77  /// @param filename String containing the filename to export to
78  /// @param plotlo Vector of double values indicating the minimum coordinate to plot to in each dimension
79  /// @param plothi Vector of double values indicating the maximum coordinate to plot to in each dimension
80  /// @param npt Vector of long integers indicating the number of points to plot in each dimension
81  /// @param binary (optional) Boolean indicating whether to print in binary
82 
83  /// The VTK routines are also designed for SERIAL data, parallel coming...
84  ///
85  /// This header is templated by the dimension of the data.
86  ///
87  /// To plot with the plotvtk_* routines:
88  /// plotvtk_begin(...)
89  /// plotvtk_data(...)
90  /// plotvtk_data(...) ...
91  /// plotvtk_end(...)
92  ///
93  /// NOTE: Paraview expects the structured mesh points in a particular
94  /// order, which is why the LowDimIndexIterator is used...
95  template<std::size_t NDIM>
96  void plotvtk_begin(World &world, const char *filename,
97  const Vector<double, NDIM> &plotlo, const Vector<double, NDIM> &plothi,
98  const Vector<long, NDIM> &npt, bool binary = false) {
99 
100  PROFILE_FUNC;
101  MADNESS_ASSERT(NDIM>=1 && NDIM<=3); // how do we plot data in more than 3-D?
102 
103  Tensor<double> cell(NDIM, 2);
104  std::size_t i;
105  for(i = 0; i < NDIM; ++i) {
106  cell(i, 0) = plotlo[i];
107  cell(i, 1) = plothi[i];
108  }
109 
110  FILE *f=0;
111  if(world.rank() == 0) {
112  f = fopen(filename, "w");
113  if(!f)
114  MADNESS_EXCEPTION("plotvtk: failed to open the plot file", 0);
115 
116  fprintf(f, "<VTKFile type=\"StructuredGrid\" version=\"0.1\"" \
117  " byte_order=\"LittleEndian\" compressor=\"" \
118  "vtkZLibDataCompressor\">\n");
119  fprintf(f, " <StructuredGrid WholeExtent=\"");
120  for(i = 0; i < NDIM; ++i)
121  fprintf(f, "0 %ld ", npt[i]-1);
122  for(; i < 3; ++i)
123  fprintf(f, "0 0 ");
124  fprintf(f, "\">\n");
125  fprintf(f, " <Piece Extent=\"");
126  for(i = 0; i < NDIM; ++i)
127  fprintf(f, "0 %ld ", npt[i]-1);
128  for(; i < 3; ++i)
129  fprintf(f, "0 0 ");
130  fprintf(f, "\">\n");
131  fprintf(f, " <Points>\n");
132  fprintf(f, " <DataArray NumberOfComponents=\"3\" " \
133  "type=\"Float32\" format=\"ascii\">\n");
134 
135  Vector<double, NDIM> space;
136  for(i = 0; i < NDIM; ++i) {
137  if(npt[i] == 1)
138  space[i] = 0.0;
139  else
140  space[i] = (cell(i, 1) - cell(i, 0)) / (npt[i] - 1);
141  }
142 
143  // go through the grid
144  for(LowDimIndexIterator it(npt); it; ++it) {
145  for(i = 0; i < NDIM; ++i)
146  fprintf(f, "%f ", plotlo[i] + it[i]*space[i]);
147  for(; i < 3; ++i)
148  fprintf(f, "0.0 ");
149  fprintf(f, "\n");
150  }
151 
152  fprintf(f, " </DataArray>\n");
153  fprintf(f, " </Points>\n");
154  fprintf(f, " <PointData>\n");
155  fclose(f);
156  }
157  world.gop.fence();
158  }

326  /// Writes a Povray DF3 format file with a cube of points on a uniform grid
327 
328  /// Collective operation but only process 0 writes the file. By convention Povray
329  /// files end in ".df3" but this choice is up to the user. The dynamic range of
330  /// function values is mapped onto [0,1] and values stored in 16-bit fixed precision.
331  template <typename T>
332  static void plotpovray(const Function<T,3>& function,
333  const char* filename,
334  const Tensor<double>& cell = FunctionDefaults<3>::get_cell(),
335  const std::vector<long>& npt = std::vector<long>(3,201L))
336  {
337  using detail::htons_x;
338 
339  MADNESS_ASSERT(npt.size() == 3);
340  unsigned short dims[3] = {htons_x(npt[0]),htons_x(npt[1]),htons_x(npt[2])};
341 
342  World& world = const_cast< Function<T,3>& >(function).world();
343  FILE *f=0;
344  if (world.rank() == 0) {
345  f = fopen(filename, "w");
346  if (!f) MADNESS_EXCEPTION("plotdx: failed to open the plot file", 0);
347  fwrite((void*) dims, sizeof(short), 3, f);
348  }
349  Tensor<T> r = function.eval_cube(cell, npt);
350  if (world.rank() == 0) {
351  double rmax = r.max();
352  double rmin = r.min();
353  double rrange = rmax + rmin;
354  double rmean = rrange*0.5;
355  double fac = 65535.0/rrange;
356 
357  printf("plot_povray: %s: min=%.2e(0.0) mean=%.2e(0.5) max=%.2e(1.0) range=%.2e\n",
358  filename,rmin,rmean,rmax,rrange);
359 
360  std::vector<unsigned short> d(npt[0]);
361  for (unsigned int i2=0; i2<npt[2]; ++i2) {
362  for (unsigned int i1=0; i1<npt[1]; ++i1) {
363  for (unsigned int i0=0; i0<npt[0]; ++i0) {
364  d[i0] = (unsigned short)(htons_x((unsigned short)(fac*(r(i0,i1,i2) - rmin))));
365  //printf("%d\n",htons_x(d[i0]));
366  }
367  fwrite((void*) &d[0], sizeof(short), npt[0], f);
368  }
369  }
370 
371  fclose(f);
372  }
373  }

386  /// The ordinate is distance from lo
387  template <typename opT, std::size_t NDIM>
388  void plot_line(World& world, const char* filename, int npt, const Vector<double,NDIM>& lo,
389  const Vector<double,NDIM>& hi, const opT& op) {
390  typedef Vector<double,NDIM> coordT;
391  coordT h = (hi - lo)*(1.0/(npt-1));
392 
393  double sum = 0.0;
394  for (std::size_t i=0; i<NDIM; ++i) sum += h[i]*h[i];
395  sum = sqrt(sum);

397  if (world.rank() == 0) {
398  FILE* file = fopen(filename,"w");
399  if(!file)
400  MADNESS_EXCEPTION("plot_line: failed to open the plot file", 0);
401  for (int i=0; i<npt; ++i) {
402  coordT r = lo + h*double(i);
403  fprintf(file, "%.14e ", i*sum);
404  plot_line_print_value(file, op(r));
405  fprintf(file,"\n");
406  }
407  fclose(file);
