54  /// A simple, fixed dimension vector.
55 
56  /// This class eliminates memory allocation cost, is just POD (it can be
57  /// copied easily and allocated on the stack), and the known
58  /// dimension permits aggressive compiler optimizations.
59  ///
60  /// Provides additional mathematical and I/O operations.
61  /// \tparam T The type of data stored in the vector.
62  /// \tparam N The size of the vector.

644  /// \param[in] l The left-hand \c Vector.
645  /// \param[in] r The right-hand \c Vector.
646  /// \return the cross product
647  template <typename T, std::size_t N>
648  typename std::enable_if<N==3, Vector<T,N> >::type
649  cross(const Vector<T,N>& l, const Vector<T,N>& r) {
650  Vector<T,N> result;
651  result[0]=l[1]*r[2] - r[1]*l[2];
652  result[1]=l[2]*r[0] - r[2]*l[0];
653  result[2]=l[0]*r[1] - r[0]*l[1];
654  return result;
655  }

658  /// Factory function for creating a \c madness::Vector.
659 
660  /// Variadic templates are used to create factories that mimic
661  /// \code
662  /// inline madness::Vector<T, N> vec(T t1, ..., T tN) {
663  /// std::Vector<T, N> ret;
664  /// ret[0] = t1;
665  /// ...
666  /// ret[N-1] = tN;
667  /// return ret;
668  /// }
669  /// \endcode
670  ///
671  /// This function counts the number of arguments passed in through the
672  /// argument pack, creates a \c std::array of the appropriate size,
673  /// forwards the arguments to the `std::array`'s constructor, and passes
674  /// the \c std::array to \c madness::Vector.
675  ///
676  /// \deprecated This function has been replaced by the list-initialization
677  /// constructor and assignment operator. Rather than
678  /// \code
679  /// Vector<double, 3> v = vec(1.4, 2.5, 4.0);
680  /// \endcode
681  /// use
682  /// \code
683  /// Vector<double, 3> v{ 1.4, 2.5, 4.0 };
684  /// \endcode
685  /// or
686  /// \code
687  /// Vector<double, 3> v = { 1.4, 2.5, 4.0 };
688  /// \endcode
689  ///
690  /// \note The first argument is separated from the pack to prevent 0-size
691  /// arrays and also so that the caller doesn't have to explicitly
692  /// specify \c T. It is assumed that all arguments are of type \c T or
693  /// are convertible to type \c T.
694  ///
695  /// \tparam T The data type for the array.
696  /// \tparam Ts The argument pack; that is, the list of arguments. The
697  /// size of the resulting \c Vector is directly determined from the
698  /// size of the argument pack.
699  /// \param[in] t The first argument.
700  /// \param[in] ts The rest of the arguments.
701  /// \return The \c Vector with the arguments put into it.
702  template<typename T, typename... Ts>
703  inline Vector<T, sizeof...(Ts) + 1> vec(T t, Ts... ts) {
704  return Vector<T, sizeof...(Ts) + 1> {
705  std::array<T, sizeof...(Ts) + 1>
706  {{ t, static_cast<T>(ts)... }}
707  };
708  }

711  /// Construct a unit-`Vector` that has the same direction as \c r.
712 
713  /// \tparam T The type of data stored in the \c Vector.
714  /// \tparam N The size of the \c Vector.
715  /// \param[in] r The \c Vector.
716  /// \param[in] eps A precision. If the norm of \c r is less than \c eps,
717  /// the zero vector is returned.
718  /// \return The desired unit-`Vector` (unless \c r is numerically the zero
719  /// \c Vector).
720  template<typename T, std::size_t N>
721  Vector<T,N> unitvec(const Vector<T,N>& r, const double eps=1.e-6) {
722  const double norm=r.normf();
723  if(norm < eps)
724  return Vector<T,N>(0.0);
725  return r * (1.0/norm);
726  }
