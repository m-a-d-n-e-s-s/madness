49 /// Represents a single shell of contracted, Cartesian, Gaussian primitives
50 class ContractedGaussianShell {
51  int type; ///< Angular momentum = 0, 1, 2, ...
52  std::vector<double> coeff;
53  std::vector<double> expnt;
54  double rsqmax;
55  int numbf; ///< Number of basis functions in shell (type+1)*(type+2)/2
56 
57  void normalize() {
58  // nwcchem cartesian normalization conventions
59  // translation of nmcoeff.F into python and thence to c++
60  int np = coeff.size();
61  if (np == 1) coeff[0] = 1.0e0;
62 
63  double pi32=pow(madness::constants::pi,1.5);
64  int l_lim = 2*type - 1;
65  double f = 1.0e00;
66  for (int n=l_lim; n>1; n-=2) f *= n;
67 
68  for (int n=0; n<np; n++)
69  coeff[n] *= pow(2.e0*expnt[n]/madness::constants::pi,0.75e0)*pow(4.e0*expnt[n],0.5E0*type)/sqrt(f);
70 
71  double sum = 0.0;
72  for (int n1=0; n1<np; n1++) {
73  for (int n2=0; n2<np; n2++) {
74  double S =pi32/pow(expnt[n1]+expnt[n2],1.5e0+type)/pow(2e0,type);
75  sum = sum + coeff[n1]*coeff[n2]*S;
76  }
77  }
78  sum *= f;
79 
80  f = 1e0/sqrt(sum);
81  for (int n=0; n<np; n++) coeff[n] *= f;
82  }
83 
84 public:
85  ContractedGaussianShell()
86  : type(-1), coeff(), expnt(), rsqmax(0.0), numbf(0) {};
87 
88  ContractedGaussianShell(int type,
89  const std::vector<double>& coeff,
90  const std::vector<double>& expnt,
91  bool donorm=true)
92  : type(type), coeff(coeff), expnt(expnt), numbf((type+1)*(type+2)/2) {
93  if (donorm) normalize();
94  double minexpnt = expnt[0];
95  for (unsigned int i=1; i<expnt.size(); i++)
96  minexpnt = std::min(minexpnt,expnt[i]);
97  rsqmax = 18.4/minexpnt; // 18.4 = 8*ln(10)
98  }

101  /// Returns square of the distance beyond which function is less than 1e-8.
102  double rangesq() const {
103  return rsqmax;
104  }
105 
106 
107  /// Evaluates the radial part of the contracted function
108  double eval_radial(double rsq) const {
109  if (rsq > rsqmax) return 0.0;
110  double sum = 0.0;
111  for (unsigned int i=0; i<coeff.size(); i++) {
112  double ersq = expnt[i]*rsq;
113  if (ersq < 18.4) sum += coeff[i]*exp(-ersq);
114  }
115  return sum;
116  }
117 
118 
119  /// Evaluates the entire shell returning the incremented result pointer
120  double* eval(double rsq, double x, double y, double z, double* bf) const {
121  double R = eval_radial(rsq);
122  if (fabs(R) < 1e-8) {
123  for (int i=0; i<numbf; i++) bf[i] = 0.0;
124 
125  }
126  else {
127  switch (type) {
128  case 0:
129  bf[0] = R;
130  break;
131  case 1:
132  bf[0] = R*x;
133  bf[1] = R*y;
134  bf[2] = R*z;
135  break;
136  case 2:
137  bf[0] = R*x*x;
138  bf[1] = R*x*y;
139  bf[2] = R*x*z ;
140  bf[3] = R*y*y ;
141  bf[4] = R*y*z ;
142  bf[5] = R*z*z;
143  break;
144  case 3:
145  bf[0] = R*x*x*x;
146  bf[1] = R*x*x*y;
147  bf[2] = R*x*x*z;
148  bf[3] = R*x*y*y;
149  bf[4] = R*x*y*z;
150  bf[5] = R*x*z*z;
151  bf[6] = R*y*y*y;
152  bf[7] = R*y*y*z;
153  bf[8] = R*y*z*z;
154  bf[9] = R*z*z*z;
155  break;
156 
157  default:
158  throw "UNKNOWN ANGULAR MOMENTUM";
159  }
160  }
161  return bf+numbf;
162  }

227 /// Represents multiple shells of contracted gaussians on a single center
228 class AtomicBasis {
229  std::vector<ContractedGaussianShell> g;
230  double rmaxsq;
231  int numbf;
232  Tensor<double> dmat, avec, bvec;
233 
234 public:
235  AtomicBasis() : g(), rmaxsq(0.0), numbf(0) {};
236 
237  AtomicBasis(const std::vector<ContractedGaussianShell>& g)
238  : g(g) {
239  rmaxsq = 0.0;
240  numbf = 0;
241  for (unsigned int i=0; i<g.size(); i++) {
242  rmaxsq = std::max(rmaxsq, g[i].rangesq());
243  numbf += g[i].nbf();
244  }
245  }

254  /// Returns the number of basis functions on the center
255  int nbf() const {
256  return numbf;
257  }
258 
259  /// Returns the number of shells on the center
260  int nshell() const {
261  return g.size();
262  }
263 
264  /// Returns a const reference to the shells
265  const std::vector<ContractedGaussianShell>& get_shells() const {
266  return g;
267  };
268 
269  /// Evaluates the basis functions at point x, y, z relative to atomic center
270 
271  /// The array bf[] must be large enough to hold nbf() values.
272  ///
273  /// Returned is the incremented pointer.
274  double* eval(double x, double y, double z, double* bf) const {
275  double rsq = x*x + y*y + z*z;
276  if (rsq > rmaxsq) {
277  for (int i=0; i<numbf; i++) bf[i] = 0.0;
278  return bf+numbf;
279  }
280 
281  double* bfstart = bf;
282  for (unsigned int i=0; i<g.size(); i++) {
283  bf = g[i].eval(rsq, x, y, z, bf);
284  }
285  // paranoia is good
286  MADNESS_ASSERT(bf-bfstart == numbf);
287  return bf;
288  }
289 
290  /// Evaluates the guess atomic density at point x, y, z relative to atomic center
291  double eval_guess_density(double x, double y, double z) const {
292  MADNESS_ASSERT(has_guess_info());
293  double rsq = x*x + y*y + z*z;
294  if (rsq > rmaxsq) return 0.0;
295 
296  double bf[numbf];
297  eval(x, y, z, bf);
298  const double* p = dmat.ptr();
299  double sum = 0.0;
300  for (int i=0; i<numbf; i++, p+=numbf) {
301  double sumj = 0.0;
302  for (int j=0; j<numbf; ++j)
303  sumj += p[j]*bf[j];
304  sum += bf[i]*sumj;
305  }
306  return sum;
307  }

309  /// Return shell that contains basis function ibf and also return index of function in the shell
310  const ContractedGaussianShell& get_shell_from_basis_function(int ibf, int& ibf_in_shell) const {
311  int n=0;
312  for (unsigned int i=0; i<g.size(); i++) {
313  int nbf_in_shell = g[i].nbf();
314  if (ibf>=n && ibf<(n+nbf_in_shell)) {
315  ibf_in_shell = ibf-n;
316  return g[i];
317  }
318  else {
319  n += g[i].nbf();
320  }
321  }
322  MADNESS_EXCEPTION("AtomicBasis: get_shell_from_basis_function", ibf*100000 + nbf());
323  }
324 
325  bool has_guess_info() const {
326  return dmat.size()>0;
327  }

361 /// Used to represent one basis function from a shell on a specific center
362 class AtomicBasisFunction {
363 private:
364  const double xx, yy, zz; // Coordinates of the center
365  const ContractedGaussianShell& shell; // Reference to the underlying atomic shell
366  const int ibf; // Index of basis function in the shell (0, 1, ...)
367  const int nbf; // Number of functions in the shell
368 
369 public:
370  AtomicBasisFunction(double x, double y, double z,
371  const ContractedGaussianShell& shell, int ibf)
372  : xx(x), yy(y), zz(z), shell(shell), ibf(ibf), nbf(shell.nbf()) {}
373 
374 
375  AtomicBasisFunction(const AtomicBasisFunction& aofunc)
376  : xx(aofunc.xx)
377  , yy(aofunc.yy)
378  , zz(aofunc.zz)
379  , shell(aofunc.shell)
380  , ibf(aofunc.ibf)
381  , nbf(aofunc.nbf) {}
382 
383  double operator()(double x, double y, double z) const {
384  double bf[nbf];
385  x-=xx;
386  y-=yy;
387  z-=zz;
388  double rsq = x*x + y*y + z*z;
389  shell.eval(rsq, x, y, z, bf);
390  return bf[ibf];
391  }

399  }

407  }

409  void get_coords(double& x, double& y, double& z) const {
410  x=xx; y=yy; z=zz;
411  return;
412  }

416  }
417 };

463  void read_file(std::string filename) {
464  static const bool debug = false;
465  TiXmlDocument doc(filename);
466  if (!doc.LoadFile()) {
467  std::cout << "AtomicBasisSet: Failed loading from file " << filename
468  << " : ErrorDesc " << doc.ErrorDesc()
469  << " : Row " << doc.ErrorRow()
470  << " : Col " << doc.ErrorCol() << std::endl;
471  MADNESS_EXCEPTION("AtomicBasisSet: Failed loading basis set",0);
472  }
473  for (TiXmlElement* node=doc.FirstChildElement(); node; node=node->NextSiblingElement()) {
474  if (strcmp(node->Value(),"name") == 0) {
475  name = node->GetText();
476  if (debug) std::cout << "Loading basis set " << name << std::endl;
477  }
478  else if (strcmp(node->Value(), "basis") == 0) {
479  const char* symbol = node->Attribute("symbol");
480  if (debug) std::cout << " found basis set for " << symbol << std::endl;
481  int atn = symbol_to_atomic_number(symbol);
482  std::vector<ContractedGaussianShell> g;
483  for (TiXmlElement* shell=node->FirstChildElement(); shell; shell=shell->NextSiblingElement()) {
484  const char* type = shell->Attribute("type");
485  int nprim=-1;
486  shell->Attribute("nprim",&nprim);
487  if (debug) std::cout << " found shell " << type << " " << nprim << std::endl;
488  std::vector<double> expnt = load_tixml_vector<double>(shell, nprim, "exponents");
489  if (strcmp(type,"L") == 0) {
490  std::vector<double> scoeff = load_tixml_vector<double>(shell, nprim, "scoefficients");
491  std::vector<double> pcoeff = load_tixml_vector<double>(shell, nprim, "pcoefficients");
492  g.push_back(ContractedGaussianShell(0,scoeff,expnt));
493  g.push_back(ContractedGaussianShell(1,pcoeff,expnt));
494  }
495  else {
496  static const char* tag[] = {"S","P","D","F","G"};
497  int i;
498  for (i=0; i<5; i++) {
499  if (strcmp(type,tag[i]) == 0) goto foundit;
500  }
501  MADNESS_EXCEPTION("Loading atomic basis set: bad shell type?",0);
502 foundit:
503  std::vector<double> coeff = load_tixml_vector<double>(shell, nprim, "coefficients");
504  g.push_back(ContractedGaussianShell(i, coeff, expnt));
505  }
506  }
507  ag[atn] = AtomicBasis(g);
508  }
509  else if (strcmp(node->Value(), "atomicguess") == 0) {
510  const char* symbol = node->Attribute("symbol");
511  if (debug) std::cout << " atomic guess info for " << symbol << std::endl;
512  int atn = symbol_to_atomic_number(symbol);
513  MADNESS_ASSERT(is_supported(atn));
514  int nbf = ag[atn].nbf();
515  Tensor<double> dmat = load_tixml_matrix<double>(node, nbf, nbf, "guessdensitymatrix");
516  Tensor<double> avec = load_tixml_matrix<double>(node, nbf, nbf, "alphavectors");
517  Tensor<double> bvec = load_tixml_matrix<double>(node, nbf, nbf, "betavectors");
518  ag[atn].set_guess_info(dmat, avec, bvec);
519  }
520  else {
521  MADNESS_EXCEPTION("Loading atomic basis set: unexpected XML element", 0);
522  }
523  }

664  const double thresh = 0.2*v.normf();

666  printf(" zero vector\n");
667  return;
668  }
669  long nbf = int(v.dim[0]);
670  long list[nbf];
671  long ngot=0;
672  for (long i=0; i<nbf; i++) {
673  if (std::abs(v(i)) > thresh) {
674  list[ngot++] = i;
675  }
676  }
677  std::sort(list,list+ngot,AnalysisSorter<T>(v));
678 
679  const char* format;

681  format = " %2s(%1d)%4s(%2ld)%6.3f ";

684  format = " %2s(%2d)%4s(%3ld)%6.3f ";
685  }

687  format = " %2s(%3d)%4s(%4ld)%6.3f ";
688  }
689  else {
690  format = " %2s(%4d)%4s(%5ld)%6.3f ";
691  }
692  for (long ii=0; ii<ngot; ii++) {
693  long ibf = list[ii];
694

696  const Atom& atom = mentity.get_atom(iat);
697  const AtomicBasisFunction ao = get_atomic_basis_function(mentity, ibf);
698  const char* desc = ao.get_desc();
699  const char* element = get_atomic_data(atom.atomic_number).symbol;
700 
701  // This will need wrapping in a template for a complex MO vector
702  printf(format, element, iat, desc, ibf, v[ibf]);
703  }
