533  vector_real_function_3d localproj;
534  int lidx = 0;
535  unsigned int maxLL = 0;
536  // loop through all of the atom types in the molecule and get the maximum L value
537  // we need this because we are going to create a fixed sized tensor to store
538  // mapping between a linear index and (ias=which atom, i=which projector, l=angular momentum,
539  // m=angular momentum projection)
540  for (unsigned int iatom = 0; iatom < natoms; iatom++) {
541  // Get atom and its associated GTH tensors
542  Atom atom = molecule.get_atom(atoms_with_projectors[iatom]);
543  unsigned int atype = atom.atomic_number;
544  real_tensor& atom_radii = radii[atype-1];
545  if (atom_radii.dim(0) > 0)
546  maxLL = std::max(maxLL,(unsigned int)atom_radii.dim(0)-1);
547  }
548 
549  // Pilm_lookup is a mapping between a linear index and (ias,i,l,m)
550  Tensor<int> Pilm_lookup((unsigned int) natoms, (unsigned long) 3, (unsigned long) maxLL+1, (unsigned long) 2*maxLL+1);

566  else localproj.push_back(real_factory_3d(world));
567  }
568  }

571  world.gop.fence();

577  //truncate(world, vpsi, FunctionDefaults<3>::get_thresh());
578  compress(world, vpsi);

581  Pilm = Pilm.reshape(natoms, 3, maxLL+1, 2*maxLL+1, norbs);
582 
583  Tensor<Q> Qilm((unsigned int) natoms, (unsigned long) 3, (unsigned long) maxLL+1, (unsigned long) 2*maxLL+1, (unsigned int) norbs);
584  for (unsigned int iorb=0; iorb<psi.size(); iorb++) {
585  for (unsigned int iatom = 0; iatom < natoms; iatom++) {
586  // Get atom and its associated GTH tensors

588  unsigned int atype = atom.atomic_number;

596  for (unsigned int j = 1; j <= 3; j++) {
597  s += atom_hlij(l,i-1,j-1)*Pilm(iatom, j-1,l,m,iorb);
598  }
599  Qilm(iatom, i-1,l,m,iorb) = s;
600  }
601  }
602  }
603  }
604  }
605  Qilm = Qilm.reshape(natoms*3*(maxLL+1)*(2*maxLL+1),norbs);
606 
607  double vtol2 = 1e-4*thresh;
608  double trantol = vtol2 / std::min(30.0, double(localproj.size()));
609  vector_real_function_3d dpsi = transform(world, localproj, Qilm, trantol, true);
610 
611  // calculate non-local energy

613  int nocc = occ.size();
