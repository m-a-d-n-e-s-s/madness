100  void set_derived_values(const Molecule& molecule) {
101 
102  // auto determine freeze parameter for the first two shells
103  // deactivate by setting freeze 0 or any other value in the input file
104  size_t freeze = 0;
105  for(const Atom& atom: molecule.get_atoms()){
106  if (atom.atomic_number < 3){
107  // no frozen core for H and He
108  }else if(atom.atomic_number < 11){
109  freeze += 1;
110  }else if(atom.atomic_number < 19){
111  freeze += 5;
112  }
113  // beyond this point the parameter is better set manually
114  }
115  set_derived_value("freeze", freeze);
116 
117  set_derived_value("no_guess", get<std::string >("no_opt"));
118  set_derived_value("restart", get<std::string>("no_guess"));
119  set_derived_value("tpno_tight", 0.01*tpno());
120 
121  // set default values for adaptive solver
122  if(adaptive_solver()){
123  const std::string gt = "exop";
124  const std::string ex = "multipole";
125  set_derived_value("guesstype", gt);
126  set_derived_value("exop", ex);
127  set_derived_value("econv_macro", thresh());
128  set_derived_value("econv_micro", thresh());
129  set_derived_value("econv_pairs", 0.1*thresh());
130  }
131 
132  }

158  std::map<std::string, std::vector<int> >partial_wave(const std::string& key = "partial_wave")const {
159  // return format atom-name, vector of numbers giving S, P, D, ... functions
160  std::string str=get<std::string >(key);
161  std::transform(str.begin(), str.end(), str.begin(), ::tolower);
162  // madness parameter format does not allow blancs so we use '-' and transform them here
163  std::replace(str.begin(), str.end(), '-', ' ');
164  // expected input format: atom-name-3s2p1d atom-name-XsYpZd...
165  std::stringstream ss(str);
166  std::string symbol, pw_string;
167  std::map<std::string, std::vector<int> > result;
168  while(ss>>symbol){
169  ss>>pw_string;
170  std::vector<int> numbers(pw_string.size()/2);
171  std::vector<char> control = {'s', 'p', 'd', 'f', 'g', 'h', 'i', 'k'};
172  for (int i=0; i<pw_string.size()/2; ++i){
173  char l = pw_string[2*i+1];
174  char n = pw_string[2*i];
175  MADNESS_ASSERT(l==control[i]);
176  numbers[i]=(n-'0'); // need to convert ascii to int
177  }
178  result[symbol] = numbers;
179  }
180  return result;
181  }
