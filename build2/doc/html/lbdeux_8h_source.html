95  template <std::size_t NDIM>
96  class LBNodeDeux {
97  static const int nchild = (1<<NDIM);
98  typedef Key<NDIM> keyT;
99  typedef LBNodeDeux<NDIM> nodeT;
100  typedef WorldContainer<keyT,nodeT> treeT;
101  volatile double child_cost[nchild];
102  volatile double my_cost;
103  volatile double total_cost;
104  volatile bool gotkids;
105  AtomicInt nsummed;
106 
107  /// Computes index of child key in this node using last bit of translations
108  int index(const keyT& key) {
109  int ind = 0;
110  for (std::size_t d=0; d<NDIM; ++d) ind += ((key.translation()[d])&0x1) << d;
111  return ind;
112  }
113 
114  public:
115  LBNodeDeux()
116  : my_cost(0.0), total_cost(0.0), gotkids(false) {
117  nsummed = 0;
118  for (int i=0; i<nchild; ++i)
119  child_cost[i] = 0.0;
120  }
121 
122  LBNodeDeux(const LBNodeDeux<NDIM>& other) :
123  my_cost(other.my_cost), total_cost(other.total_cost), gotkids(other.gotkids)
124  {
125  nsummed = other.nsummed;
126  for (int i=0; i<nchild; ++i)
127  child_cost[i] = other.child_cost[i];
128  }
129 
130  LBNodeDeux<NDIM>& operator=(const LBNodeDeux<NDIM>& other) {
131  for (int i=0; i<nchild; ++i)
132  child_cost[i] = other.child_cost[i];
133  my_cost = other.my_cost;
134  total_cost = other.total_cost;
135  gotkids = other.gotkids;
136  nsummed = other.nsummed;
137 
138  return *this;
139  }
140 
141  bool has_children() const {
142  return gotkids;
143  }
144 
145  double get_total_cost() const {
146  return total_cost;
147  }
148 
149  /// Accumulates cost into this node
150  void add(double cost, bool got_kids) {
151  total_cost = (my_cost += cost);
152  gotkids = gotkids || got_kids;
153  }
154 
155  /// Accumulates cost up the tree from children
156  void sum(const treeT& tree, const keyT& child, double value) {
157  child_cost[index(child)] = value;
158  ++nsummed;
159  if (nsummed == nchild) {
160  for (int i=0; i<nchild; ++i) total_cost += child_cost[i];
161  if (child.level() > 1) {
162  keyT key = child.parent();
163  keyT parent = key.parent();
164  const_cast<treeT&>(tree).task(parent, &nodeT::sum, tree, key, double(total_cost));
165  }
166  }
167  }
168 
169 
170  /// Logically deletes this node by setting cost to -1
171 
172  /// Cannot actually erase this node from the container since the send() handler
173  /// is holding an accessor to it.
174  void deleter(const treeT& tree, const keyT& key) {
175  total_cost = my_cost = -1.0;
176  if (has_children()) {
177  for (KeyChildIterator<NDIM> kit(key); kit; ++kit) {
178  const keyT child = kit.key();
179  const_cast<treeT&>(tree).task(child, &nodeT::deleter, tree, child);
180  }
181  }
182  }
183 
184  /// Descends tree deleting all except internal nodes and sub-tree parents
185  void partition(const treeT& tree, const keyT& key, double avg) {
186  if (has_children()) {
187  // Sort children in descending cost order
188  keyT keys[nchild];
189  double vals[nchild];
190  for (KeyChildIterator<NDIM> kit(key); kit; ++kit) {
191  const keyT child = kit.key();
192  int ind = index(child);
193  keys[ind] = child;
194  vals[ind] = child_cost[ind];
195  }
196  for (int i=0; i<nchild; ++i) {
197  for (int j=i+1; j<nchild; ++j) {
198  if (vals[i] < vals[j]) {
199  std::swap(vals[i],vals[j]);
200  std::swap(keys[i],keys[j]);
201  }
202  }
203  }
204 
205  // Split off subtrees in decreasing cost order
206  for (int i=0; i<nchild; ++i) {
207  if (total_cost <= avg) {
208  const_cast<treeT&>(tree).task(keys[i], &nodeT::deleter, tree, keys[i]);
209  }
210  else {
211  total_cost -= vals[i];
212  const_cast<treeT&>(tree).task(keys[i], &nodeT::partition, tree, keys[i], avg);
213  }
214  }
215  }
216  }

218  template <typename Archive>
219  void serialize(Archive& ar) {
220  ar & archive::wrap_opaque(this,1);
221  }

236  template <typename T, typename costT>
237  struct add_op {
238  LoadBalanceDeux* lb;
239  const costT& costfn;
240  add_op(LoadBalanceDeux* lb, const costT& costfn) : lb(lb), costfn(costfn) {}
241  void operator()(const keyT& key, const FunctionNode<T,NDIM>& node) const {
242  if (lb->tree.is_local(key))
243  lb->tree.send(key, &nodeT::add, costfn(key,node), node.has_children());
244  else
245  lb->tree.task(key, &nodeT::add, costfn(key,node), node.has_children());
246  }
247  };
248 
249  /// Sums costs up the tree returning to everyone the total cost
250  double sum() {
251  world.gop.fence();
252  const_iteratorT end = tree.end();
253  for (const_iteratorT it=tree.begin(); it!=end; ++it) {
254  const keyT& key = it->first;
255  const nodeT& node = it->second;
256  if (!node.has_children() && key.level() > 0) {
257  tree.task(key.parent(), &nodeT::sum, tree, key, node.get_total_cost());
258  }
259  }
260  world.gop.fence();
261  double total;
262  keyT key0(0);
263  if (world.rank() == tree.owner(key0)) {
264  total = tree.find(key0).get()->second.get_total_cost();
265  }
266  world.gop.broadcast(total, tree.owner(key0));
267  world.gop.fence();
268 
269  return total;
270  }
271 
272  /// Used to sort results into descending order
273  static bool compare(const std::pair<keyT,double>& a, const std::pair<keyT,double>& b) {
274  return a.second < b.second;
275  }

307  struct CostPerProc {
308  double cost;
309  int proc;
310  CostPerProc() : cost(0.0), proc(0) {}
311  CostPerProc(double cost, int proc) : cost(cost), proc(proc) {}
312  bool operator<(const CostPerProc& other) const {
313  return cost > other.cost; // Want ascending order
314  }
315  };
316 
317  /// Actually does the partitioning of the tree
318  std::shared_ptr< WorldDCPmapInterface<keyT> > load_balance(double fac = 1.0, bool printstuff=false) {
319  world.gop.fence();
320  // Compute full tree of costs
321  double avg = sum()/(world.size()*fac);
322  //if (world.rank() == 0) print_tree();
323  world.gop.fence();
324 
325  // Create partitioning
326  keyT key0(0);
327  if (world.rank() == tree.owner(key0)) {
328  tree.send(key0, &nodeT::partition, tree, key0, avg*1.1);
329  }
330  world.gop.fence();
331 
332  // Collect entire vector onto node0
333  std::vector< std::pair<keyT,double> > results;
334  const_iteratorT end = tree.end();
335  for (const_iteratorT it=tree.begin(); it!=end; ++it) {
336  if (it->second.get_total_cost() >= 0) {
337  results.push_back(std::make_pair(it->first,it->second.get_total_cost()));
338  }
339  }
340  results = world.gop.concat0(results, 128*1024*1024);
341  world.gop.fence();
342 
343  std::vector< std::pair<keyT,ProcessID> > map;
344 
345  if (world.rank() == 0) {
346 
347  std::sort(results.begin(), results.end(), compare);
348  if (printstuff) {
349  print("THESE ARE THE INITIAL SUBTREES");
350  for (unsigned int i=0; i<results.size(); ++i) print(i,results[i]);
351  }
352 
353  // Now use bin packing to cram the results together
354  map.reserve(results.size());
355 
356  // Shove the first nproc entries directly into the queue
357  unsigned int nproc = world.size();
358  std::priority_queue<CostPerProc> costs;
359  for (unsigned int p=0; p<nproc && !results.empty(); ++p) {
360  const std::pair<keyT,double>& f = results.back();
361  costs.push(CostPerProc(f.second,p));
362  map.push_back(std::make_pair(f.first,p));
363  results.pop_back();
364  }
365 
366  // Process the remainder using the sorting maintained by the priority queue
367  while (!results.empty()) {
368  const std::pair<keyT,double>& f = results.back();
369  CostPerProc top = costs.top();
370  costs.pop();
371  top.cost += f.second;
372  costs.push(top);
373  map.push_back(std::make_pair(f.first,top.proc));
374  results.pop_back();
375  }
376  if (printstuff) {
377  print("THIS IS THE MAP");
378  print(map);
379  print("THESE ARE THE COSTS PER PROCESSOR");
380  while (!costs.empty()) {
381  print(costs.top().proc,costs.top().cost);
382  costs.pop();
383  }
384  }
385  }
386 
387  world.gop.fence();
388  world.gop.broadcast_serializable(map, 0);
389  world.gop.fence();
390 
391  // Return the Procmap
392 
393  return std::shared_ptr< WorldDCPmapInterface<keyT> >(new LBDeuxPmap<NDIM>(map));
394  }
395  };
