48  /*
49  The RMI layer just does transport and does not know about World
50  or even necessarily about MPI. It also has no buffering or
51  virtualization of resources. In particular, we must be careful
52  about having too many outstanding sends and active message
53  handlers must be careful about what they do in order to avoid
54  deadlock --- especially problematic is a handler trying to send
55  a message.
56 
57  The WorldAM class provides a World-aware RMI capability that
58  limits the number of outstanding sends and can optionally manage
59  buffers. The issue of what handlers can do safely is handled by
60  the integration of WorldAM and the task queues ... if you have
61  an operation that might
62 
63  - send messages
64 
65  - take a long time
66 
67  - consume a lot of stack/heap (e.g., recursive algorithm)
68 
69  then the right thing to do is to send a task rather than
70  an active message.
71  */

276  /// Sends a managed non-blocking active message
277  void send(ProcessID dest, am_handlerT op, const AmArg* arg,
278  const int attr=RMI::ATTR_ORDERED)
279  {
280  // Setup the header
281  {
282  AmArg* argx = const_cast<AmArg*>(arg);
283 
284  argx->set_worldid(worldid);
285  argx->set_src(rank);
286  argx->set_func(op);
287  argx->clear_flags(); // Is this the right place for this?
288  }
289 
290  // Sanity check
291  MADNESS_ASSERT(arg->get_world());
292  MADNESS_ASSERT(arg->get_func());
293 
294  // Map dest from world's communicator to comm_world
295  dest = map_to_comm_world[dest];
296 
297  // Remaining code refactored to avoid blocking with lock
298  // and to enable finer grained calls into MPI send
299 
300  if (RMI::get_this_thread_is_server()) {
301 
302  // The server thread may be executing a handler that
303  // is trying to send a message (e.g., assigning a
304  // remote future). However, the server must not block
305  // in here. It also needs to send messages in order,
306  // thus it puts all send_reqs onto a queue which it
307  // processes in its main loop using the RMI::send
308  // interface.
309 
310  lock(); nsent++; unlock(); // This world must still keep track of messages
311 
312  RMI::send_req.emplace_back(std::make_unique<SendReq>((AmArg*)(arg), RMI::isend(arg, arg->size()+sizeof(AmArg), dest, handler, attr)));
313 
314  //std::cout << "sending message from server " << (void*)(arg) << " " << pthread_self() << " " << p << std::endl;
315 
316  return;
317  }
318 
319 
320  // Find a free buffer oldest first (in order to assist
321  // with flow control). Exit loop with a lock on buffer.
322 
323  // This design will need nsend >= nthreads
324  int i=-1;
325  while (i == -1) {
326  lock(); // << Protect cur_msg and nsent;
327  if (send_req[cur_msg].try_lock()) { // << matching unlock at end of routine
328  i = cur_msg;
329  cur_msg = (cur_msg + 1) % nsend;
330  nsent++;
331  }
332  unlock(); // << Protect cur_msg and nsent;
333  }
334 
335 
336  // If the buffer is still in-use wait for it to complete
337  while (!send_req[i].TestAndFree()) {
338  // If the oldest message has still not completed then
339  // there is likely severe network or end-point
340  // congestion, so pause for 100us in a rather
341  // arbitrary attempt to decrease the injection rate.
342  // Both the server thread and this call to Test()
343  // should ensure progress.
344  myusleep(100);
345  }
346 
347  // Buffer is now free but still locked by me
348  send_req[i].set((AmArg*)(arg), RMI::isend(arg, arg->size()+sizeof(AmArg), dest, handler, attr));
349  send_req[i].unlock(); // << matches try_lock above
350  }
351 
352  /// Frees as many send buffers as possible, returning the number that are free
353  int free_managed_buffers() {
354  int nfree = 0;
355  for (int i=0; i<nsend; i++) {
356  if (send_req[i].try_lock()) { // Someone may be trying to put a message into this buffer
357  if (send_req[i].TestAndFree()) nfree++;
358  send_req[i].unlock(); // matching unlock
359  }
360  }
361  return nfree;
362  }
