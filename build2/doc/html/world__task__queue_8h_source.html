504 
505  /// The operation must provide the following interface, of
506  /// which the \c operator() methods are required by \c reduce()
507  /// and the rest by the task interface.
508  /// \code
509  /// struct opT {
510  /// opT();
511  /// opT(const &opT);
512  /// resultT operator()(const rangeT::iterator& it) const;
513  /// resultT operator()(const resultT& left, const resultT& right);
514  /// template <typename Archive> void serialize(const Archive& ar);
515  /// }
516  /// \endcode
517  /// \note The serialize method does not actually have to
518  /// work unless you want to have the task be stealable.
519  ///
520  /// Adjust the chunksize in the range to control granularity.
521  /// \todo Descriptions needed and/or verified.
522  /// \tparam resultT The result type of the operation.
523  /// \tparam rangeT Description needed.
524  /// \tparam opT Function type of the operation.

526  /// \param[in] op The operation.
527  /// \return Description needed.
528  template <typename resultT, typename rangeT, typename opT>
529  Future<resultT> reduce(const rangeT& range, const opT& op) {
530  if (range.size() <= range.get_chunksize()) {
531  resultT sum = resultT();
532  for (typename rangeT::iterator it=range.begin(); it != range.end(); ++it) sum = op(sum,op(it));
533  return Future<resultT>(sum);

539  Future<resultT> rightsum = add(*this, &WorldTaskQueue::reduce<resultT,rangeT,opT>, right, op);
540  return add(&WorldTaskQueue::sum<resultT,opT>, leftsum, rightsum, op);
