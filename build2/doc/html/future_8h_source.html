229  /// \param[in] arg Description needed.
230  static void set_handler(const AmArg& arg) {
231  RemoteReference< FutureImpl<T> > ref;
232  archive::BufferInputArchive input_arch = arg & ref;
233  // The remote reference holds a copy of the shared_ptr, so no need
234  // to take another.
235  {

237 
238  ScopedMutex<Spinlock> fred(pimpl);
239  if(pimpl->remote_ref) {

241  // be forwarded to another node.

246  // will invalidate it.

266  /// Invoked locally by set routine after assignment.
267  /// \todo Description needed.
268  /// \param[in] value Description needed.
269  inline void set_assigned(const T& value) {
270  // Assume that whoever is invoking this routine is holding
271  // a copy of our shared pointer on its *stack* so that
272  // if this future is destroyed as a result of a callback
273  // the destructor of this object is not invoked until
274  // we return.
275  //
276  // Also assume that the caller either has the lock
277  // or is sure that we are single threaded.
278  MADNESS_ASSERT(!assigned);
279  assigned = true;
280 
281  assignmentT& as = const_cast<assignmentT&>(assignments);
282  callbackT& cb = const_cast<callbackT&>(callbacks);
283 
284  while (!as.empty()) {
285  MADNESS_ASSERT(as.top());
286  as.top()->set(value);
287  as.pop();
288  }
289 
290  while (!cb.empty()) {
291  MADNESS_ASSERT(cb.top());
292  cb.top()->notify();
293  cb.pop();
294  }
295 
296  as.reset();
297  cb.reset();
298  }

594  /// \brief `A.set(B)`, where `A` and `B` are futures ensures `A`
595  /// has/will have the same value as `B`.
596 
597  /// An exception is thrown if `A` is already assigned since a
598  /// \c Future is a single assignment variable. We don't yet
599  /// track multiple assignments from unassigned futures.
600  ///
601  /// If `B` is already assigned, this is the same as `A.set(B.get())`,
602  /// which sets `A` to the value of `B`.
603  ///
604  /// If `B` has not yet been assigned, the behavior is to ensure
605  /// that, when `B` is assigned, both `A` and `B` will be assigned
606  /// and have the same value (though they may/may not refer to
607  /// the same underlying copy of the data and indeed may even
608  /// be in different processes).
609  /// \todo Verification needed in the param statement.
610  /// \param[in] other The future `B` described above. `*this` is `A`.
611  void set(const Future<T>& other) {
612  MADNESS_ASSERT(f);
613  if(f != other.f) {
614  MADNESS_ASSERT(! f->probe());
615  if (other.probe()) {
616  set(other.get()); // The easy case
617  } else {
618  // Assignment is supposed to happen just once so
619  // safe to assume that this is not being messed
620  // with ... also other might invoke the assignment
621  // callback since it could have been assigned
622  // between the test above and now (and this does
623  // happen)
624  std::shared_ptr< FutureImpl<T> > ff = f; // manage lifetime of me
625  std::shared_ptr< FutureImpl<T> > of = other.f; // manage lifetime of other
626 
627  { // BEGIN CRITICAL SECTION
628  ScopedMutex<Spinlock> fred(of.get());
629  of->add_to_assignments(ff); // Recheck of assigned is performed in here
630  } // END CRITICAL SECTION
631  }
632  }
633  }

726  /// This is used for passing pointers/references to another
727  /// process. To make remote references completely safe, the
728  /// \c RemoteReference increments the internal reference count of
729  /// the \c Future. The counter is decremented by either
730  /// assigning to the remote \c Future or its destructor if it is
