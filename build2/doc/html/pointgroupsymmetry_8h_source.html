37 /*!
38  \file chem/pointgroupsymmetry.cc
39  \brief implements point group operations
40 
41  The source is
42  <a href=http://code.google.com/p/m-a-d-n-e-s-s/source/browse/local
43  /trunk/src/apps/chem/pointgroupsymmetry.h>here</a>.
44 
45  \par Use cases
46  There are 3 use cases for the projector
47  - create_symmetry_adapted_basis()
48  Given a single function generate a set of symmetry-adapted functions of a given irrep
49  - operator()()
50  Given a set of functions transform them to a set of symmetry-adapted function.
51  This is most likely the case for a localized-to-symmetrized transformation of orbitals.
52  The irreps will be determined by the projection based on a rank-revealing Cholesky
53  decomposition. It is therefore necessary that the set of input functions is normalized,
54  so that small eigenvalues are always due to numerics and not to physics.
55  There will be a rough check on closedness, i.e. that the old and the new set of functions
56  project on the same space.
57  - project_on_irreps()
58  Given a set of functions and a set of corresponding irreps, reproject the functions on
59  their irreps
60  Furthermore a reduction method is provided to do some algebra with the irreps
61 
62 */

332  /// get a mapping canonical to irrep-sorting
333  std::vector<int> get_canonical_to_irrep_map(std::vector<std::string> sirreps) const {
334 
335  // distribute into bins
336  std::map<std::string,std::vector<int> > m;
337  for (size_t i=0; i<sirreps.size(); ++i) m[sirreps[i]].push_back(i);
338 
339  // concatenate bins
340  std::vector<int> map;
341  for (auto& irrep : table_.mullikan_) {
342  map.insert(end(map), begin(m[irrep]), end(m[irrep]));
343  }
344  return map;
345  }

347  /// given a mapping resort
348  template<typename R>
349  static std::vector<R> resort(const std::vector<int> map,
350  const std::vector<R>& vrhs) {
351  MADNESS_ASSERT(map.size()==vrhs.size());
352  std::vector<R> result(vrhs.size());
353  for (size_t i=0; i<map.size(); ++i) result[i]=vrhs[map[i]];
354  return result;
355  }
356 
357  /// given a mapping resort
358  template<typename R>
359  static std::vector<R> reverse_resort(const std::vector<int> map,
360  const std::vector<R>& vrhs) {
361  MADNESS_ASSERT(map.size()==vrhs.size());
362  std::vector<R> result(vrhs.size());
363  for (int i=0; i<map.size(); ++i) result[map[i]]=vrhs[i];
364  return result;
365  }
