164  /// fit the function exp(-mu r)/r
165 
166  /// formulas taken from
167  /// G. Beylkin and L. Monzon, On approximation of functions by exponential sums,
168  /// Appl Comput Harmon A, vol. 19, no. 1, pp. 17-48, Jul. 2005.
169  /// and
170  /// R. J. Harrison, G. I. Fann, T. Yanai, and G. Beylkin,
171  /// Multiresolution Quantum Chemistry in Multiwavelet Bases,
172  /// Lecture Notes in Computer Science, vol. 2660, p. 707, 2003.
173  static void bsh_fit(double mu, double lo, double hi, double eps,
174  Tensor<double>& pcoeff, Tensor<double>& pexpnt, bool prnt) {
175 
176  if (mu < 0.0) throw "cannot handle negative mu in bsh_fit";
177 
178  if (mu > 0) {
179  // Restrict hi according to the exponential decay
180  double r = -log(4*constants::pi*0.01*eps);
181  r = -log(r * 4*constants::pi*0.01*eps);
182  if (hi > r) hi = r;
183  }
184 
185  double TT;
186  double slo, shi;
187 
188  if (eps >= 1e-2) TT = 5;
189  else if (eps >= 1e-4) TT = 10;
190  else if (eps >= 1e-6) TT = 14;
191  else if (eps >= 1e-8) TT = 18;
192  else if (eps >= 1e-10) TT = 22;
193  else if (eps >= 1e-12) TT = 26;
194  else TT = 30;
195 
196  if (mu > 0) {
197  slo = -0.5*log(4.0*TT/(mu*mu));
198  }
199  else {
200  slo = log(eps/hi) - 1.0;
201  }
202  shi = 0.5*log(TT/(lo*lo));
203  if (shi <= slo) throw "bsh_fit: logic error in slo,shi";
204 
205  // Resolution required for quadrature over s
206  double h = 1.0/(0.2-.50*log10(eps)); // was 0.5 was 0.47
207 
208  // Truncate the number of binary digits in h's mantissa
209  // so that rounding does not occur when performing
210  // manipulations to determine the quadrature points and
211  // to limit the number of distinct values in case of
212  // multiple precisions being used at the same time.
213  h = floor(64.0*h)/64.0;
214 
215 
216  // Round shi/lo up/down to an integral multiple of quadrature points
217  shi = ceil(shi/h)*h;
218  slo = floor(slo/h)*h;
219 
220  long npt = long((shi-slo)/h+0.5);
221 
222  //if (prnt)
223  //std::cout << "mu " << mu << " slo " << slo << " shi " << shi << " npt " << npt << " h " << h << " " << eps << std::endl;
224 
225  Tensor<double> coeff(npt), expnt(npt);
226 
227  for (int i=0; i<npt; ++i) {
228  double s = slo + h*(npt-i); // i+1
229  coeff[i] = h*2.0/sqrt(constants::pi)*exp(-mu*mu*exp(-2.0*s)/4.0)*exp(s);
230  coeff[i] = coeff[i]/(4.0*constants::pi);
231  expnt[i] = exp(2.0*s);
232  }
233 
234 #if ONE_TERM
235  npt=1;
236  double s=1.0;
237  coeff[0]=1.0;
238  expnt[0] = exp(2.0*s);
239  coeff=coeff(Slice(0,0));
240  expnt=expnt(Slice(0,0));
241  print("only one term in gfit",s,coeff[0],expnt[0]);
242 
243 
244 #endif
245 
246  // Prune large exponents from the fit ... never necessary due to construction
247 
248  // Prune small exponents from Coulomb fit. Evaluate a gaussian at
249  // the range midpoint, and replace it there with the next most
250  // diffuse gaussian. Then examine the resulting error at the two
251  // end points ... if this error is less than the desired
252  // precision, can discard the diffuse gaussian.
253 
254  if (mu == 0.0) {
255  double mid = lo + (hi-lo)*0.5;
256  long i;
257  for (i=npt-1; i>0; --i) {
258  double cnew = coeff[i]*exp(-(expnt[i]-expnt[i-1])*mid*mid);
259  double errlo = coeff[i]*exp(-expnt[i]*lo*lo) -
260  cnew*exp(-expnt[i-1]*lo*lo);
261  double errhi = coeff[i]*exp(-expnt[i]*hi*hi) -
262  cnew*exp(-expnt[i-1]*hi*hi);
263  if (std::max(std::abs(errlo),std::abs(errhi)) > 0.03*eps) break;
264  npt--;
265  coeff[i-1] = coeff[i-1] + cnew;
266  }
267  coeff = coeff(Slice(0,npt-1));
268  expnt = expnt(Slice(0,npt-1));
269  }
270 
271  // Modify the coeffs of the largest exponents to satisfy the moment conditions
272  //
273  // SETTING NMOM>1 TURNS OUT TO BE A BAD IDEA (AS CURRENTLY IMPLEMENTED)
274  // [It is accurate and efficient for a one-shot application but it seems to
275  // introduce fine-scale noise that amplifies during iterative solution of
276  // the SCF and DFT equations ... the symptom is negative coeffs in the fit]
277  //
278  // SET NMOM=0 or 1 (1 recommended) unless you are doing a one-shot application
279  //
280  // Determine the effective range of the four largest exponents and compute
281  // moments of the exact and remainder of the fit. Then adjust the coeffs
282  // to reproduce the exact moments in that volume.
283  //
284  // If nmom!=4 we assume that we will eliminate n=-1 which is stored first
285  // in the moment list
286  //
287  // <r^i|gj> cj = <r^i|exact-remainder>
288  const long nmom = 1;
289  if (nmom > 0) {
290  Tensor<double> q(4), qg(4);
291  double range = sqrt(-log(1e-6)/expnt[nmom-1]);
292  if (prnt) print("exponent(nmom-1)",expnt[nmom-1],"has range", range);
293 
294  bsh_spherical_moments(mu, range, q);
295  Tensor<double> M(nmom,nmom);
296  for (int i=nmom; i<npt; ++i) {
297  Tensor<double> qt(4);
298  gaussian_spherical_moments(expnt[i], range, qt);
299  qg += qt*coeff[i];
300  }
301  if (nmom != 4) {
302  q = q(Slice(1,nmom));
303  qg = qg(Slice(1,nmom));
304  }
305  if (prnt) {
306  print("moments", q);
307  print("moments", qg);
308  }
309  q = q - qg;
310  for (int j=0; j<nmom; ++j) {
311  Tensor<double> qt(4);
312  gaussian_spherical_moments(expnt[j], range, qt);
313  if (nmom != 4) qt = qt(Slice(1,nmom));
314  for (int i=0; i<nmom; ++i) {
315  M(i,j) = qt[i];
316  }
317  }
318  Tensor<double> ncoeff;
319  gesv(M, q, ncoeff);
320  if (prnt) {
321  print("M\n",M);
322  print("old coeffs", coeff(Slice(0,nmom-1)));
323  print("new coeffs", ncoeff);
324  }
325 
326  coeff(Slice(0,nmom-1)) = ncoeff;
327  }
328 
329  if (prnt) {
330  for (int i=0; i<npt; ++i)
331  std::cout << i << " " << coeff[i] << " " << expnt[i] << std::endl;
332 
333  long npt = 300;
334  //double hi = 1.0;
335  //if (mu) hi = min(1.0,30.0/mu);
336  std::cout << " x value abserr relerr" << std::endl;
337  std::cout << " ------------ ------- -------- -------- " << std::endl;
338  double step = exp(log(hi/lo)/(npt+1));
339  for (int i=0; i<=npt; ++i) {
340  double r = lo*(pow(step,i+0.5));
341  double exact = exp(-mu*r)/r/4.0/constants::pi;
342  double test = 0.0;
343  for (int j=0; j<coeff.dim(0); ++j)
344  test += coeff[j]*exp(-r*r*expnt[j]);
345  double err = 0.0;
346  if (exact) err = (exact-test)/exact;
347  printf(" %.6e %8.1e %8.1e %8.1e\n",r, exact, exact-test, err);
348  }
349  }
350  pcoeff = coeff;
351  pexpnt = expnt;
352  }

354  /// fit a Slater function using a sum of Gaussians
355 
356  /// formula inspired by the BSH fit, with the roles of r and mu exchanged
357  /// see also Eq. (A3) in
358  /// S. Ten-no, Initiation of explicitly correlated Slater-type geminal theory,
359  /// Chem. Phys. Lett., vol. 398, no. 1, pp. 56-61, 2004.
360  static void slater_fit(double gamma, double lo, double hi, double eps,
361  Tensor<double>& pcoeff, Tensor<double>& pexpnt, bool prnt) {
362 
363  // empirical number TT for the upper integration limit
364  double TT;
365  if (eps >= 1e-2) TT = 5;
366  else if (eps >= 1e-4) TT = 10;
367  else if (eps >= 1e-6) TT = 14;
368  else if (eps >= 1e-8) TT = 18;
369  else if (eps >= 1e-10) TT = 22;
370  else if (eps >= 1e-12) TT = 26;
371  else TT = 30;
372 
373  // integration limits for quadrature over s: slo and shi
374  double slo=0.5 * log(eps) - 1.0;
375  double shi=log(TT/(lo*lo))*0.5;
376 
377  // Resolution required for quadrature over s
378  double h = 1.0/(0.2-.5*log10(eps)); // was 0.5 was 0.47
379 
380  // Truncate the number of binary digits in h's mantissa
381  // so that rounding does not occur when performing
382  // manipulations to determine the quadrature points and
383  // to limit the number of distinct values in case of
384  // multiple precisions being used at the same time.
385  h = floor(64.0*h)/64.0;
386 
387  // Round shi/lo up/down to an integral multiple of quadrature points
388  shi = ceil(shi/h)*h;
389  slo = floor(slo/h)*h;
390 
391  long npt = long((shi-slo)/h+0.5);
392 
393  Tensor<double> coeff(npt), expnt(npt);
394 
395  for (int i=0; i<npt; ++i) {
396  const double s = slo + h*(npt-i); // i+1
397  coeff[i] = h*exp(-gamma*gamma*exp(2.0*s) + s);
398  coeff[i]*=2.0*gamma/sqrt(constants::pi);
399  expnt[i] = 0.25*exp(-2.0*s);
400  }
401 
402  // Prune large exponents from the fit ... never necessary due to construction
403 
404  // Prune small exponents from Coulomb fit. Evaluate a gaussian at
405  // the range midpoint, and replace it there with the next most
406  // diffuse gaussian. Then examine the resulting error at the two
407  // end points ... if this error is less than the desired
408  // precision, can discard the diffuse gaussian.
409 
410  if (gamma == 0.0) {
411  double mid = lo + (hi-lo)*0.5;
412  long i;
413  for (i=npt-1; i>0; --i) {
414  double cnew = coeff[i]*exp(-(expnt[i]-expnt[i-1])*mid*mid);
415  double errlo = coeff[i]*exp(-expnt[i]*lo*lo) -
416  cnew*exp(-expnt[i-1]*lo*lo);
417  double errhi = coeff[i]*exp(-expnt[i]*hi*hi) -
418  cnew*exp(-expnt[i-1]*hi*hi);
419  if (std::max(std::abs(errlo),std::abs(errhi)) > 0.03*eps) break;
420  npt--;
421  coeff[i-1] = coeff[i-1] + cnew;
422  }
423  coeff = coeff(Slice(0,npt-1));
424  expnt = expnt(Slice(0,npt-1));
425  }
426 
427 
428  if (prnt) {
429  std::cout << "weights and roots for a Slater function with gamma=" << gamma << std::endl;
430  for (int i=0; i<npt; ++i)
431  std::cout << i << " " << coeff[i] << " " << expnt[i] << std::endl;
432 
433  long npt = 300;
434  //double hi = 1.0;
435  //if (mu) hi = min(1.0,30.0/mu);
436  std::cout << " x value abserr relerr" << std::endl;
437  std::cout << " ------------ ------- -------- -------- " << std::endl;
438  double step = exp(log(hi/lo)/(npt+1));
439  for (int i=0; i<=npt; ++i) {
440  double r = lo*(pow(step,i+0.5));
441  double exact = exp(-gamma*r);
442  double test = 0.0;
443  for (int j=0; j<coeff.dim(0); ++j)
444  test += coeff[j]*exp(-r*r*expnt[j]);
445  double err = 0.0;
446  if (exact) err = (exact-test)/exact;
447  printf(" %.6e %8.1e %8.1e %8.1e\n",r, exact, exact-test, err);
448  }
449  }
450  pcoeff = coeff;
451  pexpnt = expnt;
452  }

575  // Returns in q[0..4] int(r^2(n+1)*exp(-alpha*r^2),r=0..R) n=-1,0,1,2
576  static void gaussian_spherical_moments(double alpha, double R, Tensor<double>& q) {
577  q[0] = -(-0.1e1 + exp(-alpha * R*R)) / alpha / 0.2e1;
