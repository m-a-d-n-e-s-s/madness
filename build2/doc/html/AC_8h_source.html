187  /// Computes correction factor for exchange correlation potential
188  /// @param[in] r: position of the electron
189  /// @param[out] : correction factor for exchange correlation potential at position r
190  double operator ()(const Vector<double, NDIM> &r)const{
191 
192  // distance between electron and nuclues
193  double d = 0.0;
194 
195  bool inside_R1 = false;
196  std::vector<atom_information<NDIM> > between;
197 
198  // get position of electron relative to nuclei, to compute right correction factor
199  for(const auto& x:ac_param_.atoms_){
200  d = get_distance(r, x.coord);
201  // check if d<R1 for at least one nucleus
202  if(d<x.R1) {
203  inside_R1 = true;
204  break;
205  }
206  // push_back index of nucleus for which electron is inbetween
207  if(x.R1 < d and d < x.R2) {
208  between.push_back(x);
209  }
210  }
211 
212  // compute correction factor
213  // at least for one nucleus r<R1
214  if(inside_R1){
215  return 1.0;}
216  // all nuclei r>R2
217  else if(between.empty()){
218  return 0.0;}
219  // for N nuclei R1<r<R2
220  else {
221  return int_factor(r, between);
222  }
223  }
224 
225  /// Computes correction factor for the exchange correlation potential if electron position is between R1 and R2
226  /// @param[in] r: position of electron
227  /// @param[in] between: vector of the atom_information of all atoms for that electron position is between R1 and R2
228  /// @param[out] int_factor: correction factor
229  double int_factor(const Vector<double, NDIM> &r, std::vector<atom_information<NDIM> > between) const{
230  // Do not forget to add interpolation scheme to ACparameters check()
231  if(ac_param_.interpolation_scheme_=="constant"){
232  return 0.5;
233  }
234  else{
235  double int_factor = 0.0;
236  for(const auto& x:between){
237  // linear interpolation scheme between R1 and R2
238  int_factor += (- get_distance(r,x.coord) + x.R2)/(x.R2-x.R1);
239  //int_factor += 0.5*(erf(-(get_distance(r, x.coord)-5.0))+1.0);
240  }
241  int_factor = int_factor/(double(between.size()));
242  return int_factor;
243  }
244  }
