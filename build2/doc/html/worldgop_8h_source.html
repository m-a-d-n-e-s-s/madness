239  /// \tparam keyT The key type
240  /// \tparam valueT The value type

246  typedef detail::DistCache<keyT> dist_cache;
247 
248  if(world_.rank() == dest) {
249  dist_cache::set_cache_value(key, value);
250  } else {
251  // The destination is not this node, so send it to the destination.
252  if(value.probe()) {
253  // Spawn a remote task to set the value
254  world_.taskq.add(dest, dist_cache::template set_cache_value<valueT>, key,
255  value.get(), TaskAttributes::hipri());
256  } else {
257  // The future is not ready, so create a callback object that will
258  // send value to the destination node when it is ready.
259  DelayedSend<keyT, valueT>* delayed_send_callback =
260  new DelayedSend<keyT, valueT>(world_, dest, key, value);
261  const_cast<Future<valueT>&>(value).register_callback(delayed_send_callback);
262 
263  }

281  /// Lazy sync parent task
282 
283  /// Send signal to the child processes in the binary tree for a lazy
284  /// sync operation. After the signal has been sent to the children, the
285  /// sync operation, \c op, will be run.
286  /// \tparam keyT The key type
287  /// \tparam opT The sync operation type
288  /// \param child0 The first child process of this process in the binary tree
289  /// \param child1 The second child process of this process in the binary tree
290  /// \param key The key associated with the sync operation
291  /// \param op The sync operation that will be run
292  template <typename keyT, typename opT>
293  void lazy_sync_children(const ProcessID child0, const ProcessID child1,
294  const keyT& key, opT& op, const ProcessID) const
295  {
296  // Signal children to execute the operation.
297  if(child0 != -1)
298  send_internal(child0, key, 1);
299  if(child1 != -1)
300  send_internal(child1, key, 1);
301 
302  // Execute the operation on this process.
303  op();
304  }

306  /// Start a distributed lazy sync operation
307 
308  /// \param key The sync key
309  /// \param op The sync operation to be executed on this process
310  template <typename tagT, typename keyT, typename opT>
311  void lazy_sync_internal(const ProcessID parent, const ProcessID child0,
312  const ProcessID child1, const keyT& key, const opT& op) const {
313  typedef ProcessKey<keyT, tagT> key_type;
314 
315  // Get signals from parent and children.
316  madness::Future<ProcessID> child0_signal = (child0 != -1 ?
317  recv_internal<ProcessID>(key_type(key, child0)) :
318  madness::Future<ProcessID>(-1));
319  madness::Future<ProcessID> child1_signal = (child1 != -1 ?
320  recv_internal<ProcessID>(key_type(key, child1)) :
321  madness::Future<ProcessID>(-1));
322  madness::Future<ProcessID> parent_signal = (parent != -1 ?
323  recv_internal<ProcessID>(key_type(key, parent)) :
324  madness::Future<ProcessID>(-1));
325 
326  // Construct the task that notifies children to run the operation
327  key_type my_key(key, world_.rank());
328  auto lazy_sync_children_fn = & WorldGopInterface::template lazy_sync_children<key_type, opT>;
329  world_.taskq.add(*this, lazy_sync_children_fn,
330  child0_signal, child1_signal, my_key, op, parent_signal,
331  TaskAttributes::hipri());
332 
333  // Send signal to parent
334  if(parent != -1) {
335  if(child0_signal.probe() && child1_signal.probe())
336  send_internal(parent, my_key, world_.rank());
337  else {
338  auto lazy_sync_parent_fn = & WorldGopInterface::template lazy_sync_parent<key_type>;
339  world_.taskq.add(*this, lazy_sync_parent_fn,
340  parent, my_key, child0_signal, child1_signal,
341  TaskAttributes::hipri());
342  }
343  }
344  }

382  /// Broadcast task
383 
384  /// This task will set the local cache with the broadcast data and send
385  /// it to child processes in the binary tree.
386  template <typename keyT, typename valueT>
387  void bcast_task(const keyT& key, const valueT& value, const ProcessID root) const {
388  typedef void (WorldGopInterface::*taskfnT)(const keyT&, const valueT&,
389  const ProcessID) const;
390 
391  // Compute binary tree data
392  ProcessID parent = -1, child0 = -1, child1 = -1;
393  world_.mpi.binary_tree_info(root, parent, child0, child1);
394 
395  // Set the local data, except on the root process
396  if(parent != -1)
397  detail::DistCache<keyT>::set_cache_value(key, value);
398 
399  if(child0 != -1) { // Check that this process has children in the binary tree
400 
401  // Get handler function and arguments
402  void (*handler)(const AmArg&) =
403  & WorldGopInterface::template bcast_handler<keyT, valueT, taskfnT>;
404  AmArg* const args0 = new_am_arg(
405  & WorldGopInterface::template bcast_task<keyT, valueT>,
406  key, value, root);
407 
408  // Send active message to children
409  if(child1 != -1) {
410  AmArg* const args1 = copy_am_arg(*args0);
411  world_.am.send(child1, handler, args1);
412  }
413  world_.am.send(child0, handler, args0);
414  }
415  }

454  /// Broadcast data from the \c root process to all processes in \c world.
455  /// The input/output data is held by \c value.
456  /// \tparam tagT The tag type that is attached to \c keyT
457  /// \tparam keyT The base key type
458  /// \tparam valueT The value type that will be broadcast
459  /// \param[in] key The key associated with this broadcast
460  /// \param[in,out] value On the \c root process, this is used as the input
461  /// data that will be broadcast to all other processes in the group.
462  /// On other processes it is used as the output to the broadcast
463  /// \param root The process that owns the data to be broadcast
464  /// \throw madness::Exception When \c value has been set, except on the
465  /// \c root process.
466  template <typename tagT, typename keyT, typename valueT>
467  void bcast_internal(const keyT& key, Future<valueT>& value, const ProcessID root) const {
468  MADNESS_ASSERT((root >= 0) && (root < world_.size()));
469  MADNESS_ASSERT((world_.rank() == root) || (! value.probe()));
470 
471  // Add operation tag to key
472  typedef TaggedKey<keyT, tagT> key_type;
473  const key_type tagged_key(key);
474 
475  if(world_.size() > 1) { // Do nothing for the trivial case
476  if(world_.rank() == root) {
477  // This process owns the data to be broadcast.
478 
479  // Spawn remote tasks that will set the local cache for this
480  // broadcast on other nodes.
481  if(value.probe())
482  // The value is ready so send it now
483  bcast_task(tagged_key, value.get(), root);
484  else {
485  // The value is not ready so spawn a task to send the
486  // data when it is ready.
487  auto bcast_task_fn = & WorldGopInterface::template bcast_task<key_type, valueT>;
488  world_.taskq.add(*this, bcast_task_fn,
489  tagged_key, value, root, TaskAttributes::hipri());
490  }
491  } else {
492  MADNESS_ASSERT(! value.probe());
493 
494  // Get the broadcast value from local cache
495  detail::DistCache<key_type>::get_cache_value(tagged_key, value);
496  }
497  }
498  }

505  /// \tparam keyT The base key type

565  /// Distributed reduce
566 
567  /// \tparam tagT The tag type to be added to the key type
568  /// \tparam keyT The key type
569  /// \tparam valueT The data type to be reduced
570  /// \tparam opT The reduction operation type
571  /// \param key The key associated with this reduction
572  /// \param value The local value to be reduced
573  /// \param op The reduction operation to be applied to local and remote data
574  /// \param root The process that will receive the result of the reduction
575  /// \return A future to the reduce value on the root process, otherwise an
576  /// uninitialized future that may be ignored.
577  template <typename tagT, typename keyT, typename valueT, typename opT>
578  Future<typename detail::result_of<opT>::type>
579  reduce_internal(const ProcessID parent, const ProcessID child0,
580  const ProcessID child1, const ProcessID root, const keyT& key,
581  const valueT& value, const opT& op)
582  {
583  // Create tagged key
584  typedef ProcessKey<keyT, tagT> key_type;
585  typedef typename detail::result_of<opT>::type result_type;
586  typedef typename remove_future<valueT>::type value_type;
587  std::vector<Future<result_type> > results;
588  results.reserve(3);
589 
590  // Add local data to vector of values to reduce
591  results.push_back(world_.taskq.add(WorldGopInterface::template reduce_task<value_type, opT>,
592  value, op, TaskAttributes::hipri()));
593 
594  // Reduce child data
595  if(child0 != -1)
596  results.push_back(recv_internal<result_type>(key_type(key, child0)));
597  if(child1 != -1)
598  results.push_back(recv_internal<result_type>(key_type(key, child1)));
599 
600  // Submit the local reduction task
601  Future<result_type> local_result =
602  world_.taskq.add(WorldGopInterface::template reduce_result_task<opT>,
603  results, op, TaskAttributes::hipri());
604 
605  // Send reduced value to parent or, if this is the root process, set the
606  // result future.
607  if(parent == -1)
608  return local_result;
609  else
610  send_internal(parent, key_type(key, world_.rank()), local_result);
611 
612  return Future<result_type>::default_initializer();
613  }

920  /// conflict with other calls to \c lazy_sync. Keys may be reused after
