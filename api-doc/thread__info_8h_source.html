41namespace madness {
42
43 /// bitfield describing thread type
44 enum ThreadTag {
45 ThreadTag_NonMADNESS = 0b0000,
46 ThreadTag_MADNESS = 0b0001,
47 ThreadTag_Main = 0b0010
48 };
49
50 namespace detail {
51 inline ThreadTag& thread_tag_accessor() {
52 thread_local ThreadTag value = ThreadTag_NonMADNESS;
53 return value;
54 }
55 } // namespace madness::detail
56
57 /// sets the thread tag for the thread invoking this function
58 /// @param tag thread tag for the calling thread
59 inline void set_thread_tag(ThreadTag tag) {
60 detail::thread_tag_accessor() = tag;
61 }
62
63 /// sets the thread tag for the thread invoking this function
64 /// @param tag thread tag for the calling thread
65 inline void set_thread_tag(int tag) {
66 detail::thread_tag_accessor() = static_cast<ThreadTag>(tag);
67 }
68
69 /// @return true if this thread is used to execute MADNESS tasks; this is true for the main thread but only true for a non-main thread if the Pthreads backend is used and the thread is part of the ThreadPool
70 inline bool is_madness_thread() {
71 return detail::thread_tag_accessor() & ThreadTag_MADNESS;
72 }
73
74 /// @return true if this is called from the main thread
75 inline bool is_main_thread() {
76 return detail::thread_tag_accessor() & ThreadTag_Main;
77 }

79} // namespace madness
