191 /// like distsq() but accounts for periodicity
192 uint64_t
193 distsq_bc(const array_of_bools<NDIM>& is_periodic) const {
194 const Translation twonm1 = (Translation(1) << level()) >> 1;
195
196 uint64_t dsq = 0;
197 for (std::size_t d = 0; d < NDIM; ++d) {
198 Translation la = translation()[d];
199 if (is_periodic[d]) {
200 if (la > twonm1) {
201 la -= twonm1 * 2;
202 MADNESS_ASSERT(la <= twonm1);
203 }
204 if (la < -twonm1) {
205 la += twonm1 * 2;
206 MADNESS_ASSERT(la >= -twonm1);
207 }
208 }
209 dsq += la * la;
210 }
211
212 return dsq;
213 }

215 /// like "periodic" distsq() but only selects the prescribed axes
216 template <std::size_t NDIM2>
217 std::enable_if_t<NDIM >= NDIM2, uint64_t>
218 distsq_bc(const array_of_bools<NDIM>& is_periodic, const std::array<std::size_t, NDIM2>& axes) const {
219 const Translation twonm1 = (Translation(1) << level()) >> 1;
220
221 uint64_t dsq = 0;
222 for (std::size_t a = 0; a < NDIM2; ++a) {
223 const auto d = axes[a];
224 MADNESS_ASSERT(d < NDIM);
225 Translation la = translation()[d];
226 if (is_periodic[d]) {
227 if (la > twonm1) {
228 la -= twonm1 * 2;
229 MADNESS_ASSERT(la <= twonm1);
230 }
231 if (la < -twonm1) {
232 la += twonm1 * 2;
233 MADNESS_ASSERT(la >= -twonm1);
234 }
235 }
236 dsq += la * la;
237 }
238
239 return dsq;

242 /// Returns the key of the parent
243
244 /// Default is the immediate parent (generation=1). To get
245 /// the grandparent use generation=2, and similarly for
246 /// great-grandparents.
247 ///
248 /// !! If there is no such parent it quietly returns the
249 /// closest match (which may be self if this is the top of the
250 /// tree).
251 Key
252 parent(int generation = 1) const {
253 Vector<Translation, NDIM> pl;
254 if (generation > n)
255 generation = n;
256 for (std::size_t i = 0; i < NDIM; ++i)
257 pl[i] = l[i] >> generation;
258 return Key(n - generation, pl);
259 }

262 bool
263 is_child_of(const Key& key) const {
264 if (this->n < key.n) {
265 return false; // I can't be child of something lower on the tree
266 }
267 else if (this->n == key.n) {
268 return (*this == key); // I am child of myself
269 }
270 else {
271 Level dn = this->n - key.n;
272 Key mama = this->parent(dn);
273 return (mama == key);
274 }
275 }

283 /// Assuming keys are at the same level, returns true if displaced by no more than 1 in any direction
284
285 /// Assumes key and this are at the same level
286 bool
287 is_neighbor_of(const Key& key, const array_of_bools<NDIM>& bperiodic) const {
288 Translation dist = 0;
289 Translation TWON1 = (Translation(1)<<n) - 1;
290 for (std::size_t i=0; i<NDIM; ++i)
291 {
292 Translation ll = std::abs(l[i] - key.l[i]);
293 if (bperiodic[i] && ll==TWON1) ll=1;
294 dist = std::max(dist, ll);
295 }
296 return (dist <= 1);
297 }

318 /// check if this MultiIndex contains point x, disregarding these two dimensions
319 bool thisKeyContains(const Vector<double,NDIM>& x, const unsigned int& dim0,
320 const unsigned int& dim1) const {
321
322 // it's sufficient if one single dimension is out
323 bool contains=true;
324 const double twotoN = std::pow(2.0,double(n));
325 MADNESS_ASSERT(dim0<NDIM and dim1<NDIM);
326
327 for (unsigned int i=0; i<NDIM; i++ ) {
328
329 // check bounds
330 MADNESS_ASSERT((x[i]>=0.0) and (x[i]<=1.0));
331
332 // leave these two dimensions out
333 if ((i==dim0) or (i==dim1)) continue;
334
335 const int ll=int (x[i]*twotoN);
336 if (not (l[i]==ll)) contains=false;
337 }
338 return contains;
339 }

341 /// break key into two low-dimensional keys
342 template<std::size_t LDIM, std::size_t KDIM>
343 void break_apart(Key<LDIM>& key1, Key<KDIM>& key2) const {
344
345 // if LDIM==NDIM the 2nd key will be constructed empty
346 // static_assert((LDIM+KDIM==NDIM) or (LDIM==NDIM));
347 Vector<Translation, LDIM> l1;
348 Vector<Translation, KDIM> l2;
349 if constexpr((LDIM+KDIM==NDIM) or (LDIM==NDIM)) {
350 for (size_t i=0; i<LDIM; ++i) {
351 l1[i]=l[i];
352 }
353 for (size_t i=LDIM; i<NDIM; ++i) {
354 l2[i-LDIM]=l[i];
355 }
356
357 key1=Key<LDIM>(n,l1);
358 key2=Key<KDIM>(n,l2);
359 } else {
360 MADNESS_EXCEPTION("Key::break_apart: LDIM+KDIM must equal NDIM or LDIM==NDIM",1);
361 }
362 }

390 /// extract a new key with the Translations complementary to the ones indicated in the v array
391 template<std::size_t VDIM>
392 Key<NDIM-VDIM> extract_complement_key(const std::array<int,VDIM>& v) const {
393
394 // return the complement of v, e.g. v={0,1}, v_complement={2,3,4} if NDIM==5
395 // v must be contiguous and ascending (i.e. 1,2,3 or 2,3,4)
396 auto v_complement = [](const std::array<int, VDIM>& v) {
397 std::array<int, NDIM - VDIM> result;
398 for (std::size_t i = 0; i < NDIM - VDIM; i++) result[i] = (v.back() + i + 1) % NDIM;
399 return result;
400 };
401 return extract_key(v_complement(v));
402 };

404 /// merge with other key (ie concatenate), use level of rhs, not of this
405 template<std::size_t LDIM>
406 Key<NDIM+LDIM> merge_with(const Key<LDIM>& rhs) const {
407 Vector<Translation,NDIM+LDIM> t;

413 /// return if the other key is pointing in the same direction and is farther out
414
415 /// unlike in distsq() the direction is taken into account, and other must be
416 /// longer than this in each dimension
417 /// @param[in] other a key
418 /// @return if other is farther out
419 bool is_farther_out_than(const Key<NDIM>& other) const {
420 for (size_t i=0; i<NDIM; ++i) {
421 if ((other.translation()[i]>0) and (other.translation()[i]>l[i])) return false;

460 /// Iterates in lexical order thru all children of a key
461
462 /// Example usage:
463 /// \code
464 /// for (KeyChildIterator<NDIM> it(key); it; ++it) print(it.key());
465 /// \endcode
466 template<std::size_t NDIM>
467 class KeyChildIterator {
468 Key<NDIM> parent;
469 Key<NDIM> child;
470 Vector<Translation, NDIM> p;
471 bool finished;
472
473 public:
474 KeyChildIterator() :
475 p(0), finished(true) {
476 }
477
478 KeyChildIterator(const Key<NDIM>& parent) :
479 parent(parent), child(parent.n + 1, parent.l * 2), p(0),
480 finished(false) {
481 }
482
483 /// Pre-increment of an iterator (i.e., ++it)
484 KeyChildIterator&
485 operator++() {
486 if (finished)
487 return *this;
488 std::size_t i;
489 for (i = 0; i < NDIM; ++i) {
490 if (p[i] == 0) {
491 ++(p[i]);
492 ++(child.l[i]);
493 for (std::size_t j = 0; j < i; ++j) {
494 --(p[j]);
495 --(child.l[j]);
496 }
497 break;
498 }
499 }
500 finished = (i == NDIM);
501 child.rehash();
502 return *this;
503 }
504
505 /// True if iterator is not at end
506 operator bool() const {
507 return !finished;
508 }
