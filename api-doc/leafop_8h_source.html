58/// helper structure for the Leaf_op
59/// The class has an operator which decides if a given key belongs to a special box (needs further refinement up to a special level)
60/// this is the default class which always gives back false
61template<typename T, std::size_t NDIM>
62struct Specialbox_op {
63public:
64 Specialbox_op() {}
65
66 virtual ~Specialbox_op() {}
67
68 virtual std::string name() const { return "default special box which only checks for the special points"; }
69
70 /// the operator which deceides if the given box is special
71 /// @param[in] the key of the current box
72 /// @param[in] the function which will be constructed (if it is needed)
73 /// @param[out] bool which states if the box is special or not
74 virtual bool operator()(const Key<NDIM>& key, const FunctionImpl<T, NDIM> *const f = NULL) const { return false; }
75
76 // check if on of the special points is in the current box (or at low refinement level in a neighbouring box)
77 /// @param[in] the key of the box
78 /// @param[in] the function which will be constructed (if it is needed)
79 bool
80 check_special_points(const Key<NDIM>& key, const FunctionImpl<T, NDIM> *const f) const {
81 const std::vector<Vector<double, NDIM> >& special_points = f->get_special_points();
82 if (special_points.empty()) return false;
83 // level 0 and 1 has only boundary boxes
84 if (key.level() > 1 and box_is_at_boundary(key)) return false;
85 // check for all special points if they are neighbours of the current box
86 BoundaryConditions<NDIM> bc = FunctionDefaults<NDIM>::get_bc();
87 const auto bperiodic = bc.is_periodic();
88 for (size_t i = 0; i < special_points.size(); ++i) {
89 Vector<double, NDIM> simpt;
90 user_to_sim(special_points[i], simpt);
91 Key<NDIM> specialkey = simpt2key(simpt, key.level());
92 // use adaptive scheme: if we are at a low level refine also neighbours
93 int ll = get_half_of_special_level(f->get_special_level());
94 if (ll < f->get_initial_level()) ll = f->get_initial_level();
95 if (key.level() > ll) {
96 if (specialkey == key) return true;
97 else return false;
98 } else {
99 if (specialkey.is_neighbor_of(key, bperiodic)) return true;
100 else return false;
101 }
102 }
103 return false;
104 }

111 /// Determine if a given box is at the boundary of the simulation box
112 /// @param[in] the key of the current box
113 /// Since boxes of level 0 and 1 are always at the boundary this case should be avoided
114 /// if the boundary conditions are periodic then all boxes are boundary boxes
115 virtual bool box_is_at_boundary(const Key<NDIM>& key) const {
116 // l runs from l=0 to l=2^n-1 for every dimension
117 // if one l is = 2^n or 0 we are at the boundary
118 // note that boxes of level 0 and 1 are always boundary boxes
119 for (size_t i = 0; i < NDIM; i++) {
120 if (key.translation()[i] == 0 or key.translation()[i] == pow(2, key.level()) - 1) {
121 if (FunctionDefaults<NDIM>::get_bc()(i, 0) != BC_PERIODIC)return true;
122 }
123 }
124 return false;
125 }
126
127 // if you want to use milder refinement conditions for special boxes with increasing refinement level
128 // Cuspybox_op needs this
129 size_t get_half_of_special_level(const size_t& sl = FunctionDefaults<NDIM>::get_special_level()) const {
130 size_t ll = sl;
131 if (sl % 2 == 0) ll = sl / 2;
132 else ll = (sl + 1) / 2;
133 return ll;
134 }

138/// the high_dimensional key is broken appart into two lower dimensional keys, if they are neighbours of each other then refinement
139/// up to the special_level is enforced
140/// For general class description see the base class Specialbox_op
141template<typename T, std::size_t NDIM>
142struct ElectronCuspyBox_op : public Specialbox_op<T, NDIM> {
143public:
144 ElectronCuspyBox_op() {}
145
146 ~ElectronCuspyBox_op() {}
147
148 template<typename Archive>
149 void serialize(Archive& ar) {}
150
151 std::string name() const { return "Cuspybox_op"; }

153 /// Operator which decides if the key belongs to a special box
154 /// The key is broken apart in two lower dimensional keys (for electron cusps this is 6D -> 2x3D)
155 /// if the keys are neighbours then refinement up to the special level is enforced (means the 6D box is close to the cusp or contains it)
156 /// if the refinement level is already beyond half of the special_level then refinement is only enforded if the broken keys are the same (6D box contains cusp)
157 bool operator()(const Key<NDIM>& key, const FunctionImpl<T, NDIM> *const f) const {
158 // do not treat boundary boxes beyond level 2 as special (level 1 and 0 consist only of boundary boxes)
159 if (not(key.level() < 2)) {
160 if (this->box_is_at_boundary(key)) return false;
161 }
162
163 // Cuspybox_op only valid for even dimensions, if uneven dims are needed just make a new class with NDIM+1/2 and NDIM-LDIM
164 if constexpr (NDIM % 2 == 0) {
165 BoundaryConditions<NDIM> bc = FunctionDefaults<NDIM>::get_bc();
166 const auto bperiodic = bc.template is_periodic<NDIM / 2>();
167 Key<NDIM / 2> key1;
168 Key<NDIM / 2> key2;
169 key.break_apart(key1, key2);
170 int ll = this->get_half_of_special_level();
171 if (ll < f->get_initial_level())
172 ll = f->get_initial_level();
173 if (key.level() > ll) {
174 if (key1 == key2)
175 return true;
176 else
177 return false;
178 } else {
179 if (key1.is_neighbor_of(key2, bperiodic))
180 return true;
181 else
182 return false;
183 }
184 }
185 MADNESS_EXCEPTION("We should not end up here (check further of cuspy box)", 1);
186 return false;
187 }
188
189};

191/// This works similar to the Cuspybox_op: The key is broken apart (2N-dimensional -> 2x N-Dimensional)
192/// then it is checked if one of the lower dimensional keys contains a lower dimensional special points (which will be the nuclear-coordinates)
193template<typename T, std::size_t NDIM>
194struct NuclearCuspyBox_op : public Specialbox_op<T, NDIM> {
195public:
196 NuclearCuspyBox_op() : particle(-1) {
197 // failsafe
198 if (NDIM % 2 != 0) MADNESS_EXCEPTION("NuclearCuspyBox works only for even dimensions", 1);
199 }
200
201 NuclearCuspyBox_op(const size_t p) : particle(p) {
202 // failsafe
203 if (NDIM % 2 != 0) MADNESS_EXCEPTION("NuclearCuspyBox works only for even dimensions", 1);
204 MADNESS_ASSERT(particle == 1 or particle == 2 or particle == 0);
205 }
206
207 ~NuclearCuspyBox_op() {}
208
209 // particle==0: check both particles
210 // particle==1 or particle==2: check only particle1 or 2
211 int particle;
212
213 template<typename Archive>
214 void serialize(Archive& ar) { ar & particle; }
215
216 std::string name() const { return "Cuspybox_op for nuclear cusps"; }
217
218 /// Operator which decides if the key belongs to a special box
219 /// The key is broken appart in two lower dimensional keys (6D -> 2x3D)
220 /// The special points should be given in a format which can also be broken appart
221 /// so: 6D special_points = (3D-special-points, 3D-special-points)
222 /// if the refinement level is already beyond half of the special_level then refinement is only enforded if the broken keys are the same (6D box contains cusp)
223 bool operator()(const Key<NDIM>& key, const FunctionImpl<T, NDIM> *const f) const {
224 if (not(key.level() < 2)) {
225 if (this->box_is_at_boundary(key)) return false;
226 }
227 MADNESS_ASSERT(particle == 1 or particle == 2 or particle == 0);
228 if (f == NULL) MADNESS_EXCEPTION("NuclearCuspyBox: Pointer to function is NULL", 1);
229 const std::vector<Vector<double, NDIM> >& special_points = f->get_special_points();
230 if (special_points.empty()) MADNESS_EXCEPTION(
231 "Demanded NuclearCuspyBox but the special points of the function are empty", 1);
232
233 // break the special points into 3D points
234 std::vector<Vector<double, NDIM / 2> > lowdim_sp;
235 for (size_t i = 0; i < special_points.size(); i++) {
236 Vector<double, NDIM / 2> lowdim_tmp;
237 for (size_t j = 0; j < NDIM / 2; j++) {
238 lowdim_tmp[j] = special_points[i][j];
239 // check if the formatting is correct
240 if (special_points[i][j] != special_points[i][NDIM / 2 + j]) MADNESS_EXCEPTION(
241 "NuclearCuspyBox: Wrong format of special_point: ", 1);
242 }
243 lowdim_sp.push_back(lowdim_tmp);
244 }
245
246 // now break the key appart and check if one if the results is in the neighbourhood of a special point
247 BoundaryConditions<NDIM / 2> bc = FunctionDefaults<NDIM / 2>::get_bc();
248 const auto bperiodic = bc.is_periodic();
249 Key<NDIM / 2> key1;
250 Key<NDIM / 2> key2;
251
252 key.break_apart(key1, key2);
253 for (size_t i = 0; i < lowdim_sp.size(); ++i) {
254 Vector<double, NDIM / 2> simpt;
255 user_to_sim(lowdim_sp[i], simpt);
256 Key<NDIM / 2> specialkey = simpt2key(simpt, key1.level());
257 // use adaptive scheme: if we are at a low level refine also neighbours
258 int ll = this->get_half_of_special_level(f->get_special_level());
259 if (ll < f->get_initial_level()) ll = f->get_initial_level();
260 if (key.level() > ll) {
261 if (particle == 1 and specialkey == key1) return true;
262 else if (particle == 2 and specialkey == key2) return true;
263 else if (particle == 0 and (specialkey == key1 or specialkey == key2)) return true;
264 else return false;
265 } else {
266 if (particle == 1 and specialkey.is_neighbor_of(key1, bperiodic)) return true;
267 else if (particle == 2 and specialkey.is_neighbor_of(key2, bperiodic)) return true;
268 else if (particle == 0 and
269 (specialkey.is_neighbor_of(key1, bperiodic) or specialkey.is_neighbor_of(key2, bperiodic)))
270 return true;
271 else return false;
272 }
273 }
274 return false;
275 }

280template<typename T, std::size_t NDIM, typename opT, typename specialboxT>

331public:
332 /// pre-determination
333 /// the decision if the current box will be a leaf box is made from information of another function
334 /// this is needed for on demand function
335 /// not that the on-demand function that is being constructed is not the function in this class
336 virtual bool

338 MADNESS_EXCEPTION("pre-screening was called for leaf_op != leaf_op_other", 1);
339 return false;
340 }
341
342 /// post-determination
343 /// determination if the box will be a leaf box after the coefficients are made
344 /// the function that is being constructed is the function in this class
345 /// the function will use the opartor op in order to screen, if op is a NULL pointer the result is always: false
346 /// @param[in] key: the key to the current box
347 /// @param[in] coeff: Coefficients of the current box
348 virtual bool
349 post_screening(const Key<NDIM>& key, const GenTensor<T>& coeff) const {
350 if (op == NULL) return false;
351 if (key.level() < this->f->get_initial_level()) return false;
352 sanity();
353 const double cnorm = coeff.normf();
354 BoundaryConditions<NDIM> bc = FunctionDefaults<NDIM>::get_bc();
355 // const auto bperiodic = bc.is_periodic();
356
357 typedef Key<opT::opdim> opkeyT;
358 const opkeyT source = op->get_source_key(key);
359
360 const double thresh = (this->f->truncate_tol(this->f->get_thresh(), key));
361 const std::vector<opkeyT>& disp = op->get_disp(key.level());
362 const opkeyT& d = *disp.begin(); // use the zero-displacement for screening
363 const double opnorm = op->norm(key.level(), d, source);
364 const double norm = opnorm * cnorm;
365
366 return norm < thresh;
367 }

369 /// determines if a node is well represented compared to the parents
370 /// @param[in] key the FunctionNode which we want to determine if it's a leaf node
371 /// @param[in] coeff the coeffs of key
372 /// @param[in] parent the coeffs of key's parent node
373 /// @return is the FunctionNode of key a leaf node?
374 bool
375 compare_to_parent(const Key<NDIM>& key, const GenTensor<T>& coeff, const GenTensor<T>& parent) const {
376 if (key.level() < this->f->get_initial_level()) return false;
377 //this->sanity();
378 if (parent.has_no_data()) return false;
379 if (key.level() < this->f->get_initial_level()) return false;
380 GenTensor<T> upsampled = this->f->upsample(key, parent);
381 upsampled.scale(-1.0);
382 upsampled += coeff;
383 const double dnorm = upsampled.normf();
384 const bool is_leaf = (dnorm < this->f->truncate_tol(this->f->get_thresh(), key.level()));
385 return is_leaf;
386 }

388 /// check if the box is a special box
389 /// first check: Is one of the special points defined in the function f in the current box or a neighbour box
390 /// second check: Is the box special according to the specialbox operator (see class description of Specialbox_op)
391 /// @param[in] the key of the box
392 bool
393 special_refinement_needed(const Key<NDIM>& key) const {
394 if (key.level() > f->get_special_level()) return false;
395 else if (specialbox.check_special_points(key, f)) return true;
396 else if (specialbox(key, f)) return true;
397 else return false;
398 }
