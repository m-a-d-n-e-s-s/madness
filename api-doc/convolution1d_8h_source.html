161 /// !!! Note that if Rnormf is zero then ***ALL*** of the tensors are empty
162 template <typename Q>
163 struct ConvolutionData1D {
164
165
166 Tensor<Q> R, T; ///< if NS: R=ns, T=T part of ns; if modified NS: T=\uparrow r^(n-1)
167 Tensor<Q> RU, RVT, TU, TVT; ///< SVD approximations to R and T
168 Tensor<typename Tensor<Q>::scalar_type> Rs, Ts; ///< hold relative errors, NOT the singular values..
169
170 // norms for NS form
171 double Rnorm, Tnorm, Rnormf, Tnormf, NSnormf;
172
173 // norms for modified NS form
174 double N_up, N_diff, N_F; ///< the norms according to Beylkin 2008, Eq. (21) ff
175
176
177 /// ctor for NS form
178 /// make the operator matrices r^n and \uparrow r^(n-1)
179 /// @param[in] R operator matrix of the requested level; NS: unfilter(r^(n+1)); modified NS: r^n
180 /// @param[in] T upsampled operator matrix from level n-1; NS: r^n; modified NS: filter( r^(n-1) )
181 ConvolutionData1D(const Tensor<Q>& R, const Tensor<Q>& T) : R(R), T(T) {
182 Rnormf = R.normf();
183 // Making the approximations is expensive ... only do it for
184 // significant components
185 if (Rnormf > 1e-20) {
186 Tnormf = T.normf();
187 make_approx(T, TU, Ts, TVT, Tnorm);
188 make_approx(R, RU, Rs, RVT, Rnorm);
189 int k = T.dim(0);
190
191 Tensor<Q> NS = copy(R);
192 for (int i=0; i<k; ++i)
193 for (int j=0; j<k; ++j)
194 NS(i,j) = 0.0;
195 NSnormf = NS.normf();
196
197 }
198 else {
199 Rnorm = Tnorm = Rnormf = Tnormf = NSnormf = 0.0;
200 N_F = N_up = N_diff = 0.0;
201 }
202 }
203
204 /// ctor for modified NS form
205 /// make the operator matrices r^n and \uparrow r^(n-1)
206 /// @param[in] R operator matrix of the requested level; NS: unfilter(r^(n+1)); modified NS: r^n
207 /// @param[in] T upsampled operator matrix from level n-1; NS: r^n; modified NS: filter( r^(n-1) )
208 /// @param[in] modified use (un) modified NS form
209 ConvolutionData1D(const Tensor<Q>& R, const Tensor<Q>& T, const bool modified) : R(R), T(T) {
210
211 // note that R can be small, but T still be large
212
213 Rnormf = R.normf();
214 Tnormf = T.normf();
215 // Making the approximations is expensive ... only do it for
216 // significant components
217 if (Rnormf > 1e-20) make_approx(R, RU, Rs, RVT, Rnorm);
218 if (Tnormf > 1e-20) make_approx(T, TU, Ts, TVT, Tnorm);
219
220 // norms for modified NS form: follow Beylkin, 2008, Eq. (21) ff
221 N_F=Rnormf;
222 N_up=Tnormf;
223 N_diff=(R-T).normf();
224 }

228 /// approximate the operator matrices using SVD, and abuse Rs to hold the error instead of
229 /// the singular values (seriously, who named this??)
230 void make_approx(const Tensor<Q>& R,
231 Tensor<Q>& RU, Tensor<typename Tensor<Q>::scalar_type>& Rs, Tensor<Q>& RVT, double& norm) {
232 int n = R.dim(0);
233 svd(R, RU, Rs, RVT);
234 for (int i=0; i<n; ++i) {
235 for (int j=0; j<n; ++j) {
236 RVT(i,j) *= Rs[i];
237 }
238 }
239 for (int i=n-1; i>1; --i) { // Form cumulative sum of norms
240 Rs[i-1] += Rs[i];
241 }
242
243 norm = Rs[0];
244 if (Rs[0]>0.0) { // Turn into relative errors
245 double rnorm = 1.0/norm;
246 for (int i=0; i<n; ++i) {
247 Rs[i] *= rnorm;
248 }
249 }
250 }
251 };

316 /// @return true if the block of [r^n_l]_ij is expected to be small
317 /// @note unlike issmall(), this handles periodicity and range restriction
318 bool get_issmall(Level n, Translation lx) const {
319 // issmall modufulated by range restriction
320 auto is_small = [this,n](Translation l) {
321 if (!range_restricted())
322 return issmall(n, l);
323 else {
324 // [r^n_l]_ij = superposition of [r^n_l]_p and [r^n_l-1]_p
325 // so rnlij is out of the range only if both rnlp contributions are
326 const auto closest_l = l<=0 ? l : l-1;
327 return rnlp_is_zero(n, closest_l) || issmall(n, l);
328 }
329 };
330
331 // handle lattice summation, if needed
332 if (lattice_summed()) {
333 const Translation twon = Translation(1) << n;
334 for (int R = -maxR; R <= maxR; ++R) {
335 if (!is_small(R * twon + lx))
336 return false;
337 }
338 return true;
339 } else { // !lattice_summed
340 return is_small(lx);
341 }
342 }

344 /// @return true if `[r^n_l]` is zero due to range restriction
345 bool rnlp_is_zero(Level n, Translation l) const {
346 bool result = false;
347 if (range_restricted()) {
348 if (n == 0) {
349 // result = l > 0 || l < -1;
350 if (l >= 0)
351 result = Translation(range.iextent_x2()) <= 2*l;
352 else
353 result = Translation(range.iextent_x2()) <= 2*(-l-1);
354 } else { // n > 0
355 if (l >= 0)
356 result = (1 << (n - 1)) * Translation(range.iextent_x2()) <= l;
357 else
358 result = ((1 << (n - 1)) * Translation(range.iextent_x2())) <= (-l-1);
359 }
360 }
361 return result;
362 }

367 /// Computes the transition matrix elements for the convolution for n,l
368
369 /// Returns the tensor
370 /// \code
371 /// r(i,j) = int(K(x-y) phi[n0](x) phi[nl](y), x=0..1, y=0..1)
372 /// \endcode
373 /// This is computed from the matrix elements over the correlation
374 /// function which in turn are computed from the matrix elements
375 /// over the double order legendre polynomials.
376 /// \note if `this->range_restricted()==true`, `(D/2 - |x-y|) K(x-y)` is used as the kernel
377 const Tensor<Q>& rnlij(Level n, Translation lx, bool do_transpose=false) const {
378 const Tensor<Q>* p=rnlij_cache.getptr(n,lx);
379 if (p) return *p;
380
381 // PROFILE_MEMBER_FUNC(Convolution1D); // Too fine grain for routine profiling
382
383 long twok = 2*k;
384 Tensor<Q> R(2*twok);
385 R(Slice(0,twok-1)) = get_rnlp(n,lx-1);
386 R(Slice(twok,2*twok-1)) = get_rnlp(n,lx);
387
388 R.scale(pow(0.5,0.5*n));
389 R = inner(c,R);
390 if (do_transpose) R = transpose(R);
391 rnlij_cache.set(n,lx,R);
392 return *rnlij_cache.getptr(n,lx);
393 };

778 /// Compute the projection of the operator onto the double order polynomials
779
780 /// The returned reference is to a cached tensor ... if you want to
781 /// modify it, take a copy first.
782 ///
783 /// Return in \c v[p] \c p=0..2*k-1
784 /// \code
785 /// r(n,l,p) = 2^(-n) * int(K(2^(-n)*(z+l)) * phi(p,z), z=0..1)
786 /// \endcode
787 /// The kernel is coeff*exp(-expnt*z^2)*z^m (with m>0). This is equivalent to
788 /// \code
789 /// r(n,l,p) = 2^(-n*(m+1))*coeff * int( ((d/dz)^m exp(-beta*z^2)) * phi(p,z-l), z=l..l+1)
790 /// \endcode
791 /// where
792 /// \code
793 /// beta = alpha * 2^(-2*n)
794 /// \endcode
795 Tensor<Q> rnlp(Level n, const Translation lx) const final {
796 int twok = 2*this->k;
797 Tensor<Q> v(twok); // Can optimize this away by passing in
798 //KahanAccumulator<Q> v_accumulator[twok];
799 std::vector<KahanAccumulator<Q>> v_accumulator(twok);
800 constexpr bool use_kahan = false; // change to true to use Kahan accumulator
801
802 // integration range is [0,1] ...
803 std::pair<double, double> integration_limits{0,1};
804 // ... unless using hard (step-wise) range restriction
805 if (this->range.finite_hard()) {
806 const auto two_to_nm1 = (1ul << n) * 0.5;
807 if (lx < 0) {
808 integration_limits = std::make_pair(
809 std::min(std::max(-two_to_nm1 * this->range.iextent_x2() - lx, 0.), 1.), 1.);
810 } else {
811 integration_limits = std::make_pair(
812 0., std::max(std::min(two_to_nm1 * this->range.iextent_x2() - lx, 1.), 0.));
813 }
814 // early return if empty integration range (this indicates that
815 // the range restriction makes the kernel zero everywhere in the box)
816 if (integration_limits.first == integration_limits.second) {
817 MADNESS_ASSERT(this->rnlp_is_zero(n, lx));
818 return v;
819 }
820 else {
821 MADNESS_ASSERT(!this->rnlp_is_zero(n, lx));
822 }
823 }
824 // integration range lower bound, upper bound, length
825 const auto x0 = integration_limits.first;
826 const auto x1 = integration_limits.second;
827 const auto L = x1 - x0;
828
829 /* Apply high-order Gauss Legendre onto subintervals
830
831 coeff*int(exp(-beta(x+l)**2) * z^m * phi[p](x),x=0..1);
832
833 The translations internally considered are all +ve, so
834 significant pieces will be on the left. Finish after things
835 become insignificant.
836
837 The resulting coefficients are accurate to about 1e-20.
838 */
839
840 // Rescale expnt & coeff onto level n so integration range
841 // is [l,l+1]
842 Q scaledcoeff = coeff*pow(0.5,0.5*n*(2*m+1));
843
844 // Subdivide interval into nbox boxes of length h
845 // ... estimate appropriate size from the exponent. A
846 // Gaussian with real-part of the exponent beta falls in
847 // magnitude by a factor of 1/e at x=1/sqrt(beta), and by
848 // a factor of e^-49 ~ 5e-22 at x=7/sqrt(beta) (and with
849 // polyn of z^2 it is 1e-20). So, if we use a box of size
850 // 1/sqrt(beta) we will need at most 7 boxes. Incorporate
851 // the coefficient into the screening since it may be
852 // large. We can represent exp(-x^2) over [l,l+1] with a
853 // polynomial of order 21 to a relative
854 // precision of better than machine precision for
855 // l=0,1,2,3 and for l>3 the absolute error is less than
856 // 1e-23. We want to compute matrix elements with
857 // polynomials of order 2*k-1+m, so the total order is
858 // 2*k+20+m, which can be integrated with a quadrature rule
859 // of npt=k+11+(m+1)/2. npt is set in the constructor.
860
861 double fourn = std::pow(4.0,double(n));
862 double beta = expnt * pow(0.25,double(n));
863 double h = 1.0/sqrt(beta); // 2.0*sqrt(0.5/beta);
864 long nbox = long(1.0/h);
865 if (nbox < 1) nbox = 1;
866
867 // corner case: soft range restriction, range boundary within in interval
868 // since the integrand changes rapidly at x=0, 1/2, or 1, need finer
869 // integration
870 // N.B. this should have almost no impact on performance since this
871 // will produce a large number of boxes only when 2^n * sigma << 1
872 if (this->range.finite_soft()) {
873 // range boundary within in interval if 2^{n-1}*range.N() \in [l,l+1] (l>=0) or [-l,-l+1] (l<0)
874 const auto range_edge_in_interval = [&]() {
875 const auto two_to_nm1 = (1ul << n) * 0.5;
876 const auto range_boundary = two_to_nm1 * this->range.N();
877 if (lx >= 0) {
878 return range_boundary >= lx && range_boundary <= lx+1;
879 }
880 else { // lx < 0
881 return range_boundary <= -lx && range_boundary >= -lx-1;
882 }
883 };
884
885 // ensure that box is at least 1/sigma in size
886 if (range_edge_in_interval()) {
887 nbox = std::max(nbox, static_cast<long>(ceil(1./((1ul << n) * this->range.sigma()))));
888 }
889 }
890
891 h = L/nbox;
892
893 // Find argmax such that h*scaledcoeff*exp(-argmax)=1e-22 ... if
894 // beta*xlo*xlo is already greater than argmax we can neglect this
895 // and subsequent boxes.
896
897 // The derivatives add a factor of expnt^m to the size of
898 // the function at long range.
899 double sch = std::abs(scaledcoeff*h);
900 if (m == 1) sch *= expnt;
901 else if (m == 2) sch *= expnt*expnt;
902 double argmax = std::abs(log(1e-22/sch)); // perhaps should be -log(1e-22/sch) ?
903
904 // to screen need to iterate over boxes in the order of decreasing kernel values
905 const bool left_to_right = lx >= 0;
906 // if going left-to-right, start at left, else at right
907 const double xstartedge = left_to_right ? x0+lx : lx + 1;
908
909 // with oscillatory integrands the heuristic for reducing roundoff
910 // is to sum from large to small, i.e. proceed in same direction as the order of boxes
911 // WARNING: the grid points in quad_{x,w} are in order of decreasing x!
912 // hence decrement grid point indices for left_to_right, increment otherwise
913 const long first_pt = left_to_right ? this->npt-1: 0;
914 const long sentinel_pt = left_to_right ? -1 : this->npt;
915 const auto next_pt = [left_to_right](auto i) { return left_to_right ? i-1 : i+1; };
916
917 double xlo = left_to_right ? xstartedge : xstartedge-h;
918 double xhi;
919 for (long box=0; box!=nbox; ++box, xlo = (left_to_right ? xhi : xlo-h)) {
920
921 // can ignore this and rest of boxes if the Gaussian has decayed enough at the side of the box closest to the origin
922 xhi=xlo+h;
923 const auto xabs_min = std::min(std::abs(xhi),std::abs(xlo));
924 if (beta*xabs_min*xabs_min > argmax) break;
925
926 for (long i=first_pt; i!=sentinel_pt; i=next_pt(i)) {
927#ifdef IBMXLC
928 double phix[80];
929#else
930 double phix[2*MAXK];
931#endif
932 double xx = xlo + h*this->quad_x(i);
933 Q ee = scaledcoeff*exp(-beta*xx*xx)*this->quad_w(i)*h;
934 if (this->range && this->range.finite()) {
935 const auto x = xx * pow(0.5,double(n));
936 ee *= this->range.value(x);
937 }
938
939 // Differentiate as necessary
940 if (m == 1) {
941 ee *= -2.0*expnt*xx;
942 }
943 else if (m == 2) {
944 ee *= (4.0*xx*xx*expnt*expnt - 2.0*expnt*fourn);
945 }
946
947 legendre_scaling_functions(xx-lx,twok,phix);
948 for (long p=0; p<twok; ++p) {
949 if constexpr (use_kahan)
950 v_accumulator[p] += ee * phix[p];
951 else
952 v(p) += ee * phix[p];
953 }
954 }
955 }
956
957 if constexpr (use_kahan) {
958 for (long p = 0; p < twok; ++p)
959 v(p) = static_cast<Q>(v_accumulator[p]);
960 }
961
962 return v;
963 }
964
965 /// @return true if the block of [r^n_l]_ij is expected to be small
966 bool issmall(Level n, Translation lx) const final {
967 // [r^n_l]_ij = superposition of [r^n_l]_p and [r^n_l-1]_ij
968 // lx>0? the nearest box is lx-1 -> the edge closest to the origin is lx - 1
969 // lx<0? the nearest box is lx -> the edge closest to the origin is lx + 1
970 // lx==0? interactions within same box are never small
971 if (lx == 0)
972 return false;
973 else {
974 const double beta = expnt * pow(0.25,double(n));
975 const double overly_large_beta_r2 = 49.0; // 49 -> 5e-22 69 -> 1e-30
976 if (lx > 0) {
977 const auto ll = lx - 1;
978 return beta * ll * ll > overly_large_beta_r2;
979 } else {
980 const auto ll = lx + 1;
981 return beta * ll * ll > overly_large_beta_r2;
982 }
983 }
984 };
985 };
