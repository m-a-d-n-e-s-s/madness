63 /// SVD - MATLAB syntax
64
65 /// call as
66 /// auto [U,s,VT] = svd(A);
67 /// with A=U*S*VT
68 template <typename T>
69 std::tuple<Tensor<T>, Tensor< typename Tensor<T>::scalar_type >, Tensor<T>>
70 svd(const Tensor<T>& A) {
71 Tensor<T> U,VT;
72 Tensor< typename Tensor<T>::scalar_type > s;
73 svd(A,U,s,VT);
