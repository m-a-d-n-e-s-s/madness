39 struct MemKey {
40 unsigned long world_id=1;
41 unsigned long rank=0;
42 std::string hostname="localhost";
43 std::size_t DIM=0;
44 MemKey() = default;
45 MemKey(World& world) : world_id(world.id()), rank(world.rank()) {
46 hostname=MemoryMeasurer::get_hostname();
47 }
48
49 template<typename T, std::size_t NDIM>
50 MemKey(const FunctionImpl<T,NDIM>& fimpl) : MemKey(fimpl.world) {
51 DIM=NDIM;
52 }
53 MemKey(const MemKey& other) = default;
54
55 template<typename Archive>
56 void serialize(Archive& ar) const {
57 ar & world_id & rank & hostname & DIM;
58 }
59 };
60 friend bool operator<(const MemKey& lhs, const MemKey& other) {
61 if (lhs.hostname!=other.hostname) return lhs.hostname<other.hostname;
62 if (lhs.world_id!=other.world_id) return lhs.world_id<other.world_id;
63 if (lhs.rank!=other.rank) return lhs.rank<other.rank;
64 return lhs.DIM<other.DIM;
65 }

80 template<typename T, std::size_t NDIM>
81 const FunctionImpl<T,NDIM>* cast_to_funcimpl_ptr(const uniqueidT obj_id) {
82 World& world=*World::world_from_id(obj_id.get_world_id());
83 auto ptr_opt = world.ptr_from_id< WorldObject< FunctionImpl<T,NDIM> > >(obj_id);

85 MADNESS_EXCEPTION("FunctionImpl: remote operation attempting to use a locally uninitialized object",0);

89 /// keeps track of the memory usage of all objects of one or many worlds **on this rank**
90 MemInfoMapT world_memory_map;
91 bool debug=false;
92
93 template<typename T, std::size_t NDIM>
94 void add_memory_to_map(const FunctionImpl<T,NDIM>& f) {
95 const double toGB=double(sizeof(T))/(1024*1024*1024); // convert to GB
96 auto sz=f.nCoeff_local();
97 if (debug) print("funcimpl<T,",NDIM,"> id",f.id(), "rank",f.world.rank(),"size in GB",sz*toGB);
98
99 // accumulate the sizes into the world_memory_map
100 world_memory_map[MemKey(f)].num_functions++;
101 world_memory_map[MemKey(f)].memory_GB+=sz*toGB;
102 }

106 /// add all FunctionImpl<T,NDIM> objects of the given world to the memory map
107 /// the memory map is a rank-local object
108 void search_world(World& world) {
109
110 auto all_objects=world.get_object_ids();
111 if (debug and (world.rank()==0)) print("objects in this world ",all_objects);
112
113 for (const auto& obj : all_objects) {
114 if (auto funcimpl=cast_to_funcimpl_ptr<double,1>(obj)) add_memory_to_map(*funcimpl);
115 if (auto funcimpl=cast_to_funcimpl_ptr<double,2>(obj)) add_memory_to_map(*funcimpl);
116 if (auto funcimpl=cast_to_funcimpl_ptr<double,3>(obj)) add_memory_to_map(*funcimpl);
117 if (auto funcimpl=cast_to_funcimpl_ptr<double,4>(obj)) add_memory_to_map(*funcimpl);
118 if (auto funcimpl=cast_to_funcimpl_ptr<double,5>(obj)) add_memory_to_map(*funcimpl);
119 if (auto funcimpl=cast_to_funcimpl_ptr<double,6>(obj)) add_memory_to_map(*funcimpl);
120 }
121 }

123 /// add all FunctionImpl<T,NDIM> objects **of all worlds** to the memory map
124 /// the memory map is a rank-local object
125 void search_all_worlds() {
126 auto all_worlds=World::get_world_ids(); // all worlds but the default world
127 all_worlds.push_back(World::get_default().id()); // add the default world
128 if (debug) print("searching worlds",all_worlds);
129 for (auto world_id : all_worlds) {
130 if (debug) print("searching world",world_id);
131 World* thisworld=World::world_from_id(world_id);
132 if (World::exists(thisworld)) search_world(*thisworld);
133 }
134 }

141 /// gather all information of the map on rank 0 of the universe
142 void reduce_map(World& universe) {
143 // turn map into vector
144 std::vector<std::pair<MemKey,MemInfo>> memory_vec(world_memory_map.begin(),world_memory_map.end());
145 // gather all data on rank 0
146 memory_vec=universe.gop.concat0(memory_vec);
147 // turn back into map
148 clear_map();
149 for (const auto& [memkey,memval] : memory_vec) {
150 world_memory_map[memkey]=memval;
151 }
152 }

155 /// given the hostname, return number of ranks and total rss on that node
156 static std::map<std::string,std::pair<int,double>> host_to_nrank_and_rss_map(World& universe) {
157 auto accumulate_left =[](std::pair<int,double>& a, const std::pair<int,double>& b) {
158 a.first++;
159 a.second+=b.second;
160 };
161 auto rank_to_host=rank_to_host_and_rss_map(universe);
162 std::map<std::string,std::pair<int,double>> host_to_rank;
163 for (const auto& [rank,hostname_and_rss] : rank_to_host) {
164 accumulate_left(host_to_rank[hostname_and_rss.first],std::pair<int,double>(rank,hostname_and_rss.second));
165 }
166 return host_to_rank;
167 }

170 /// accumulate the memory usage of all objects of all worlds for this rank per host
171
172 /// integrate out world and dim from MemKey, result lives on rank 0 only
173 std::vector<std::pair<std::pair<std::string,long>,double>> memory_per_host_and_rank(World& world) const {
174
175 std::map<std::pair<std::string,long>,double> memory_per_host;
176 for (const auto& [memkey,memval] : world_memory_map) {
177 memory_per_host[{memkey.hostname,memkey.rank}]+=memval.memory_GB;
178 }
179
180 // turn map into vector and sort
181 std::vector<std::pair<std::pair<std::string,long>,double>> memory_per_host_vec(memory_per_host.begin(),memory_per_host.end());
182 std::sort(memory_per_host_vec.begin(),memory_per_host_vec.end(),[](const auto& a, const auto& b){return a.first<b.first;});
183
184 return memory_per_host_vec;
185 }

187 /// accumulate the memory usage of all objects of all worlds over all ranks per host
188
189 /// integrate out world, dim and rank, only hostname is left
190 std::vector<std::pair<std::string,double>> memory_per_host_all_ranks(
191 const std::vector<std::pair<std::pair<std::string,long>,double>>& mem_per_host_and_rank) const {
192 std::map<std::string,double> mem_per_host;
193 for (auto& [hostname_and_rank,memory] : mem_per_host_and_rank) {
194 auto hostname=hostname_and_rank.first;
195 mem_per_host[hostname]+=memory;
196 }
197 // turn map into vector
198 std::vector<std::pair<std::string,double>> mem_per_host_vec(mem_per_host.begin(),mem_per_host.end());
199 return mem_per_host_vec;
200 }

202 /// return the total memory usage over all hosts
203 double total_memory(World& world) const {
204 double total_memory=0.0;
205 for (const auto& [memkey,memval] : world_memory_map) {
206 total_memory+=memval.memory_GB;

211 /// @param[in] msg a message to print before the memory map
212 /// @param[in] world used only for clean printing
213 void print_memory_map(World& world, std::string msg="") {
214 reduce_map(world);
215 world.gop.fence();
216 if (world.rank()==0) {
217 print("final memory map:",msg);
218 print("hostname world rank DIM #funcs memory_GB");
219 }
220 constexpr std::size_t bufsize=256;
221 char line[bufsize];
222
223 // print all information
224 world.gop.fence();
225 // turn into vector
226 std::vector<std::pair<MemKey,MemInfo>> memory_vec(world_memory_map.begin(),world_memory_map.end());
227 std::sort(memory_vec.begin(),memory_vec.end(),[](const std::pair<MemKey,MemInfo>& a, const std::pair<MemKey,MemInfo>& b){return a.first<b.first;});
228 for (const auto& [memkey,memval] : memory_vec) {
229 snprintf(line, bufsize, "%20s %12lu %5lu %5lu %5lu %e", memkey.hostname.c_str(), memkey.world_id, memkey.rank, memkey.DIM, memval.num_functions, memval.memory_GB);
230 print(std::string(line));
231 }
232 world.gop.fence();
233
234 // print memory on each host
235 auto mem_per_host_and_rank=memory_per_host_and_rank(world);
236 auto host_to_nrank_and_rss=host_to_nrank_and_rss_map(world);
237 if (world.rank()==0) {
238 print("memory per host");
239 auto info=memory_per_host_all_ranks(mem_per_host_and_rank);
240 print("hostname memory_GB nrank(universe) rss_GB/host");
241 // print("hostname memory_GB");
242 for (const auto& [hostname,memory] : info) {
243 snprintf(line, bufsize, "%20s %e %d %e", hostname.c_str(), memory,
244 host_to_nrank_and_rss[hostname].first, host_to_nrank_and_rss[hostname].second);
245 print(std::string(line));
246 }
247 }
248 if (world.rank()==0) {
249 auto info=memory_per_host_all_ranks(mem_per_host_and_rank);
250 double total_mem=total_memory(world);
251 double total_rss=0.0;
252 for (auto& [hostname,memory] : info) {
253 total_rss+=host_to_nrank_and_rss[hostname].second;
254 }
255 std::string word="all hosts";
256 snprintf(line, bufsize, "%20s %e %d %e",
257 word.c_str(), total_mem, world.size(), total_rss);
258 print(std::string(line));
259 }

261 }
